14/// Run Ifá-Lang code and return the output
15#[wasm_bindgen]
16pub fn run_code(source: &str) -> String {
17    // Initialize interpreter
18    let mut interpreter = Interpreter::new();
19
20    // Parse source code
21    match parse(source) {
22        Ok(program) => {
23            // Execute program
24            match interpreter.execute(&program) {
25                Ok(_) => {
26                    // Get text output
27                    let mut output = interpreter.get_output().join("\n");
28
29                    // Check if canvas has content (non-blank)
30                    let canvas_output = interpreter.get_canvas();
31                    if !canvas_output.chars().all(|c| c == ' ' || c == '\n') {
32                        if !output.is_empty() {
33                            output.push_str("\n\n");
34                        }
35                        output.push_str("═══ Canvas Output ═══\n");
36                        output.push_str(&canvas_output);
37                    }
38
39                    output
40                }
41                Err(e) => {
42                    format!("Runtime Error: {}", e)
43                }
44            }
45        }
46        Err(e) => {
47            format!("Parse Error: {}", e)
48        }
49    }
50}

58/// Cast the Opele and return an Odu name
59#[wasm_bindgen]
60pub fn cast_opele() -> String {
61    let seed = js_sys::Date::now() as u64;
62    let random = seed
63        .wrapping_mul(6364136223846793005)
64        .wrapping_add(1442695040888963407);
65    let odu_byte = ((random >> 32) as u8) % 16;
66
67    let names = [
68        "Ogbe", "Oyeku", "Iwori", "Odi", "Irosu", "Owonrin", "Obara", "Okanran", "Ogunda", "Osa",
69        "Ika", "Oturupon", "Otura", "Irete", "Ose", "Ofun",
70    ];
71
72    names[odu_byte as usize].to_string()
73}
