12/// Context for linting - tracks state as we walk the AST
13#[derive(Debug)]
14pub struct LintContext {
15    /// Variables that have been defined
16    pub defined_vars: HashSet<String>,
17    /// Variables that have been used
18    pub used_vars: HashSet<String>,
19    /// Imports
20    pub imports: HashSet<String>,
21    /// Current function name (if inside one)
22    pub current_function: Option<String>,
23    /// Whether we've seen a return in current function
24    pub has_return: bool,
25    /// Resource lifecycle tracking (open -> close)
26    pub open_resources: HashMap<String, (usize, usize)>, // resource -> (line, col)
27    /// Ìwà Engine - resource lifecycle validation
28    pub iwa_engine: IwaEngine,
29    /// Èèwọ̀ Enforcer - architectural constraints
30    pub taboo_enforcer: TabooEnforcer,
31}
32
33impl Default for LintContext {
34    fn default() -> Self {
35        Self::new()
36    }
37}
38
39impl LintContext {
40    pub fn new() -> Self {
41        Self {
42            defined_vars: HashSet::new(),
43            used_vars: HashSet::new(),
44            imports: HashSet::new(),
45            current_function: None,
46            has_return: false,
47            open_resources: HashMap::new(),
48            iwa_engine: IwaEngine::new(true),
49            taboo_enforcer: TabooEnforcer::new(),
50        }
51    }
52
53    pub fn define_var(&mut self, name: &str) {
54        self.defined_vars.insert(name.to_string());
55    }
56
57    pub fn use_var(&mut self, name: &str) {
58        self.used_vars.insert(name.to_string());
59    }
60
61    pub fn enter_function(&mut self, name: &str) {
62        self.current_function = Some(name.to_string());
63        self.has_return = false;
64    }
65
66    pub fn exit_function(&mut self) {
67        self.current_function = None;
68        self.has_return = false;
69    }
70}

72/// Check a program and return diagnostics
73pub fn check_program(program: &Program, filename: &str) -> Babalawo {
74    let mut babalawo = Babalawo::new();
75    let mut ctx = LintContext::new();
76
77    // First pass: collect definitions
78    for stmt in &program.statements {
79        collect_definitions(stmt, &mut ctx);
80    }
81
82    // Second pass: check for issues (including Ìwà and Èèwọ̀)
83    for stmt in &program.statements {
84        check_statement(stmt, &mut ctx, &mut babalawo, filename);
85    }
86
87    // Final checks
88    check_unused_vars(&ctx, &mut babalawo, filename);
89    check_unclosed_resources(&ctx, &mut babalawo, filename);
90
91    // Ìwà Engine: check resource balance
92    if !ctx.iwa_engine.check_balance() {
93        for debt in ctx.iwa_engine.unclosed_resources() {
94            babalawo.error(
95                "UNCLOSED_RESOURCE",
96                &format!(
97                    "Resource '{}' opened at line {} was never closed (needs '{}')",
98                    debt.opener, debt.line, debt.required
99                ),
100                filename,
101                debt.line,
102                debt.column,
103            );
104        }
105    }
106
107    // Èèwọ̀ Enforcer: check taboo violations
108    if !ctx.taboo_enforcer.is_clean() {
109        for v in ctx.taboo_enforcer.get_violations() {
110            babalawo.error(
111                "TABOO_VIOLATION",
112                &format!(
113                    "Forbidden dependency: '{}' cannot call '{}'",
114                    v.caller, v.callee
115                ),
116                filename,
117                v.line,
118                v.column,
119            );
120        }
121    }
122
123    babalawo
124}

126/// Check a program with custom taboos
127#[allow(dead_code)]
128pub fn check_program_with_taboos(
129    program: &Program,
130    filename: &str,
131    taboos: Vec<(&str, &str)>, // (source, target) forbidden pairs
132) -> Babalawo {
133    let mut babalawo = Babalawo::new();
134    let mut ctx = LintContext::new();
135
136    // Register taboos
137    for (source, target) in taboos {
138        ctx.taboo_enforcer.add_taboo(source, "", target, "", false);
139    }
140
141    // First pass: collect definitions
142    for stmt in &program.statements {
143        collect_definitions(stmt, &mut ctx);
144    }
145
146    // Second pass: check for issues
147    for stmt in &program.statements {
148        check_statement(stmt, &mut ctx, &mut babalawo, filename);
149    }
150
151    // Final checks
152    check_unused_vars(&ctx, &mut babalawo, filename);
153    check_iwa_balance(&mut ctx, &mut babalawo, filename);
154    check_taboo_violations(&ctx, &mut babalawo, filename);
155
156    babalawo
157}

159/// Check Ìwà Engine balance
160#[allow(dead_code)]
161fn check_iwa_balance(ctx: &mut LintContext, baba: &mut Babalawo, file: &str) {
162    if !ctx.iwa_engine.check_balance() {
163        for debt in ctx.iwa_engine.unclosed_resources() {
164            baba.error(
165                "UNCLOSED_RESOURCE",
166                &format!(
167                    "Resource '{}' was never closed (needs '{}')",
168                    debt.opener, debt.required
169                ),
170                file,
171                debt.line,
172                debt.column,
173            );
174        }
175    }
176}

192/// Collect variable and function definitions + Taboos and Opon directives
193fn collect_definitions(stmt: &Statement, ctx: &mut LintContext) {
194    match stmt {
195        Statement::VarDecl { name, .. } => {
196            ctx.define_var(name);
197        }
198        Statement::EseDef {
199            name, params, body, ..
200        } => {
201            ctx.define_var(name);
202            // Parameters are also definitions within the function
203            for param in params {
204                ctx.define_var(&param.name);
205            }
206            for s in body {
207                collect_definitions(s, ctx);
208            }
209        }
210        Statement::OduDef { name, body, .. } => {
211            ctx.define_var(name);
212            for s in body {
213                collect_definitions(s, ctx);
214            }
215        }
216        Statement::For { var, body, .. } => {
217            ctx.define_var(var);
218            for s in body {
219                collect_definitions(s, ctx);
220            }
221        }
222        Statement::If {
223            then_body,
224            else_body,
225            ..
226        } => {
227            for s in then_body {
228                collect_definitions(s, ctx);
229            }
230            if let Some(else_stmts) = else_body {
231                for s in else_stmts {
232                    collect_definitions(s, ctx);
233                }
234            }
235        }
236        Statement::While { body, .. } => {
237            for s in body {
238                collect_definitions(s, ctx);
239            }
240        }
241        // Register taboo declarations for enforcement
242        Statement::Taboo { source, target, .. } => {
243            ctx.taboo_enforcer.add_taboo(source, "", target, "", false);
244        }
245        // Opon directives - currently informational only
246        Statement::Opon { size, .. } => {
247            // Could store opon size in context for memory limit checks
248            let _ = size;
249        }
250        _ => {}
251    }
252}

254/// Check a statement for issues
255fn check_statement(stmt: &Statement, ctx: &mut LintContext, baba: &mut Babalawo, file: &str) {
256    match stmt {
257        Statement::VarDecl {
258            name, value, span, ..
259        } => {
260            check_expression(value, ctx, baba, file);
261
262            // Check for self-referencing initialization
263            if expression_uses_var(value, name) {
264                baba.error(
265                    "UNINITIALIZED",
266                    &format!("Variable '{}' used in its own initialization", name),
267                    file,
268                    span.line,
269                    span.column,
270                );
271            }
272        }
273
274        Statement::Assignment {
275            target,
276            value,
277            span,
278        } => {
279            check_expression(value, ctx, baba, file);
280
281            // Check if target variable is defined
282            if let ifa_core::ast::AssignTarget::Variable(name) = target {
283                if !ctx.defined_vars.contains(name) {
284                    baba.error(
285                        "UNDEFINED_VARIABLE",
286                        &format!("Variable '{}' assigned before declaration", name),
287                        file,
288                        span.line,
289                        span.column,
290                    );
291                }
292            }
293        }
294
295        Statement::Instruction { call, span } => {
296            // Check for division by zero
297            if call.method == "pin" || call.method == "div" {
298                if let Some(Expression::Int(0)) = call.args.get(1) {
299                    baba.error(
300                        "DIVISION_BY_ZERO",
301                        "Division by zero detected",
302                        file,
303                        span.line,
304                        span.column,
305                    );
306                }
307            }
308
309            // Track resource lifecycle
310            let domain = format!("{:?}", call.domain).to_lowercase();
311            if call.method == "si" || call.method == "open" {
312                ctx.open_resources.insert(
313                    format!("{}:{}", domain, span.line),
314                    (span.line, span.column),
315                );
316            }
317            if call.method == "pa" || call.method == "close" {
318                ctx.open_resources
319                    .remove(&format!("{}:{}", domain, span.line));
320            }
321
322            // Check taboo violations - get current context (caller) from function or "global"
323            let caller = ctx
324                .current_function
325                .clone()
326                .unwrap_or_else(|| "global".to_string());
327            let callee = format!("{:?}", call.domain).to_lowercase();
328            ctx.taboo_enforcer
329                .check_call(&caller, &callee, span.line, span.column);
330
331            // Check arguments
332            for arg in &call.args {
333                check_expression(arg, ctx, baba, file);
334            }
335        }
336
337        Statement::EseDef {
338            name,
339            body,
340            span,
341            visibility: _,
342            ..
343        } => {
344            ctx.enter_function(name);
345
346            for s in body {
347                check_statement(s, ctx, baba, file);
348            }
349
350            // Check for missing return (only warn, not error)
351            if !ctx.has_return && !body.is_empty() {
352                // Only warn if function seems to return something
353                if function_should_return(body) {
354                    baba.warning(
355                        "MISSING_RETURN",
356                        &format!("Function '{}' may not return on all paths", name),
357                        file,
358                        span.line,
359                        span.column,
360                    );
361                }
362            }
363
364            ctx.exit_function();
365        }
366
367        Statement::OduDef { body, .. } => {
368            for s in body {
369                check_statement(s, ctx, baba, file);
370            }
371        }
372
373        Statement::If {
374            condition,
375            then_body,
376            else_body,
377            ..
378        } => {
379            check_expression(condition, ctx, baba, file);
380
381            for s in then_body {
382                check_statement(s, ctx, baba, file);
383            }
384
385            if let Some(else_stmts) = else_body {
386                for s in else_stmts {
387                    check_statement(s, ctx, baba, file);
388                }
389            }
390        }
391
392        Statement::While {
393            condition, body, ..
394        } => {
395            check_expression(condition, ctx, baba, file);
396
397            for s in body {
398                check_statement(s, ctx, baba, file);
399            }
400        }
401
402        Statement::For {
403            var,
404            iterable,
405            body,
406            span: _,
407        } => {
408            check_expression(iterable, ctx, baba, file);
409            ctx.use_var(var);
410
411            for s in body {
412                check_statement(s, ctx, baba, file);
413            }
414        }
415
416        Statement::Return { value, .. } => {
417            ctx.has_return = true;
418            if let Some(v) = value {
419                check_expression(v, ctx, baba, file);
420            }
421        }
422
423        _ => {}
424    }
425}

427/// Check an expression for issues
428fn check_expression(expr: &Expression, ctx: &mut LintContext, baba: &mut Babalawo, file: &str) {
429    match expr {
430        Expression::Identifier(name) => {
431            ctx.use_var(name);
432
433            // Check if variable is defined
434            if !ctx.defined_vars.contains(name) && !is_builtin(name) {
435                baba.error(
436                    "UNDEFINED_VARIABLE",
437                    &format!("Variable '{}' used before declaration", name),
438                    file,
439                    1, // TODO: get proper line from expression
440                    1,
441                );
442            }
443        }
444
445        Expression::BinaryOp {
446            left, right, op, ..
447        } => {
448            check_expression(left, ctx, baba, file);
449            check_expression(right, ctx, baba, file);
450
451            // Check for division by zero in binary op
452            if matches!(
453                op,
454                ifa_core::ast::BinaryOperator::Div | ifa_core::ast::BinaryOperator::Mod
455            ) {
456                if let Expression::Int(0) = **right {
457                    baba.error(
458                        "DIVISION_BY_ZERO",
459                        "Division by zero in expression",
460                        file,
461                        1,
462                        1,
463                    );
464                }
465            }
466        }
467
468        Expression::List(items) => {
469            for item in items {
470                check_expression(item, ctx, baba, file);
471            }
472        }
473
474        Expression::Map(entries) => {
475            for (k, v) in entries {
476                check_expression(k, ctx, baba, file);
477                check_expression(v, ctx, baba, file);
478            }
479        }
480
481        Expression::Index { object, index } => {
482            check_expression(object, ctx, baba, file);
483            check_expression(index, ctx, baba, file);
484        }
485
486        Expression::MethodCall { object, args, .. } => {
487            check_expression(object, ctx, baba, file);
488            for arg in args {
489                check_expression(arg, ctx, baba, file);
490            }
491        }
492
493        Expression::OduCall(call) => {
494            for arg in &call.args {
495                check_expression(arg, ctx, baba, file);
496            }
497        }
498
499        _ => {}
500    }
501}

503/// Check for unused variables
504fn check_unused_vars(ctx: &LintContext, baba: &mut Babalawo, file: &str) {
505    for var in &ctx.defined_vars {
506        if !ctx.used_vars.contains(var) && !var.starts_with('_') {
507            baba.warning(
508                "UNUSED_VARIABLE",
509                &format!("Variable '{}' is defined but never used", var),
510                file,
511                1,
512                1,
513            );
514        }
515    }
516}

518/// Check for unclosed resources
519fn check_unclosed_resources(ctx: &LintContext, baba: &mut Babalawo, file: &str) {
520    for (resource, (line, col)) in &ctx.open_resources {
521        baba.warning(
522            "UNCLOSED_RESOURCE",
523            &format!("Resource '{}' opened but never closed", resource),
524            file,
525            *line,
526            *col,
527        );
528    }
529}

531/// Check if an expression uses a variable
532fn expression_uses_var(expr: &Expression, var_name: &str) -> bool {
533    match expr {
534        Expression::Identifier(name) => name == var_name,
535        Expression::BinaryOp { left, right, .. } => {
536            expression_uses_var(left, var_name) || expression_uses_var(right, var_name)
537        }
538        Expression::List(items) => items.iter().any(|i| expression_uses_var(i, var_name)),
539        Expression::Index { object, index } => {
540            expression_uses_var(object, var_name) || expression_uses_var(index, var_name)
541        }
542        _ => false,
543    }
544}

546/// Check if a function body suggests it should return something
547fn function_should_return(body: &[Statement]) -> bool {
548    // Simple heuristic: if there's any expression statement, it probably should return
549    for stmt in body {
550        if matches!(stmt, Statement::Return { value: Some(_), .. }) {
551            return true;
552        }
553    }
554    false
555}

557/// Check if a name is a builtin
558fn is_builtin(name: &str) -> bool {
559    matches!(
560        name,
561        "true" | "false" | "nil" | "otito" | "iro" | "ohunkohun"
562    )
563}
