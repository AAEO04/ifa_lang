9/// Severity level for diagnostics
10#[derive(Debug, Clone, Copy, PartialEq, Eq)]
11pub enum Severity {
12    Error,   // Aṣiṣe - must fix
13    Warning, // Ìkìlọ̀ - should fix
14    Info,    // Ìmọ̀ràn - suggestion
15    Style,   // Style recommendation
16}
17
18impl fmt::Display for Severity {
19    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
20        match self {
21            Severity::Error => write!(f, "error"),
22            Severity::Warning => write!(f, "warning"),
23            Severity::Info => write!(f, "info"),
24            Severity::Style => write!(f, "style"),
25        }
26    }
27}

29/// An Ifá error with Babalawo-style messaging
30#[derive(Debug, Clone)]
31pub struct IfaError {
32    pub code: String,
33    pub message: String,
34    pub file: String,
35    pub line: usize,

40impl IfaError {

45            file: file.to_string(),

58/// A diagnostic message with severity and wisdom
59#[derive(Debug, Clone)]
60pub struct Diagnostic {
61    pub severity: Severity,
62    pub error: IfaError,
63    pub odu: String,
64    pub wisdom: Option<String>,

67/// The Babalawo - Error diagnosis system
68pub struct Babalawo {
69    pub diagnostics: Vec<Diagnostic>,
70    pub verbose: bool,

73impl Default for Babalawo {

79impl Babalawo {
80    pub fn new() -> Self {
81        Self {
82            diagnostics: Vec::new(),
83            verbose: false,
84        }
85    }
86
87    pub fn verbose(mut self) -> Self {
88        self.verbose = true;
89        self
90    }

92    /// Add an error diagnostic
93    pub fn error(&mut self, code: &str, msg: &str, file: &str, line: usize, col: usize) {
94        self.add_diagnostic(Severity::Error, code, msg, file, line, col);
95    }
96
97    /// Add a warning diagnostic
98    pub fn warning(&mut self, code: &str, msg: &str, file: &str, line: usize, col: usize) {
99        self.add_diagnostic(Severity::Warning, code, msg, file, line, col);
100    }
101
102    /// Add an info diagnostic
103    pub fn info(&mut self, code: &str, msg: &str, file: &str, line: usize, col: usize) {
104        self.add_diagnostic(Severity::Info, code, msg, file, line, col);
105    }
106
107    fn add_diagnostic(
108        &mut self,
109        severity: Severity,
110        code: &str,
111        msg: &str,
112        file: &str,
113        line: usize,
114        col: usize,
115    ) {
116        let odu_key = ERROR_TO_ODU.get(code).copied().unwrap_or("OKANRAN");
117        let wisdom = ODU_WISDOM.get(odu_key);
118
119        let diagnostic = Diagnostic {
120            severity,
121            error: IfaError::new(code, msg, file, line, col),
122            odu: odu_key.to_string(),
123            wisdom: wisdom.map(|w| w.advice.to_string()),
124        };
125
126        self.diagnostics.push(diagnostic);
127    }
128
129    /// Get error count
130    pub fn error_count(&self) -> usize {
131        self.diagnostics
132            .iter()
133            .filter(|d| d.severity == Severity::Error)
134            .count()
135    }
136
137    /// Get warning count
138    pub fn warning_count(&self) -> usize {
139        self.diagnostics
140            .iter()
141            .filter(|d| d.severity == Severity::Warning)
142            .count()
143    }
144
145    /// Check if there are any errors
146    pub fn has_errors(&self) -> bool {
147        self.error_count() > 0
148    }

150    /// Format output (minimal by default)
151    pub fn format(&self) -> String {
152        let mut output = String::new();
153
154        if self.diagnostics.is_empty() {
155            return "No issues found. Àṣẹ!\n".to_string();
156        }
157
158        output.push_str("babalawo:\n\n");
159
160        for diag in &self.diagnostics {
161            let odu_name = ODU_WISDOM
162                .get(diag.odu.as_str())
163                .map(|w| w.name)
164                .unwrap_or("Ọ̀kànràn");
165
166            // Minimal format: error[Odu] file:line:col
167            output.push_str(&format!(
168                "{}[{}] {}:{}:{}\n",
169                diag.severity, odu_name, diag.error.file, diag.error.line, diag.error.column
170            ));
171
172            // Message
173            output.push_str(&format!("  {}\n", diag.error.message));
174
175            // Wisdom (only if verbose or it's an error)
176            if self.verbose || diag.severity == Severity::Error {
177                if let Some(wisdom) = &diag.wisdom {
178                    output.push_str(&format!("  Wisdom: {}\n", wisdom));
179                }
180            }
181
182            output.push('\n');
183        }
184
185        // Summary
186        let errors = self.error_count();
187        let warnings = self.warning_count();
188        output.push_str(&format!(
189            "{} error{}, {} warning{}. Àṣẹ!\n",
190            errors,
191            if errors == 1 { "" } else { "s" },
192            warnings,
193            if warnings == 1 { "" } else { "s" }
194        ));
195
196        output
197    }

199    /// Format as JSON for IDE integration
200    pub fn format_json(&self) -> String {
201        let items: Vec<serde_json::Value> = self
202            .diagnostics
203            .iter()
204            .map(|d| {
205                serde_json::json!({
206                    "severity": d.severity.to_string(),
207                    "code": d.error.code,
208                    "message": d.error.message,
209                    "file": d.error.file,
210                    "line": d.error.line,
211                    "column": d.error.column,
212                    "odu": d.odu,
213                })
214            })
215            .collect();
216
217        serde_json::to_string_pretty(&items).unwrap_or_default()
218    }
219
220    /// Format compact (one line per error)
221    pub fn format_compact(&self) -> String {
222        self.diagnostics
223            .iter()
224            .map(|d| {
225                format!(
226                    "{}:{}:{}: {}: {}\n",
227                    d.error.file, d.error.line, d.error.column, d.severity, d.error.message
228                )
229            })
230            .collect()
231    }
232}

244    #[test]
245    fn test_babalawo_error() {
246        let mut baba = Babalawo::new();
247        baba.error(
248            "UNDEFINED_VARIABLE",
249            "Variable 'x' not defined",
250            "test.ifa",
251            10,
252            5,
253        );
254
255        assert_eq!(baba.error_count(), 1);
256        assert!(baba.has_errors());
257
258        let output = baba.format();
259        assert!(output.contains("error[Ogbè]"));
260        assert!(output.contains("Variable 'x' not defined"));
261    }
262
263    #[test]
264    fn test_babalawo_warning() {
265        let mut baba = Babalawo::new();
266        baba.warning(
267            "UNUSED_VARIABLE",
268            "Variable 'temp' is unused",
269            "test.ifa",
270            5,
271            10,
272        );
273
274        assert_eq!(baba.warning_count(), 1);
275        assert!(!baba.has_errors());
276    }
