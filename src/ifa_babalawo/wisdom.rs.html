263/// Error code to Odù domain mapping
264pub static ERROR_TO_ODU: Lazy<HashMap<&'static str, &'static str>> = Lazy::new(|| {
265    let mut m = HashMap::new();
266
267    // Initialization errors → Ogbè
268    m.insert("UNINITIALIZED", "OGBE");
269    m.insert("NULL_REFERENCE", "OGBE");
270    m.insert("UNDEFINED_VARIABLE", "OGBE");
271
272    // Termination errors → Oyeku
273    m.insert("UNCLOSED_RESOURCE", "OYEKU");
274    m.insert("ORPHAN_PROCESS", "OYEKU");
275    m.insert("INCOMPLETE_SHUTDOWN", "OYEKU");
276
277    // Loop errors → Iwori
278    m.insert("INFINITE_LOOP", "IWORI");
279    m.insert("ITERATOR_EXHAUSTED", "IWORI");
280    m.insert("LOOP_INVARIANT_VIOLATED", "IWORI");
281
282    // File errors → Odi
283    m.insert("FILE_NOT_FOUND", "ODI");
284    m.insert("FILE_ALREADY_OPEN", "ODI");
285    m.insert("FILE_NOT_CLOSED", "ODI");
286    m.insert("PERMISSION_DENIED", "ODI");
287    m.insert("PRIVATE_ACCESS", "ODI");
288
289    // Output errors → Irosu
290    m.insert("FORMAT_ERROR", "IROSU");
291    m.insert("OUTPUT_OVERFLOW", "IROSU");
292
293    // Random errors → Owonrin
294    m.insert("SEED_ERROR", "OWONRIN");
295
296    // Math errors → Obara/Oturupon
297    m.insert("OVERFLOW", "OBARA");
298    m.insert("UNDERFLOW", "OTURUPON");
299    m.insert("DIVISION_BY_ZERO", "OTURUPON");
300    m.insert("ARITHMETIC_ERROR", "OBARA");
301
302    // Exception errors → Okanran
303    m.insert("UNHANDLED_EXCEPTION", "OKANRAN");
304    m.insert("ASSERTION_FAILED", "OKANRAN");
305    m.insert("UNUSED_VARIABLE", "OKANRAN");
306
307    // Array errors → Ogunda
308    m.insert("INDEX_OUT_OF_BOUNDS", "OGUNDA");
309    m.insert("ARRAY_EMPTY", "OGUNDA");
310
311    // Control flow errors → Osa
312    m.insert("UNREACHABLE_CODE", "OSA");
313    m.insert("INVALID_JUMP", "OSA");
314    m.insert("MISSING_RETURN", "OSA");
315
316    // String errors → Ika
317    m.insert("INVALID_ENCODING", "IKA");
318    m.insert("STRING_OVERFLOW", "IKA");
319
320    // Network errors → Otura
321    m.insert("CONNECTION_REFUSED", "OTURA");
322    m.insert("TIMEOUT", "OTURA");
323    m.insert("NETWORK_UNREACHABLE", "OTURA");
324
325    // Memory errors → Irete
326    m.insert("MEMORY_LEAK", "IRETE");
327    m.insert("DOUBLE_FREE", "IRETE");
328    m.insert("OUT_OF_MEMORY", "IRETE");
329
330    // Graphics errors → Ose
331    m.insert("INVALID_COORDINATES", "OSE");
332    m.insert("BUFFER_OVERFLOW", "OSE");
333
334    // Object errors → Ofun
335    m.insert("TYPE_ERROR", "OFUN");
336    m.insert("INHERITANCE_ERROR", "OFUN");
337    m.insert("OBJECT_NOT_FOUND", "OFUN");
338
339    m
340});

342#[cfg(test)]
343mod tests {
344    use super::*;
345
346    #[test]
347    fn test_odu_wisdom_lookup() {
348        let ogbe = ODU_WISDOM.get("OGBE").unwrap();
349        assert_eq!(ogbe.name, "Ogbè");
350        assert_eq!(ogbe.title, "The Light");
351        assert!(!ogbe.proverbs.is_empty());
352    }
353
354    #[test]
355    fn test_error_to_odu_mapping() {
356        assert_eq!(ERROR_TO_ODU.get("UNDEFINED_VARIABLE"), Some(&"OGBE"));
357        assert_eq!(ERROR_TO_ODU.get("DIVISION_BY_ZERO"), Some(&"OTURUPON"));
358        assert_eq!(ERROR_TO_ODU.get("FILE_NOT_FOUND"), Some(&"ODI"));
359    }
