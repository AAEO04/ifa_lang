1//! # Ifá FFI - Foreign Function Interface
2//!
3//! Cross-language bridge for calling external libraries from Ifá-Lang.
4//!
5//! Design philosophy (Linus-approved):
6//! - Simple types, clear mappings
7//! - No fancy abstractions - just get the job done
8//! - Security by default (whitelist, not blacklist)
9
10use std::collections::{HashMap, HashSet};
11use std::ffi::CString;
12use std::os::raw::{c_char, c_int, c_void};

14/// FFI Error type
15#[derive(Debug)]
16pub enum FfiError {
17    LibraryNotFound(String),
18    FunctionNotFound(String),
19    TypeMismatch(String),
20    SecurityViolation(String),
21    CallFailed(String),
22}
23
24impl std::fmt::Display for FfiError {
25    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
26        match self {
27            FfiError::LibraryNotFound(s) => write!(f, "Library not found: {}", s),
28            FfiError::FunctionNotFound(s) => write!(f, "Function not found: {}", s),
29            FfiError::TypeMismatch(s) => write!(f, "Type mismatch: {}", s),
30            FfiError::SecurityViolation(s) => write!(f, "Security violation: {}", s),
31            FfiError::CallFailed(s) => write!(f, "Call failed: {}", s),
32        }
33    }
34}
35
36impl std::error::Error for FfiError {}
37
38pub type FfiResult<T> = Result<T, FfiError>;

137/// Platform-agnostic library handle
138#[cfg(windows)]
139pub type LibHandle = *mut c_void;
140
141#[cfg(not(windows))]
142pub type LibHandle = *mut c_void;
143
144/// Load a shared library
145#[cfg(windows)]
146pub fn load_library(path: &str) -> FfiResult<LibHandle> {
147    let wide: Vec<u16> = path.encode_utf16().chain(std::iter::once(0)).collect();
148
149    // LoadLibraryW
150    extern "system" {
151        fn LoadLibraryW(lpFileName: *const u16) -> *mut c_void;
152    }
153
154    let handle = unsafe { LoadLibraryW(wide.as_ptr()) };
155    if handle.is_null() {
156        Err(FfiError::LibraryNotFound(path.to_string()))
157    } else {
158        Ok(handle)
159    }
160}
161
162#[cfg(unix)]
163pub fn load_library(path: &str) -> FfiResult<LibHandle> {
164    unsafe extern "C" {
165        fn dlopen(filename: *const c_char, flags: c_int) -> *mut c_void;
166    }
167
168    const RTLD_NOW: c_int = 2;
169
170    let c_path = CString::new(path).map_err(|_| FfiError::LibraryNotFound(path.to_string()))?;
171    let handle = unsafe { dlopen(c_path.as_ptr(), RTLD_NOW) };
172
173    if handle.is_null() {
174        Err(FfiError::LibraryNotFound(path.to_string()))
175    } else {
176        Ok(handle)
177    }
178}
179
180/// Get function pointer from library
181///
182/// # Safety
183/// The handle must be a valid library handle returned by load_library.
184#[cfg(windows)]
185pub unsafe fn get_proc(handle: LibHandle, name: &str) -> FfiResult<*mut c_void> {
186    unsafe extern "system" {
187        fn GetProcAddress(hModule: *mut c_void, lpProcName: *const c_char) -> *mut c_void;
188    }
189
190    let c_name = CString::new(name).map_err(|_| FfiError::FunctionNotFound(name.to_string()))?;
191    let proc = unsafe { GetProcAddress(handle, c_name.as_ptr()) };
192
193    if proc.is_null() {
194        Err(FfiError::FunctionNotFound(name.to_string()))
195    } else {
196        Ok(proc)
197    }
198}
199
200/// Get function pointer from library
201///
202/// # Safety
203/// The handle must be a valid library handle returned by load_library.
204#[cfg(unix)]
205pub unsafe fn get_proc(handle: LibHandle, name: &str) -> FfiResult<*mut c_void> {
206    unsafe extern "C" {
207        fn dlsym(handle: *mut c_void, symbol: *const c_char) -> *mut c_void;
208    }
209
210    let c_name = CString::new(name).map_err(|_| FfiError::FunctionNotFound(name.to_string()))?;
211    let proc = unsafe { dlsym(handle, c_name.as_ptr()) };
212
213    if proc.is_null() {
214        Err(FfiError::FunctionNotFound(name.to_string()))
215    } else {
216        Ok(proc)
217    }
218}

238/// The FFI Bridge - main interface for foreign calls
239pub struct IfaFfi {
240    libraries: HashMap<String, LibHandle>,
241    functions: HashMap<String, BoundFunction>,
242}
243
244impl IfaFfi {
245    pub fn new() -> Self {
246        IfaFfi {
247            libraries: HashMap::new(),
248            functions: HashMap::new(),
249        }
250    }
251
252    /// Load a shared library
253    /// Ifa syntax: ffi.load("mylib")
254    pub fn load(&mut self, name: &str, path: Option<&str>) -> FfiResult<()> {
255        let lib_path = path.map(String::from).unwrap_or_else(|| {
256            #[cfg(windows)]
257            {
258                format!("{}.dll", name)
259            }
260            #[cfg(target_os = "macos")]
261            {
262                format!("lib{}.dylib", name)
263            }
264            #[cfg(all(unix, not(target_os = "macos")))]
265            {
266                format!("lib{}.so", name)
267            }
268        });
269
270        let handle = load_library(&lib_path)?;
271        self.libraries.insert(name.to_string(), handle);
272        println!("[FFI] Loaded: {}", name);
273        Ok(())
274    }
275
276    /// Bind a function with signature
277    /// Ifa syntax: ffi.bind("lib", "func", ["i32", "i32"], "i32")
278    pub fn bind(&mut self, lib: &str, func: &str, args: &[&str], ret: &str) -> FfiResult<()> {
279        let handle = self
280            .libraries
281            .get(lib)
282            .ok_or_else(|| FfiError::LibraryNotFound(lib.to_string()))?;
283
284        // Safety: handle was obtained from load_library and is valid
285        let ptr = unsafe { get_proc(*handle, func)? };
286
287        let arg_types: Vec<IfaType> = args.iter().filter_map(|s| IfaType::from_str(s)).collect();
288
289        let ret_type = IfaType::from_str(ret).unwrap_or(IfaType::Void);
290
291        let key = format!("{}.{}", lib, func);
292        self.functions.insert(
293            key.clone(),
294            BoundFunction {
295                name: func.to_string(),
296                ptr,
297                sig: FfiSignature {
298                    arg_types,
299                    ret_type,
300                },
301            },
302        );
303
304        Ok(())
305    }
306
307    /// Call a bound function
308    /// Ifa syntax: ffi.call("lib", "func", arg1, arg2)
309    pub fn call(&self, lib: &str, func: &str, args: &[FfiValue]) -> FfiResult<FfiValue> {
310        let key = format!("{}.{}", lib, func);
311        let bound = self
312            .functions
313            .get(&key)
314            .ok_or_else(|| FfiError::FunctionNotFound(key))?;
315
316        // Type check
317        if args.len() != bound.sig.arg_types.len() {
318            return Err(FfiError::TypeMismatch(format!(
319                "Expected {} args, got {}",
320                bound.sig.arg_types.len(),
321                args.len()
322            )));
323        }
324
325        // For now, return placeholder - real implementation needs libffi
326        println!("[FFI] Call: {}({:?})", bound.name, args);
327        Ok(FfiValue::Null)
328    }
329
330    /// Get list of loaded libraries
331    pub fn loaded_libraries(&self) -> Vec<&str> {
332        self.libraries.keys().map(|s| s.as_str()).collect()
333    }
334}

336impl Default for IfaFfi {
337    fn default() -> Self {
338        Self::new()
339    }

346/// Secure FFI with whitelist
347pub struct SecureFfi {
348    inner: IfaFfi,
349    whitelist: HashSet<String>,
350    blocked_symbols: HashSet<String>,
351}
352
353impl SecureFfi {
354    pub fn new() -> Self {
355        let mut blocked = HashSet::new();
356        // Block dangerous symbols
357        blocked.insert("system".to_string());
358        blocked.insert("exec".to_string());
359        blocked.insert("execve".to_string());
360        blocked.insert("fork".to_string());
361        blocked.insert("popen".to_string());
362        blocked.insert("dlopen".to_string()); // Prevent nested loading
363
364        SecureFfi {
365            inner: IfaFfi::new(),
366            whitelist: HashSet::new(),
367            blocked_symbols: blocked,
368        }
369    }
370
371    /// Add library to whitelist
372    pub fn allow(&mut self, lib: &str) {
373        self.whitelist.insert(lib.to_string());
374    }
375
376    /// Load library (only if whitelisted)
377    pub fn load(&mut self, name: &str, path: Option<&str>) -> FfiResult<()> {
378        if !self.whitelist.contains(name) {
379            return Err(FfiError::SecurityViolation(format!(
380                "Library '{}' not whitelisted. Use allow() first.",
381                name
382            )));
383        }
384        self.inner.load(name, path)
385    }
386
387    /// Bind function (blocks dangerous symbols)
388    pub fn bind(&mut self, lib: &str, func: &str, args: &[&str], ret: &str) -> FfiResult<()> {
389        if self.blocked_symbols.contains(func) {
390            return Err(FfiError::SecurityViolation(format!(
391                "Symbol '{}' is blocked for security reasons",
392                func
393            )));
394        }
395        self.inner.bind(lib, func, args, ret)
396    }
397
398    /// Call function
399    pub fn call(&self, lib: &str, func: &str, args: &[FfiValue]) -> FfiResult<FfiValue> {
400        self.inner.call(lib, func, args)
401    }
402}
403
404impl Default for SecureFfi {
405    fn default() -> Self {
406        Self::new()
407    }
408}

414/// Generate C header for Ifa exports
415pub fn generate_c_header(exports: &[(String, FfiSignature)]) -> String {
416    let mut out = String::new();
417
418    out.push_str("/* IFA-LANG C API - Auto-generated */\n");
419    out.push_str("#ifndef IFA_API_H\n");
420    out.push_str("#define IFA_API_H\n\n");
421    out.push_str("#include <stdint.h>\n\n");
422    out.push_str("#ifdef __cplusplus\n");
423    out.push_str("extern \"C\" {\n");
424    out.push_str("#endif\n\n");
425
426    for (name, sig) in exports {
427        let ret = sig.ret_type.c_name();
428        let args: Vec<String> = sig
429            .arg_types
430            .iter()
431            .enumerate()
432            .map(|(i, t)| format!("{} arg{}", t.c_name(), i))
433            .collect();
434        let args_str = if args.is_empty() {
435            "void".to_string()
436        } else {
437            args.join(", ")
438        };
439
440        let c_name = name.replace(".", "_");
441        out.push_str(&format!("{} ifa_{}({});\n", ret, c_name, args_str));
442    }
443
444    out.push_str("\n#ifdef __cplusplus\n");
445    out.push_str("}\n");
446    out.push_str("#endif\n\n");
447    out.push_str("#endif /* IFA_API_H */\n");
448
449    out
450}

452/// Generate Rust FFI bindings
453pub fn generate_rust_bindings(exports: &[(String, FfiSignature)]) -> String {
454    let mut out = String::new();
455
456    out.push_str("// IFA-LANG Rust Bindings - Auto-generated\n");
457    out.push_str("use std::os::raw::{c_char, c_void};\n\n");
458    out.push_str("extern \"C\" {\n");
459
460    for (name, sig) in exports {
461        let ret = sig.ret_type.rust_name();
462        let args: Vec<String> = sig
463            .arg_types
464            .iter()
465            .enumerate()
466            .map(|(i, t)| format!("arg{}: {}", i, t.rust_name()))
467            .collect();
468        let args_str = args.join(", ");
469
470        let rust_name = name.replace(".", "_");
471        out.push_str(&format!(
472            "    pub fn ifa_{}({}) -> {};\n",
473            rust_name, args_str, ret
474        ));
475    }
476
477    out.push_str("}\n");
478
479    out
480}

486/// Endpoint information
487#[derive(Debug, Clone)]
488pub struct Endpoint {
489    pub name: String,
490    pub handler_id: usize,
491    pub arg_types: Vec<IfaType>,
492    pub ret_type: IfaType,
493}
494
495/// Handler function type
496pub type ApiHandler = Box<dyn Fn(&[FfiValue]) -> FfiResult<FfiValue> + Send + Sync>;
497
498/// API Layer for exposing Ifa functions
499pub struct IfaApi {
500    endpoints: HashMap<String, Endpoint>,
501    handlers: Vec<ApiHandler>,
502}
503
504impl IfaApi {
505    pub fn new() -> Self {
506        IfaApi {
507            endpoints: HashMap::new(),
508            handlers: Vec::new(),
509        }
510    }
511
512    /// Expose a function as an API endpoint
513    pub fn expose<F>(&mut self, name: &str, arg_types: &[IfaType], ret_type: IfaType, handler: F)
514    where
515        F: Fn(&[FfiValue]) -> FfiResult<FfiValue> + Send + Sync + 'static,
516    {
517        let handler_id = self.handlers.len();
518        self.handlers.push(Box::new(handler));
519
520        self.endpoints.insert(
521            name.to_string(),
522            Endpoint {
523                name: name.to_string(),
524                handler_id,
525                arg_types: arg_types.to_vec(),
526                ret_type,
527            },
528        );
529    }
530
531    /// Call an exposed endpoint
532    pub fn call(&self, name: &str, args: &[FfiValue]) -> FfiResult<FfiValue> {
533        let endpoint = self
534            .endpoints
535            .get(name)
536            .ok_or_else(|| FfiError::FunctionNotFound(name.to_string()))?;
537
538        // Type check arg count
539        if args.len() != endpoint.arg_types.len() {
540            return Err(FfiError::TypeMismatch(format!(
541                "{}: expected {} args, got {}",
542                name,
543                endpoint.arg_types.len(),
544                args.len()
545            )));
546        }
547
548        let handler = &self.handlers[endpoint.handler_id];
549        handler(args)
550    }
551
552    /// List all endpoints
553    pub fn list_endpoints(&self) -> Vec<&Endpoint> {
554        self.endpoints.values().collect()
555    }
556
557    /// Generate JSON schema for all endpoints
558    pub fn to_json_schema(&self) -> String {
559        let mut out = String::from("{\n");
560        let mut first = true;
561
562        for (name, ep) in &self.endpoints {
563            if !first {
564                out.push_str(",\n");
565            }
566            first = false;
567
568            let args: Vec<&str> = ep.arg_types.iter().map(|t| t.rust_name()).collect();
569            out.push_str(&format!(
570                "  \"{}\": {{ \"args\": {:?}, \"returns\": \"{}\" }}",
571                name,
572                args,
573                ep.ret_type.rust_name()
574            ));
575        }
576
577        out.push_str("\n}");
578        out
579    }
580
581    /// Export as C header
582    pub fn generate_c_header(&self) -> String {
583        let exports: Vec<(String, FfiSignature)> = self
584            .endpoints
585            .values()
586            .map(|ep| {
587                (
588                    ep.name.clone(),
589                    FfiSignature {
590                        arg_types: ep.arg_types.clone(),
591                        ret_type: ep.ret_type,
592                    },
593                )
594            })
595            .collect();
596        generate_c_header(&exports)
597    }
598
599    /// Export as Rust bindings
600    pub fn generate_rust_bindings(&self) -> String {
601        let exports: Vec<(String, FfiSignature)> = self
602            .endpoints
603            .values()
604            .map(|ep| {
605                (
606                    ep.name.clone(),
607                    FfiSignature {
608                        arg_types: ep.arg_types.clone(),
609                        ret_type: ep.ret_type,
610                    },
611                )
612            })
613            .collect();
614        generate_rust_bindings(&exports)
615    }
616}

636/// JSON-RPC Response
637#[derive(Debug)]
638pub struct RpcResponse {
639    pub id: u64,
640    pub result: Option<FfiValue>,
641    pub error: Option<String>,
642}
643
644impl RpcResponse {
645    pub fn success(id: u64, result: FfiValue) -> Self {
646        RpcResponse {
647            id,
648            result: Some(result),
649            error: None,
650        }
651    }
652
653    pub fn error(id: u64, msg: String) -> Self {
654        RpcResponse {
655            id,
656            result: None,
657            error: Some(msg),
658        }
659    }
660
661    /// Convert to JSON string
662    pub fn to_json(&self) -> String {
663        if let Some(ref err) = self.error {
664            format!(
665                "{{\"jsonrpc\":\"2.0\",\"error\":{{\"code\":-1,\"message\":\"{}\"}},\"id\":{}}}",
666                err, self.id
667            )
668        } else {
669            let result_json = match &self.result {
670                Some(FfiValue::I32(v)) => v.to_string(),
671                Some(FfiValue::I64(v)) => v.to_string(),
672                Some(FfiValue::F64(v)) => v.to_string(),
673                Some(FfiValue::Str(s)) => format!("\"{}\"", s),
674                Some(FfiValue::U8(v)) => v.to_string(),
675                Some(FfiValue::Ptr(v)) => v.to_string(),
676                Some(FfiValue::Null) | None => "null".to_string(),
677            };
678            format!(
679                "{{\"jsonrpc\":\"2.0\",\"result\":{},\"id\":{}}}",
680                result_json, self.id
681            )
682        }
683    }
684}

686/// Simple RPC Server
687pub struct IfaRpcServer {
688    api: IfaApi,
689    port: u16,
690}
691
692impl IfaRpcServer {
693    pub fn new(api: IfaApi, port: u16) -> Self {
694        IfaRpcServer { api, port }
695    }
696
697    /// Handle a JSON-RPC request string
698    pub fn handle_request(&self, json: &str) -> String {
699        // Simple JSON parsing (no dependencies)
700        let id = self.extract_id(json).unwrap_or(1);
701        let method = match self.extract_string(json, "method") {
702            Some(m) => m,
703            None => return RpcResponse::error(id, "Missing method".to_string()).to_json(),
704        };
705
706        let params = self.extract_params(json);
707
708        match self.api.call(&method, &params) {
709            Ok(result) => RpcResponse::success(id, result).to_json(),
710            Err(e) => RpcResponse::error(id, e.to_string()).to_json(),
711        }
712    }
713
714    /// Extract id from JSON
715    fn extract_id(&self, json: &str) -> Option<u64> {
716        let id_start = json.find("\"id\":")?;
717        let rest = &json[id_start + 5..];
718        let id_str: String = rest
719            .chars()
720            .skip_while(|c| c.is_whitespace())
721            .take_while(|c| c.is_ascii_digit())
722            .collect();
723        id_str.parse().ok()
724    }
725
726    /// Extract string field from JSON
727    fn extract_string(&self, json: &str, field: &str) -> Option<String> {
728        let pattern = format!("\"{}\":\"", field);
729        let start = json.find(&pattern)? + pattern.len();
730        let rest = &json[start..];
731        let end = rest.find('"')?;
732        Some(rest[..end].to_string())
733    }
734
735    /// Extract params array (simplified)
736    fn extract_params(&self, json: &str) -> Vec<FfiValue> {
737        // Find params array
738        let params_start = match json.find("\"params\":[") {
739            Some(p) => p + 10,
740            None => return Vec::new(),
741        };
742
743        let rest = &json[params_start..];
744        let params_end = match rest.find(']') {
745            Some(e) => e,
746            None => return Vec::new(),
747        };
748
749        let params_str = &rest[..params_end];
750
751        // Parse simple values
752        params_str
753            .split(',')
754            .filter_map(|s| {
755                let s = s.trim();
756                if s.is_empty() {
757                    None
758                } else if s.starts_with('"') && s.ends_with('"') {
759                    Some(FfiValue::Str(s[1..s.len() - 1].to_string()))
760                } else if s.contains('.') {
761                    s.parse::<f64>().ok().map(FfiValue::F64)
762                } else {
763                    s.parse::<i64>().ok().map(FfiValue::I64)
764                }
765            })
766            .collect()
767    }
768
769    /// Start HTTP server (blocking)
770    #[cfg(feature = "full")]
771    #[allow(dead_code)]
772    pub fn start(&self) -> std::io::Result<()> {
773        use std::io::{Read, Write};
774        use std::net::TcpListener;
775
776        let listener = TcpListener::bind(format!("127.0.0.1:{}", self.port))?;
777        println!("[RPC] Server started on port {}", self.port);
778
779        for stream in listener.incoming() {
780            match stream {
781                Ok(mut stream) => {
782                    let mut buffer = [0u8; 4096];
783                    if let Ok(n) = stream.read(&mut buffer) {
784                        let request = String::from_utf8_lossy(&buffer[..n]);
785
786                        // Extract body from HTTP request
787                        if let Some(body_start) = request.find("\r\n\r\n") {
788                            let body = &request[body_start + 4..];
789                            let response = self.handle_request(body);
790
791                            let http_response = format!(
792                                "HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: {}\r\n\r\n{}",
793                                response.len(),
794                                response
795                            );
796                            let _ = stream.write_all(http_response.as_bytes());
797                        }
798                    }
799                }
800                Err(e) => eprintln!("[RPC] Connection error: {}", e),
801            }
802        }
803        Ok(())
804    }
805
806    /// Get port
807    pub fn port(&self) -> u16 {
808        self.port
809    }

816/// Create API from Ifa stdlib (16 Odu domains)
817pub fn create_stdlib_api() -> IfaApi {
818    let mut api = IfaApi::new();
819
820    // Obara (Math - Add/Mul)
821    api.expose(
822        "obara.fikun",
823        &[IfaType::I64, IfaType::I64],
824        IfaType::I64,
825        |args| {
826            let a = args.get(0).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
827            let b = args.get(1).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
828            Ok(FfiValue::I64(a + b))
829        },
830    );
831
832    api.expose(
833        "obara.isodipupo",
834        &[IfaType::I64, IfaType::I64],
835        IfaType::I64,
836        |args| {
837            let a = args.get(0).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
838            let b = args.get(1).and_then(|v| v.as_i32()).unwrap_or(1) as i64;
839            Ok(FfiValue::I64(a * b))
840        },
841    );
842
843    // Oturupon (Math - Sub/Div)
844    api.expose(
845        "oturupon.din",
846        &[IfaType::I64, IfaType::I64],
847        IfaType::I64,
848        |args| {
849            let a = args.get(0).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
850            let b = args.get(1).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
851            Ok(FfiValue::I64(a - b))
852        },
853    );
854
855    api.expose(
856        "oturupon.pin",
857        &[IfaType::I64, IfaType::I64],
858        IfaType::F64,
859        |args| {
860            let a = args.get(0).and_then(|v| v.as_f64()).unwrap_or(0.0);
861            let b = args.get(1).and_then(|v| v.as_f64()).unwrap_or(1.0);
862            if b == 0.0 {
863                Err(FfiError::CallFailed("Division by zero".to_string()))
864            } else {
865                Ok(FfiValue::F64(a / b))
866            }
867        },
868    );
869
870    // Ika (Strings)
871    api.expose("ika.gigun", &[IfaType::Str], IfaType::I64, |args| {
872        let s = args.get(0).and_then(|v| v.as_str()).unwrap_or("");
873        Ok(FfiValue::I64(s.len() as i64))
874    });
875
876    // Iwori (Time)
877    api.expose("iwori.epoch", &[], IfaType::I64, |_args| {
878        use std::time::{SystemTime, UNIX_EPOCH};
879        let epoch = SystemTime::now()
880            .duration_since(UNIX_EPOCH)
881            .map(|d| d.as_secs() as i64)
882            .unwrap_or(0);
883        Ok(FfiValue::I64(epoch))
884    });
885
886    // Owonrin (Random)
887    api.expose(
888        "owonrin.afesona",
889        &[IfaType::I64, IfaType::I64],
890        IfaType::I64,
891        |args| {
892            let min = args.get(0).and_then(|v| v.as_i32()).unwrap_or(0) as i64;
893            let max = args.get(1).and_then(|v| v.as_i32()).unwrap_or(100) as i64;
894
895            // Simple LCG random
896            use std::time::{SystemTime, UNIX_EPOCH};
897            let seed = SystemTime::now()
898                .duration_since(UNIX_EPOCH)
899                .map(|d| d.as_nanos() as u64)
900                .unwrap_or(12345);
901
902            let range = (max - min).max(1) as u64;
903            let random =
904                ((seed.wrapping_mul(6364136223846793005).wrapping_add(1)) % range) as i64 + min;
905            Ok(FfiValue::I64(random))
906        },
907    );
908
909    // Ogbe (System)
910    api.expose("ogbe.version", &[], IfaType::Str, |_args| {
911        Ok(FfiValue::Str("1.0.0".to_string()))
912    });
913
914    api
915}
