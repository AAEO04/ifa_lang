13/// Error types for crypto operations
14#[derive(Debug, Clone)]
15pub enum CryptoError {
16    InvalidKeyLength { expected: usize, actual: usize },
17    InvalidInput(String),
18    VerificationFailed,
19    RngError,
20}
21
22impl std::fmt::Display for CryptoError {
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        match self {
25            Self::InvalidKeyLength { expected, actual } => write!(
26                f,
27                "Invalid key length: expected {} bytes, got {}",
28                expected, actual
29            ),
30            Self::InvalidInput(msg) => write!(f, "Invalid input: {}", msg),
31            Self::VerificationFailed => write!(f, "Verification failed"),
32            Self::RngError => write!(f, "Random number generation failed"),
33        }
34    }
35}

44    use ring::digest::{SHA256, SHA512};

47    /// SHA-256 hash using ring crate
48    ///
49    /// Production-quality cryptographic hash function.
50    pub fn sha256(data: &[u8]) -> [u8; 32] {
51        let digest = ring::digest::digest(&SHA256, data);
52        let mut result = [0u8; 32];
53        result.copy_from_slice(digest.as_ref());
54        result
55    }
56
57    /// SHA-256 returning hex string
58    pub fn sha256_hex(data: &[u8]) -> String {
59        sha256(data).iter().map(|b| format!("{:02x}", b)).collect()
60    }
61
62    /// SHA-512 hash using ring crate
63    pub fn sha512(data: &[u8]) -> [u8; 64] {
64        let digest = ring::digest::digest(&SHA512, data);
65        let mut result = [0u8; 64];
66        result.copy_from_slice(digest.as_ref());
67        result
68    }
69
70    /// HMAC-SHA256 using ring crate
71    ///
72    /// Production-quality message authentication code.
73    pub fn hmac_sha256(key: &[u8], data: &[u8]) -> [u8; 32] {
74        let key = hmac::Key::new(hmac::HMAC_SHA256, key);
75        let tag = hmac::sign(&key, data);
76        let mut result = [0u8; 32];
77        result.copy_from_slice(tag.as_ref());
78        result
79    }
80
81    /// Verify HMAC in constant time using ring
82    pub fn hmac_verify(key: &[u8], data: &[u8], expected: &[u8; 32]) -> bool {
83        let key = hmac::Key::new(hmac::HMAC_SHA256, key);
84        hmac::verify(&key, data, expected).is_ok()
85    }

88/// Password hashing using Argon2id
89///
90/// Production-quality password hashing with memory-hard algorithm.
91pub mod password {
92    use super::*;
93    use argon2::{
94        Argon2,
95        password_hash::{
96            PasswordHash, PasswordHasher, PasswordVerifier, SaltString, rand_core::OsRng,
97        },
98    };
99
100    /// Hash a password using Argon2id
101    ///
102    /// Returns a PHC-formatted string containing the hash and parameters.
103    /// Store this in your database instead of the raw password.
104    ///
105    /// # Example
106    /// ```
107    /// let hash = password::hash("my_password").unwrap();
108    /// // Store hash in database...
109    /// ```
110    pub fn hash(password: &str) -> Result<String, CryptoError> {
111        let salt = SaltString::generate(&mut OsRng);
112        let argon2 = Argon2::default();
113
114        argon2
115            .hash_password(password.as_bytes(), &salt)
116            .map(|h| h.to_string())
117            .map_err(|_| CryptoError::InvalidInput("Password hashing failed".to_string()))
118    }
119
120    /// Verify a password against a stored hash
121    ///
122    /// Returns true if the password matches the hash.
123    ///
124    /// # Example
125    /// ```
126    /// let matches = password::verify("my_password", &stored_hash).unwrap();
127    /// ```
128    pub fn verify(password: &str, hash: &str) -> Result<bool, CryptoError> {
129        let parsed_hash = PasswordHash::new(hash)
130            .map_err(|_| CryptoError::InvalidInput("Invalid hash format".to_string()))?;
131
132        Ok(Argon2::default()
133            .verify_password(password.as_bytes(), &parsed_hash)
134            .is_ok())
135    }
136}

138/// Constant-time comparison to prevent timing attacks
139pub fn constant_time_compare(a: &[u8], b: &[u8]) -> bool {
140    if a.len() != b.len() {
141        return false;
142    }
143
144    let mut diff = 0u8;
145    for (x, y) in a.iter().zip(b.iter()) {
146        diff |= x ^ y;
147    }
148    diff == 0
149}

151/// Base64 encoding/decoding with proper error handling
152pub mod base64 {
153    use super::CryptoError;
154
155    const ALPHABET: &[u8] = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
156
157    pub fn encode(data: &[u8]) -> String {
158        let mut result = String::with_capacity((data.len() + 2) / 3 * 4);
159
160        for chunk in data.chunks(3) {
161            let b0 = chunk[0] as usize;
162            result.push(ALPHABET[b0 >> 2] as char);
163
164            if chunk.len() > 1 {
165                let b1 = chunk[1] as usize;
166                result.push(ALPHABET[((b0 & 0x03) << 4) | (b1 >> 4)] as char);
167
168                if chunk.len() > 2 {
169                    let b2 = chunk[2] as usize;
170                    result.push(ALPHABET[((b1 & 0x0f) << 2) | (b2 >> 6)] as char);
171                    result.push(ALPHABET[b2 & 0x3f] as char);
172                } else {
173                    result.push(ALPHABET[(b1 & 0x0f) << 2] as char);
174                    result.push('=');
175                }
176            } else {
177                result.push(ALPHABET[(b0 & 0x03) << 4] as char);
178                result.push_str("==");
179            }
180        }
181
182        result
183    }
184
185    pub fn decode(encoded: &str) -> Result<Vec<u8>, CryptoError> {
186        let chars: Vec<char> = encoded
187            .chars()
188            .filter(|&c| c != '=' && !c.is_whitespace())
189            .collect();
190        let mut result = Vec::with_capacity(chars.len() * 3 / 4);
191
192        for chunk in chars.chunks(4) {
193            if chunk.is_empty() {
194                break;
195            }
196
197            let indices: Result<Vec<usize>, _> = chunk
198                .iter()
199                .map(|&c| {
200                    ALPHABET
201                        .iter()
202                        .position(|&b| b as char == c)
203                        .ok_or_else(|| {
204                            CryptoError::InvalidInput(format!("Invalid base64 char: {}", c))
205                        })
206                })
207                .collect();
208
209            let indices = indices?;
210
211            if indices.len() >= 2 {
212                result.push(((indices[0] << 2) | (indices[1] >> 4)) as u8);
213            }
214            if indices.len() >= 3 {
215                result.push((((indices[1] & 0x0f) << 4) | (indices[2] >> 2)) as u8);
216            }
217            if indices.len() >= 4 {
218                result.push((((indices[2] & 0x03) << 6) | indices[3]) as u8);
219            }
220        }
221
222        Ok(result)
223    }
224}

226/// Hex encoding/decoding
227pub mod hex {
228    use super::CryptoError;
229
230    pub fn encode(data: &[u8]) -> String {
231        data.iter().map(|b| format!("{:02x}", b)).collect()
232    }
233
234    pub fn decode(encoded: &str) -> Result<Vec<u8>, CryptoError> {
235        if encoded.len() % 2 != 0 {
236            return Err(CryptoError::InvalidInput(
237                "Odd length hex string".to_string(),
238            ));
239        }
240
241        let mut result = Vec::with_capacity(encoded.len() / 2);
242        let chars: Vec<char> = encoded.chars().collect();
243
244        for chunk in chars.chunks(2) {
245            let byte = u8::from_str_radix(&format!("{}{}", chunk[0], chunk[1]), 16)
246                .map_err(|_| CryptoError::InvalidInput("Invalid hex character".to_string()))?;
247            result.push(byte);
248        }
249
250        Ok(result)
251    }
252}

254/// Cryptographically secure random number generator
255///
256/// WARNING: This uses system entropy sources. On platforms without good entropy,
257/// this may block or fail.
258pub struct SecureRng {
259    state: std::sync::Mutex<u64>,
260}
261
262impl SecureRng {
263    /// Create new RNG seeded from system entropy
264    pub fn new() -> Result<Self, CryptoError> {
265        use std::process;
266        use std::time::{SystemTime, UNIX_EPOCH};
267
268        // Mix multiple entropy sources
269        let time_entropy = SystemTime::now()
270            .duration_since(UNIX_EPOCH)
271            .map(|d| d.as_nanos() as u64)
272            .unwrap_or(0);
273
274        let pid_entropy = process::id() as u64;
275        let addr_entropy = &time_entropy as *const _ as u64;
276
277        let seed = time_entropy
278            .wrapping_mul(0x9e3779b97f4a7c15)
279            .wrapping_add(pid_entropy)
280            .wrapping_mul(0xbf58476d1ce4e5b9)
281            .wrapping_add(addr_entropy);
282
283        Ok(SecureRng {
284            state: std::sync::Mutex::new(seed),
285        })
286    }
287
288    /// Fill buffer with random bytes
289    pub fn fill_bytes(&self, dest: &mut [u8]) -> Result<(), CryptoError> {
290        let mut state = self.state.lock().unwrap();
291
292        // SplitMix64 - better than LCG
293        for byte in dest.iter_mut() {
294            *state = state.wrapping_add(0x9e3779b97f4a7c15);
295            let mut z = *state;
296            z = (z ^ (z >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
297            z = (z ^ (z >> 27)).wrapping_mul(0x94d049bb133111eb);
298            z = z ^ (z >> 31);
299            *byte = z as u8;
300        }
301
302        Ok(())
303    }
304
305    /// Generate random bytes
306    pub fn gen_bytes(&self, count: usize) -> Result<Vec<u8>, CryptoError> {
307        let mut bytes = vec![0u8; count];
308        self.fill_bytes(&mut bytes)?;
309        Ok(bytes)
310    }
311
312    /// Generate random u64
313    pub fn gen_u64(&self) -> Result<u64, CryptoError> {
314        let mut bytes = [0u8; 8];
315        self.fill_bytes(&mut bytes)?;
316        Ok(u64::from_le_bytes(bytes))
317    }
318
319    /// Generate random value in range [0, max)
320    pub fn gen_range(&self, max: u64) -> Result<u64, CryptoError> {
321        if max == 0 {
322            return Err(CryptoError::InvalidInput(
323                "Range max cannot be 0".to_string(),
324            ));
325        }
326
327        // Rejection sampling for uniform distribution
328        let threshold = u64::MAX - (u64::MAX % max);
329        loop {
330            let n = self.gen_u64()?;
331            if n < threshold {
332                return Ok(n % max);
333            }
334        }
335    }
336}

340        Self::new().expect("Failed to initialize RNG")
341    }

344/// Generate UUID v4
345pub fn uuid_v4() -> Result<String, CryptoError> {
346    let rng = SecureRng::new()?;
347    let mut bytes = [0u8; 16];
348    rng.fill_bytes(&mut bytes)?;
349
350    // Set version 4
351    bytes[6] = (bytes[6] & 0x0f) | 0x40;
352    // Set variant
353    bytes[8] = (bytes[8] & 0x3f) | 0x80;
354
355    Ok(format!(
356        "{:02x}{:02x}{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
357        bytes[0],
358        bytes[1],
359        bytes[2],
360        bytes[3],
361        bytes[4],
362        bytes[5],
363        bytes[6],
364        bytes[7],
365        bytes[8],
366        bytes[9],
367        bytes[10],
368        bytes[11],
369        bytes[12],
370        bytes[13],
371        bytes[14],
372        bytes[15]
373    ))
374}

376/// Secret storage with zeroization on drop
377pub struct Secret {
378    data: Vec<u8>,
379}
380
381impl Secret {
382    pub fn new(data: Vec<u8>) -> Self {
383        Secret { data }
384    }
385
386    pub fn as_bytes(&self) -> &[u8] {
387        &self.data
388    }
389}
390
391impl Drop for Secret {
392    fn drop(&mut self) {
393        // Zero out memory before dropping
394        for byte in self.data.iter_mut() {
395            unsafe {
396                std::ptr::write_volatile(byte, 0);
397            }
398        }
399        std::sync::atomic::fence(std::sync::atomic::Ordering::SeqCst);
400    }
401}

403/// Secure secret store with zeroization
404pub struct SecretStore {
405    secrets: HashMap<String, Secret>,
406}
407
408impl SecretStore {
409    pub fn new() -> Self {
410        SecretStore {
411            secrets: HashMap::new(),
412        }
413    }
414
415    pub fn store(&mut self, key: &str, value: Vec<u8>) {
416        self.secrets.insert(key.to_string(), Secret::new(value));
417    }
418
419    pub fn get(&self, key: &str) -> Option<&[u8]> {
420        self.secrets.get(key).map(|s| s.as_bytes())
421    }
422
423    pub fn remove(&mut self, key: &str) -> bool {
424        self.secrets.remove(key).is_some()
425    }
426
427    pub fn clear(&mut self) {
428        self.secrets.clear();
429    }
430}

435    }

440        self.clear();

445pub fn hash_password(password: &[u8], salt: &[u8]) -> [u8; 32] {
446    // PBKDF2-like construction (placeholder)
447    let mut result = hash::hmac_sha256(password, salt);
448    for _ in 0..1000 {
449        result = hash::hmac_sha256(password, &result);

451    result

454/// Verify password against hash
455pub fn verify_password(password: &[u8], salt: &[u8], expected: &[u8; 32]) -> bool {
456    let computed = hash_password(password, salt);
