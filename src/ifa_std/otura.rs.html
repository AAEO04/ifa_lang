13/// Blocked hosts for SSRF protection
14const BLOCKED_HOSTS: &[&str] = &[
15    "localhost",
16    "127.0.0.1",
17    "0.0.0.0",
18    "::1",
19    "metadata.google.internal",
20    "169.254.169.254", // AWS/Cloud metadata
21];

38    /// Check if host is allowed (SSRF protection + Capability)
39    pub fn ṣàyẹ̀wò(&self, host: &str) -> bool {
40        // First check capability
41        if !self.capabilities.check(&Ofun::Network {
42            domains: vec![host.to_string()],
43        }) {
44            return false;
45        }
46
47        let host_lower = host.to_lowercase();
48
49        // Block known dangerous hosts
50        for blocked in BLOCKED_HOSTS {
51            if host_lower == *blocked || host_lower.ends_with(blocked) {
52                return false;
53            }
54        }
55
56        // Block private IP ranges
57        if let Ok(ip) = host.parse::<IpAddr>() {
58            match ip {
59                IpAddr::V4(v4) => {
60                    if v4.is_loopback() || v4.is_private() || v4.is_link_local() {
61                        return false;
62                    }
63                }
64                IpAddr::V6(v6) => {
65                    if v6.is_loopback() {
66                        return false;
67                    }
68                }
69            }
70        }
71
72        true
73    }
74}

76#[cfg(feature = "full")]
77impl Otura {
78    /// HTTP GET request (gbà)
79    pub async fn gba(&self, url: &str) -> IfaResult<String> {
80        // Extract host for SSRF check
81        if let Some(host) = url.split("://").nth(1).and_then(|s| s.split('/').next()) {
82            let host = host.split(':').next().unwrap_or(host);
83            if !self.ṣàyẹ̀wò(host) {
84                return Err(IfaError::SsrfBlocked(host.to_string()));
85            }
86        } else {
87            return Err(IfaError::Runtime("Invalid URL".into()));
88        }
89
90        reqwest::get(url)
91            .await
92            .map_err(|e| IfaError::ConnectionFailed(e.to_string()))?
93            .text()
94            .await
95            .map_err(|e| IfaError::Custom(format!("Response error: {}", e)))
96    }
97
98    /// HTTP POST request (rán)
99    pub async fn ran(&self, url: &str, body: &str) -> IfaResult<String> {
100        // SSFR Check
101        if let Some(host) = url.split("://").nth(1).and_then(|s| s.split('/').next()) {
102            let host = host.split(':').next().unwrap_or(host);
103            if !self.ṣàyẹ̀wò(host) {
104                return Err(IfaError::SsrfBlocked(host.to_string()));
105            }
106        }
107
108        let client = reqwest::Client::new();
109        client
110            .post(url)
111            .body(body.to_string())
112            .send()
113            .await
114            .map_err(|e| IfaError::ConnectionFailed(e.to_string()))?
115            .text()
116            .await
117            .map_err(|e| IfaError::Custom(format!("Response error: {}", e)))
118    }
119
120    /// TCP listen (dè)
121    pub async fn de(&self, addr: &str) -> IfaResult<TcpListener> {
122        if !self.capabilities.check(&Ofun::Network {
123            domains: vec!["*".to_string()],
124        }) {
125            return Err(IfaError::PermissionDenied("Network bind denied".into()));
126        }
127
128        TcpListener::bind(addr)
129            .await
130            .map_err(|e| IfaError::Custom(format!("Bind error: {}", e)))
131    }
132
133    /// TCP connect (sọ̀rọ̀)
134    pub async fn soro(&self, addr: &str) -> IfaResult<TcpStream> {
135        // Extract host
136        let host = addr.split(':').next().unwrap_or(addr);
137        if !self.ṣàyẹ̀wò(host) {
138            return Err(IfaError::SsrfBlocked(host.to_string()));
139        }
140
141        TcpStream::connect(addr)
142            .await
143            .map_err(|e| IfaError::ConnectionFailed(e.to_string()))
144    }
145}

155#[cfg(test)]
156mod tests {
157    use super::*;
158
159    #[test]
160    fn test_ssrf_protection() {
161        // Create Otura with network capabilities granted for test domains
162        let mut caps = CapabilitySet::default();
163        caps.grant(Ofun::Network {
164            domains: vec!["example.com".to_string(), "8.8.8.8".to_string()],
165        });
166        let otura = Otura::new(caps);
167
168        // Should block (localhost blocked even with capability due to SSRF protection)
169        assert!(!otura.ṣàyẹ̀wò("localhost"));
170        assert!(!otura.ṣàyẹ̀wò("127.0.0.1"));
171        assert!(!otura.ṣàyẹ̀wò("169.254.169.254"));
172
173        // Should allow (granted in capability)
174        assert!(otura.ṣàyẹ̀wò("example.com"));
175        assert!(otura.ṣàyẹ̀wò("8.8.8.8"));
176    }
