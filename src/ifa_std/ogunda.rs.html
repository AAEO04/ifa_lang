1//! # Ògúndá Domain (1110)
2//!
3//! The Warrior - Arrays and Process Management
4//!
5//! Vec with iterators and subprocess spawning via tokio-process.

7use crate::impl_odu_domain;
8use ifa_core::error::{IfaError, IfaResult};
9use std::process::{Command, Output, Stdio};
10
11/// Ògúndá - The Warrior (Arrays/Processes)
12pub struct Ogunda;
13
14impl_odu_domain!(Ogunda, "Ògúndá", "1110", "The Warrior - Arrays/Processes");
15
16impl Ogunda {
17    // =========================================================================
18    // ARRAY OPERATIONS
19    // =========================================================================
20
21    /// Create new list (ṣẹ̀dá)
22    pub fn seda<T>(&self) -> Vec<T> {
23        Vec::new()
24    }
25
26    /// Create list with capacity
27    pub fn seda_agbara<T>(&self, capacity: usize) -> Vec<T> {
28        Vec::with_capacity(capacity)
29    }
30
31    /// Push to list (fí)
32    pub fn fi<T>(&self, list: &mut Vec<T>, item: T) {
33        list.push(item);
34    }
35
36    /// Pop from list (mú)
37    pub fn mu<T>(&self, list: &mut Vec<T>) -> Option<T> {
38        list.pop()
39    }
40
41    /// Get length (ìwọ̀n)
42    pub fn iwon<T>(&self, list: &[T]) -> usize {
43        list.len()
44    }
45
46    /// Check if empty (ṣófo)
47    pub fn sofo<T>(&self, list: &[T]) -> bool {
48        list.is_empty()
49    }
50
51    /// Reverse list (padà)
52    pub fn pada<T: Clone>(&self, list: &[T]) -> Vec<T> {
53        list.iter().cloned().rev().collect()
54    }
55
56    /// Sort list (tò)
57    pub fn to<T: Ord + Clone>(&self, list: &[T]) -> Vec<T> {
58        let mut sorted = list.to_vec();
59        sorted.sort();
60        sorted
61    }
62
63    /// Concatenate lists (dàpọ̀)
64    pub fn dapo<T: Clone>(&self, a: &[T], b: &[T]) -> Vec<T> {
65        let mut result = a.to_vec();
66        result.extend(b.iter().cloned());
67        result
68    }
69
70    /// Filter list (yàn)
71    pub fn yan<T: Clone, F: Fn(&T) -> bool>(&self, list: &[T], predicate: F) -> Vec<T> {
72        list.iter().filter(|x| predicate(x)).cloned().collect()
73    }
74
75    /// Map over list (yí padà)
76    pub fn yi_pada<T, U, F: Fn(T) -> U>(&self, list: Vec<T>, mapper: F) -> Vec<U> {
77        list.into_iter().map(mapper).collect()
78    }
79
80    /// Find first matching (wá)
81    pub fn wa<'a, T, F: Fn(&T) -> bool>(&self, list: &'a [T], predicate: F) -> Option<&'a T> {
82        list.iter().find(|x| predicate(x))
83    }
84
85    /// Check if any match (èyíkéyìí)
86    pub fn eyikeyi<T, F: Fn(&T) -> bool>(&self, list: &[T], predicate: F) -> bool {
87        list.iter().any(|x| predicate(x))
88    }
89
90    /// Check if all match (gbogbo)
91    pub fn gbogbo<T, F: Fn(&T) -> bool>(&self, list: &[T], predicate: F) -> bool {
92        list.iter().all(|x| predicate(x))
93    }
94
95    /// Slice list
96    pub fn ge<T: Clone>(&self, list: &[T], start: usize, end: usize) -> Vec<T> {
97        list.get(start..end).map(|s| s.to_vec()).unwrap_or_default()
98    }
99
100    // =========================================================================
101    // PROCESS OPERATIONS
102    // =========================================================================
103
104    /// Run command and get output (ṣiṣẹ́)
105    pub fn sise(&self, command: &str, args: &[&str]) -> IfaResult<Output> {
106        Command::new(command)
107            .args(args)
108            .output()
109            .map_err(|e| IfaError::Custom(format!("Process error: {}", e)))
110    }
111
112    /// Run command and get stdout as string
113    pub fn sise_ka(&self, command: &str, args: &[&str]) -> IfaResult<String> {
114        let output = self.sise(command, args)?;
115        String::from_utf8(output.stdout)
116            .map_err(|e| IfaError::Custom(format!("UTF-8 error: {}", e)))
117    }
118
119    /// Spawn detached process (bẹ̀rẹ̀)
120    pub fn bere(&self, command: &str, args: &[&str]) -> IfaResult<u32> {
121        let child = Command::new(command)
122            .args(args)
123            .stdin(Stdio::null())
124            .stdout(Stdio::null())
125            .stderr(Stdio::null())
126            .spawn()
127            .map_err(|e| IfaError::Custom(format!("Spawn error: {}", e)))?;
128        Ok(child.id())
129    }
130
131    /// Get environment variable
132    pub fn ayika(&self, key: &str) -> Option<String> {
133        std::env::var(key).ok()
134    }
135}

137#[cfg(test)]
138mod tests {
139    use super::*;
140
141    #[test]
142    fn test_list_ops() {
143        let ogunda = Ogunda;
144        let mut list = vec![1, 2, 3];
145
146        ogunda.fi(&mut list, 4);
147        assert_eq!(list, vec![1, 2, 3, 4]);
148
149        assert_eq!(ogunda.mu(&mut list), Some(4));
150        assert_eq!(ogunda.iwon(&list), 3);
151    }
152
153    #[test]
154    fn test_sort() {
155        let ogunda = Ogunda;
156        let list = vec![3, 1, 4, 1, 5, 9, 2, 6];
157        let sorted = ogunda.to(&list);
158        assert_eq!(sorted, vec![1, 1, 2, 3, 4, 5, 6, 9]);
159    }
160
161    #[test]
162    fn test_filter() {
163        let ogunda = Ogunda;
164        let list = vec![1, 2, 3, 4, 5, 6];
165        let evens = ogunda.yan(&list, |x| x % 2 == 0);
166        assert_eq!(evens, vec![2, 4, 6]);
167    }
