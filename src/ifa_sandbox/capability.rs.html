1use serde::{Deserialize, Serialize};
2use std::path::PathBuf;
3
4/// Ọ̀fún Capability Definition
5#[derive(Debug, Clone, Hash, Eq, PartialEq, Serialize, Deserialize)]
6pub enum Ofun {
7    /// Read access to filesystem path
8    ReadFiles { root: PathBuf },
9    /// Write access to filesystem path
10    WriteFiles { root: PathBuf },
11    /// Network access to specific domains
12    Network { domains: Vec<String> },
13    /// Execute subprocesses
14    Execute { programs: Vec<String> },
15    /// Access environment variables
16    Environment { keys: Vec<String> },
17    /// High-resolution time
18    Time,
19    /// Random number generation
20    Random,
21    /// Standard I/O (stdin/stdout/stderr)
22    Stdio,
23}
24
25/// A set of granted capabilities
26#[derive(Debug, Clone, Default)]
27pub struct CapabilitySet {
28    capabilities: Vec<Ofun>,
29    violations: Vec<CapabilityViolation>,
30}
31
32#[derive(Debug, Clone)]
33pub struct CapabilityViolation {
34    pub capability: Ofun,
35    pub call_site: String,
36    pub timestamp: String,
37}
38
39impl CapabilitySet {
40    pub fn new() -> Self {
41        Self::default()
42    }
43
44    pub fn grant(&mut self, cap: Ofun) {
45        self.capabilities.push(cap);
46    }
47
48    /// Check if an operation is allowed
49    pub fn check(&self, required: &Ofun) -> bool {
50        self.capabilities
51            .iter()
52            .any(|granted| match (granted, required) {
53                (Ofun::ReadFiles { root: g }, Ofun::ReadFiles { root: r }) => r.starts_with(g),
54                (Ofun::WriteFiles { root: g }, Ofun::WriteFiles { root: r }) => r.starts_with(g),
55                (Ofun::Network { domains: g }, Ofun::Network { domains: r }) => {
56                    // Simple exact match for now, could add globbing
57                    r.iter().all(|d| g.contains(d))
58                }
59                (Ofun::Environment { keys: g }, Ofun::Environment { keys: r }) => {
60                    r.iter().all(|k| g.contains(k))
61                }
62                (Ofun::Execute { programs: g }, Ofun::Execute { programs: r }) => {
63                    r.iter().all(|p| g.contains(p))
64                }
65                (Ofun::Time, Ofun::Time) => true,
66                (Ofun::Random, Ofun::Random) => true,
67                (Ofun::Stdio, Ofun::Stdio) => true,
68                _ => false,
69            })
70    }
71
72    /// Get all granted capabilities
73    pub fn all(&self) -> &[Ofun] {
74        &self.capabilities
75    }
76
77    /// Get recorded violations (for audit/debugging)
78    pub fn violations(&self) -> &[CapabilityViolation] {
79        &self.violations
80    }
81
82    /// Record a capability violation
83    pub fn record_violation(&mut self, cap: Ofun, call_site: &str) {
84        self.violations.push(CapabilityViolation {
85            capability: cap,
86            call_site: call_site.to_string(),
87            timestamp: format!("{:?}", std::time::SystemTime::now()),
88        });
89    }
90}
