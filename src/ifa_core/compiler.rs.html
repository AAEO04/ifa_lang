1//! # Bytecode Compiler
2//!
3//! Compiles AST to bytecode for the If√°-Lang VM.

5use crate::ast::*;
6use crate::bytecode::{Bytecode, OpCode};
7use crate::error::IfaResult;
8use crate::lexer::OduDomain;
9use std::collections::HashMap;

11/// Bytecode compiler - transforms AST to executable bytecode
12pub struct Compiler {
13    bytecode: Bytecode,
14    /// Local variables: name -> stack slot
15    locals: Vec<HashMap<String, usize>>,
16    /// Current scope depth
17    scope_depth: usize,
18    /// Label counter for jumps
19    _label_counter: usize,
20}
21
22impl Compiler {
23    pub fn new(source_name: &str) -> Self {
24        Compiler {
25            bytecode: Bytecode::new(source_name),
26            locals: vec![HashMap::new()],
27            scope_depth: 0,
28            _label_counter: 0,
29        }
30    }
31
32    /// Compile a program to bytecode
33    pub fn compile(mut self, program: &Program) -> IfaResult<Bytecode> {
34        for stmt in &program.statements {
35            self.compile_statement(stmt)?;
36        }
37        self.emit(OpCode::Halt);
38        Ok(self.bytecode)
39    }
40
41    fn emit(&mut self, op: OpCode) {
42        self.bytecode.code.push(op as u8);
43    }
44
45    fn emit_byte(&mut self, byte: u8) {
46        self.bytecode.code.push(byte);
47    }
48
49    fn emit_i64(&mut self, value: i64) {
50        self.bytecode.code.extend_from_slice(&value.to_le_bytes());
51    }
52
53    fn emit_f64(&mut self, value: f64) {
54        self.bytecode.code.extend_from_slice(&value.to_le_bytes());
55    }
56
57    fn emit_string(&mut self, s: &str) {
58        let bytes = s.as_bytes();
59        self.emit_byte(bytes.len() as u8);
60        self.bytecode.code.extend_from_slice(bytes);
61    }
62
63    fn current_offset(&self) -> usize {
64        self.bytecode.code.len()
65    }
66
67    fn emit_jump(&mut self, op: OpCode) -> usize {
68        self.emit(op);
69        let offset = self.current_offset();
70        // Placeholder for 16-bit offset
71        self.emit_byte(0);
72        self.emit_byte(0);
73        offset
74    }
75
76    fn patch_jump(&mut self, offset: usize) {
77        let jump = (self.current_offset() - offset - 2) as u16;
78        self.bytecode.code[offset] = (jump & 0xff) as u8;
79        self.bytecode.code[offset + 1] = ((jump >> 8) & 0xff) as u8;
80    }
81
82    fn begin_scope(&mut self) {
83        self.scope_depth += 1;
84        self.locals.push(HashMap::new());
85    }
86
87    fn end_scope(&mut self) {
88        self.scope_depth -= 1;
89        self.locals.pop();
90    }
91
92    fn declare_local(&mut self, name: &str) -> usize {
93        let slot = self.locals.iter().map(|m| m.len()).sum();
94        if let Some(scope) = self.locals.last_mut() {
95            scope.insert(name.to_string(), slot);
96        }
97        slot
98    }
99
100    fn resolve_local(&self, name: &str) -> Option<usize> {
101        for scope in self.locals.iter().rev() {
102            if let Some(&slot) = scope.get(name) {
103                return Some(slot);
104            }
105        }
106        None
107    }
108
109    fn compile_statement(&mut self, stmt: &Statement) -> IfaResult<()> {
110        match stmt {
111            Statement::VarDecl { name, value, .. } => {
112                self.compile_expression(value)?;
113                let slot = self.declare_local(name);
114                self.emit(OpCode::StoreLocal);
115                self.emit_byte(slot as u8);
116            }
117
118            Statement::Assignment { target, value, .. } => {
119                self.compile_expression(value)?;
120                match target {
121                    AssignTarget::Variable(name) => {
122                        if let Some(slot) = self.resolve_local(name) {
123                            self.emit(OpCode::StoreLocal);
124                            self.emit_byte(slot as u8);
125                        } else {
126                            self.emit(OpCode::StoreGlobal);
127                            self.emit_string(name);
128                        }
129                    }
130                    AssignTarget::Index { name, index } => {
131                        // Push container, index, value
132                        if let Some(slot) = self.resolve_local(name) {
133                            self.emit(OpCode::LoadLocal);
134                            self.emit_byte(slot as u8);
135                        } else {
136                            self.emit(OpCode::LoadGlobal);
137                            self.emit_string(name);
138                        }
139                        self.compile_expression(index)?;
140                        // Swap so stack is: value, container, index
141                        // Then call SetIndex
142                        self.emit(OpCode::SetIndex);
143                    }
144                }
145            }
146
147            Statement::Instruction { call, .. } => {
148                self.compile_odu_call(call)?;
149                self.emit(OpCode::Pop); // Discard result
150            }
151
152            Statement::If {
153                condition,
154                then_body,
155                else_body,
156                ..
157            } => {
158                self.compile_expression(condition)?;
159                let else_jump = self.emit_jump(OpCode::JumpIfFalse);
160                self.emit(OpCode::Pop); // Pop condition
161
162                self.begin_scope();
163                for s in then_body {
164                    self.compile_statement(s)?;
165                }
166                self.end_scope();
167
168                if let Some(else_stmts) = else_body {
169                    let end_jump = self.emit_jump(OpCode::Jump);
170                    self.patch_jump(else_jump);
171                    self.emit(OpCode::Pop); // Pop condition
172
173                    self.begin_scope();
174                    for s in else_stmts {
175                        self.compile_statement(s)?;
176                    }
177                    self.end_scope();
178                    self.patch_jump(end_jump);
179                } else {
180                    self.patch_jump(else_jump);
181                }
182            }
183
184            Statement::While {
185                condition, body, ..
186            } => {
187                let loop_start = self.current_offset();
188
189                self.compile_expression(condition)?;
190                let exit_jump = self.emit_jump(OpCode::JumpIfFalse);
191                self.emit(OpCode::Pop); // Pop condition
192
193                self.begin_scope();
194                for s in body {
195                    self.compile_statement(s)?;
196                }
197                self.end_scope();
198
199                // Jump back to start
200                self.emit(OpCode::Jump);
201                let back_offset = (self.current_offset() - loop_start + 2) as i16;
202                self.emit_byte((-back_offset as u16 & 0xff) as u8);
203                self.emit_byte(((-back_offset as u16 >> 8) & 0xff) as u8);
204
205                self.patch_jump(exit_jump);
206                self.emit(OpCode::Pop); // Pop condition
207            }
208
209            Statement::For {
210                var: _var,
211                iterable,
212                body,
213                ..
214            } => {
215                // Compile iterable
216                self.compile_expression(iterable)?;
217
218                // Create iterator - push index 0
219                self.emit(OpCode::PushInt);
220                self.emit_i64(0);
221
222                let loop_start = self.current_offset();
223
224                // Check if index < len
225                self.emit(OpCode::Dup); // Dup index
226                // Get length - this is simplified, real impl needs GetLen opcode
227
228                // For now, emit a placeholder
229                self.emit(OpCode::Pop); // Simplified
230
231                for s in body {
232                    self.compile_statement(s)?;
233                }
234
235                // Increment index and loop
236                self.emit(OpCode::PushInt);
237                self.emit_i64(1);
238                self.emit(OpCode::Add);
239
240                // Jump back
241                let _ = loop_start; // TODO: proper for loop compilation
242            }
243
244            Statement::Return { value, .. } => {
245                if let Some(expr) = value {
246                    self.compile_expression(expr)?;
247                } else {
248                    self.emit(OpCode::PushNull);
249                }
250                self.emit(OpCode::Return);
251            }
252
253            Statement::Ase { .. } => {
254                self.emit(OpCode::Halt);
255            }
256
257            Statement::Expr { expr, .. } => {
258                self.compile_expression(expr)?;
259                self.emit(OpCode::Pop);
260            }
261
262            Statement::EseDef {
263                name, params, body, ..
264            } => {
265                // Functions are compiled as separate bytecode chunks
266                // For now, store as a constant and skip
267                let _ = (name, params, body);
268                // TODO: Implement function compilation
269            }
270
271            Statement::OduDef { name, .. } => {
272                // Class definitions
273                let _ = name;
274                // TODO: Implement class compilation
275            }
276
277            Statement::Import { path, .. } => {
278                // Import handling
279                let _ = path;
280                // TODO: Implement import compilation
281            }
282
283            Statement::Taboo { source, target, .. } => {
284                // Taboo is a compile-time directive, no bytecode emitted
285                // Could be stored in metadata for later validation
286                let _ = (source, target);
287            }
288
289            Statement::Ewo {
290                condition, message, ..
291            } => {
292                // Compile the condition expression
293                self.compile_expression(condition)?;
294                // Note: Assertion is verified at bytecode interpretation time
295                // For now, we just compile the condition check
296                let _ = message;
297            }
298
299            Statement::Opon { size, .. } => {
300                // Opon is a compile-time directive, no bytecode emitted
301                // Memory size is configured at VM initialization
302                let _ = size;
303            }
304        }
305        Ok(())
306    }
307
308    fn compile_expression(&mut self, expr: &Expression) -> IfaResult<()> {
309        match expr {
310            Expression::Int(n) => {
311                self.emit(OpCode::PushInt);
312                self.emit_i64(*n);
313            }
314
315            Expression::Float(f) => {
316                self.emit(OpCode::PushFloat);
317                self.emit_f64(*f);
318            }
319
320            Expression::String(s) => {
321                self.emit(OpCode::PushStr);
322                self.emit_string(s);
323            }
324
325            Expression::Bool(b) => {
326                self.emit(if *b {
327                    OpCode::PushTrue
328                } else {
329                    OpCode::PushFalse
330                });
331            }
332
333            Expression::Nil => {
334                self.emit(OpCode::PushNull);
335            }
336
337            Expression::Identifier(name) => {
338                if let Some(slot) = self.resolve_local(name) {
339                    self.emit(OpCode::LoadLocal);
340                    self.emit_byte(slot as u8);
341                } else {
342                    self.emit(OpCode::LoadGlobal);
343                    self.emit_string(name);
344                }
345            }
346
347            Expression::BinaryOp { left, op, right } => {
348                self.compile_expression(left)?;
349                self.compile_expression(right)?;
350
351                let opcode = match op {
352                    BinaryOperator::Add => OpCode::Add,
353                    BinaryOperator::Sub => OpCode::Sub,
354                    BinaryOperator::Mul => OpCode::Mul,
355                    BinaryOperator::Div => OpCode::Div,
356                    BinaryOperator::Mod => OpCode::Mod,
357                    BinaryOperator::Eq => OpCode::Eq,
358                    BinaryOperator::NotEq => OpCode::Ne,
359                    BinaryOperator::Lt => OpCode::Lt,
360                    BinaryOperator::LtEq => OpCode::Le,
361                    BinaryOperator::Gt => OpCode::Gt,
362                    BinaryOperator::GtEq => OpCode::Ge,
363                    BinaryOperator::And => OpCode::And,
364                    BinaryOperator::Or => OpCode::Or,
365                };
366                self.emit(opcode);
367            }
368
369            Expression::UnaryOp { op, expr } => {
370                self.compile_expression(expr)?;
371                match op {
372                    UnaryOperator::Neg => self.emit(OpCode::Neg),
373                    UnaryOperator::Not => self.emit(OpCode::Not),
374                }
375            }
376
377            Expression::List(items) => {
378                for item in items {
379                    self.compile_expression(item)?;
380                }
381                self.emit(OpCode::BuildList);
382                self.emit_byte(items.len() as u8);
383            }
384
385            Expression::Map(entries) => {
386                for (key, value) in entries {
387                    self.compile_expression(key)?;
388                    self.compile_expression(value)?;
389                }
390                self.emit(OpCode::BuildMap);
391                self.emit_byte(entries.len() as u8);
392            }
393
394            Expression::Index { object, index } => {
395                self.compile_expression(object)?;
396                self.compile_expression(index)?;
397                self.emit(OpCode::GetIndex);
398            }
399
400            Expression::OduCall(call) => {
401                self.compile_odu_call(call)?;
402            }
403
404            Expression::Call { name, args } => {
405                // Push function
406                if let Some(slot) = self.resolve_local(name) {
407                    self.emit(OpCode::LoadLocal);
408                    self.emit_byte(slot as u8);
409                } else {
410                    self.emit(OpCode::LoadGlobal);
411                    self.emit_string(name);
412                }
413
414                // Push arguments
415                for arg in args {
416                    self.compile_expression(arg)?;
417                }
418
419                self.emit(OpCode::Call);
420                self.emit_byte(args.len() as u8);
421            }
422
423            Expression::MethodCall {
424                object,
425                method,
426                args,
427            } => {
428                self.compile_expression(object)?;
429                for arg in args {
430                    self.compile_expression(arg)?;
431                }
432                self.emit(OpCode::CallMethod);
433                self.emit_string(method);
434                self.emit_byte(args.len() as u8);
435            }
436        }
437        Ok(())
438    }
439
440    fn compile_odu_call(&mut self, call: &OduCall) -> IfaResult<()> {
441        // Push arguments
442        for arg in &call.args {
443            self.compile_expression(arg)?;
444        }
445
446        // Emit OduCall opcode with domain and method
447        self.emit(OpCode::CallOdu);
448        self.emit_byte(domain_to_byte(&call.domain));
449        self.emit_string(&call.method);
450        self.emit_byte(call.args.len() as u8);
451
452        Ok(())
453    }
454}

456/// Convert OduDomain to byte
457fn domain_to_byte(domain: &OduDomain) -> u8 {
458    match domain {
459        OduDomain::Ogbe => 0,
460        OduDomain::Oyeku => 1,
461        OduDomain::Iwori => 2,
462        OduDomain::Odi => 3,
463        OduDomain::Irosu => 4,
464        OduDomain::Owonrin => 5,
465        OduDomain::Obara => 6,
466        OduDomain::Okanran => 7,
467        OduDomain::Ogunda => 8,
468        OduDomain::Osa => 9,
469        OduDomain::Ika => 10,
470        OduDomain::Oturupon => 11,
471        OduDomain::Otura => 12,
472        OduDomain::Irete => 13,
473        OduDomain::Ose => 14,
474        OduDomain::Ofun => 15,
475        OduDomain::Coop => 16,  // Pseudo-domain for FFI
476        OduDomain::Opele => 17, // Pseudo-domain for Divination
477    }

480/// Compile source code to bytecode
481pub fn compile(source: &str) -> IfaResult<Bytecode> {
482    let program = crate::parser::parse(source)?;
483    let compiler = Compiler::new("<main>");
484    compiler.compile(&program)
485}
