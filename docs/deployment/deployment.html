Deployment Targets

Multi-Platform Runtime Architecture

Overview

Ifá-Lang supports multiple deployment targets through a unified codebase with platform-specific optimizations. The three-tier runtime architecture (AST Interpreter, Bytecode VM, Transpiler) adapts to different environments while maintaining language consistency.

Native Deployment (ifa-cli)

Full-Featured Runtime

The native CLI provides the complete Ifá-Lang experience with all features enabled:

Core Features

  • Complete 16 Odù domains
  • Three-tier runtime
  • Full standard library
  • REPL and LSP integration

Performance

  • JIT compilation
  • Native code generation
  • Optimized VM
  • Multi-threading support

Development Tools

  • Babalawo LSP server
  • Self-updating installer
  • Debugging support
  • Profiling tools

Embedded Configuration

            

                

                    Embedded Setup
                

                
// Cargo.toml for embedded target
[dependencies]
ifa-embedded = { version = "1.3.0", default-features = false }
heapless = "0.8"
cortex-m = "0.7"

// Memory configuration
const HEAP_SIZE: usize = 1024;  // 1KB heap
const STACK_SIZE: usize = 2048; // 2KB stack

// Bounded collections
use heapless::{Vec, String};

fn embedded_main() -> ! {
    // Initialize embedded runtime
    let runtime = IfaEmbedded::new()
        .with_heap_size(HEAP_SIZE)
        .with_stack_size(STACK_SIZE)
        .with_gpio_pins(13);  // LED pin
    
    // Run Ifá-Lang code
    runtime.run(`
        opon: kekere;
        Irosu.fo("Embedded Ifá-Lang!");
        
        // Blink LED
        nigita otito {
            Ogbe.gpio_set(13, otito);
            Iwori.delay(1000);
            Ogbe.gpio_set(13, eke);
            Iwori.delay(1000);
        }
    `);
    
    loop {}
}

Sandbox Configuration

            

                

                    Sandbox Setup
                

                
// Create sandbox instance
let sandbox = IfaSandbox::new()
    .with_memory_limit(10 * 1024 * 1024)  // 10MB
    .with_time_limit(Duration::from_secs(30))
    .with_capabilities(&[
        Capability::Console,
        Capability::Math,
        Capability::Strings,
    ])
    .deny_capabilities(&[
        Capability::FileSystem,
        Capability::Network,
        Capability::NativeFFI,
    ]);

// Execute untrusted code
let result = sandbox.execute(`
    // This code runs safely
    Irosu.fo("Hello from sandbox!");
    ayanmo result = Obara.fikun(5, 3);
    pada result;
    
    // This would be blocked
    // Odi.si("sensitive_file.txt");  // Denied!
    // Otura.so("evil.com", 4444);   // Denied!
`);

Conditional Compilation

            

                

                    Platform-Specific Code
                

                
// Platform-specific implementations
#[cfg(feature = "native")]
mod native_runtime {
    pub fn create_runtime() -> NativeRuntime {
        NativeRuntime::new()
            .with_full_domains()
            .with_gpu_support()
            .with_audio_support()
    }
}

#[cfg(feature = "wasm")]
mod wasm_runtime {
    pub fn create_runtime() -> WasmRuntime {
        WasmRuntime::new()
            .with_core_domains()
            .with_webgpu_support()
    }
}

#[cfg(feature = "embedded")]
mod embedded_runtime {
    pub fn create_runtime() -> EmbeddedRuntime {
        EmbeddedRuntime::new()
            .with_minimal_domains()
            .with_bounded_memory(1024)
    }
}

// Unified interface
pub trait IfaRuntime {
    fn run(&self, code: &str) -> Result;
    fn get_domains(&self) -> &[OduDomain];
}

CI/CD Pipeline

            

                

                    GitHub Actions
                

                
# .github/workflows/build.yml
name: Build and Deploy

on: [push, pull_request]

jobs:
  build:
    strategy:
      matrix:
        target: [native, wasm, embedded]
        include:
          - target: native
            os: ubuntu-latest
            features: full-domains
          - target: wasm
            os: ubuntu-latest
            features: core-domains
          - target: embedded
            os: ubuntu-latest
            features: core-domains
    
    runs-on: ${{ matrix.os }}
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Install Rust
      uses: actions-rs/toolchain@v1
      with:
        toolchain: stable
        target: ${{ matrix.target }}
    
    - name: Build
      run: |
        cargo build --target ${{ matrix.target }} --features ${{ matrix.features }}
    
    - name: Test
      run: |
        cargo test --target ${{ matrix.target }} --features ${{ matrix.features }}
    
    - name: Deploy
      if: github.ref == 'refs/heads/main'
      run: |
        # Deploy to appropriate platform
        ./scripts/deploy.sh ${{ matrix.target }}

// Runtime metrics collection
let metrics = runtime.get_metrics();

// Domain-specific metrics
