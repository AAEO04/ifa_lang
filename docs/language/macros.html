Available Macros

#[derive(Ebo)]

Auto-implement Drop for RAII resource cleanup.

// Rust usage
#[derive(Ebo)]
#[ebo(cleanup = "close")]  // Optional: specify cleanup method
struct MyFile {
    handle: std::fs::File,
}

// Drop is auto-implemented calling self.close()

#[iwa_pele]

Compile-time balance checking for paired operations.

// Ensures open/close pairs match
#[iwa_pele]
fn network_task() {
    let conn = Otura.so("example.com", 80);  // open
    // ... use connection ...
    Otura.pa(conn);  // close - required!
}

// Pairs checked:
// - so / pa (open/close)
// - si / ti (write open/close)
// - mu / fi (acquire/release)
// - bere / da (start/stop)

ajose!

Create reactive bindings between values.

// Reactive binding declaration
ajose!(counter.value => label.text);

// When counter.value changes, label.text updates automatically

ebo_block!

Scoped RAII block with guaranteed cleanup.

ebo_block! {
    let file = std::fs::File::open("data.txt")?;
    // ... use file ...
    // file auto-closed when block exits (even on error)
}

#[derive(Observable)]

Make struct fields observable for reactive updates.

#[derive(Observable)]
struct Counter {
    value: i32,
}

// Generates Counter::watch_value() method
let counter = Counter { value: 0 };
counter.watch_value(|new_val| {
    println!("Value changed to: {}", new_val);
});

Usage in Rust

// Add to Cargo.toml
[dependencies]
ifa-macros = "1.2"

// Import in Rust code
use ifa_macros::{Ebo, iwa_pele, ajose, ebo_block};

How Ìwà P?`l?´ Works

The #[iwa_pele] macro performs AST analysis to count method calls:

  1. Parses the function body into AST
  2. Counts calls to paired methods (so/pa, mu/fi, etc.)
  3. Emits compile error if counts don't match
