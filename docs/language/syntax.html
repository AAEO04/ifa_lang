Syntax Reference

Complete Grammar and Language Structure

Program Structure

An Ifá-Lang program is a sequence of statements, from top to bottom execution:

Basic Program 
// Import statements
iba std.irosu;
iba { fo, ka } from std.irosu;

// Constants
const PI = 3.14159;

// Variables
ayanmo name = "Ifá-Lang";

// Functions
ese greet(who) {
    Irosu.fo("Kaabo, " + who + "!");
}

// Main execution
greet("World");

Statements

Variable Declaration

Variables 
// Yoruba keywords
ayanmo x = 5;                    // Mutable variable
const NAME = "Ifá";              // Constant

// English keywords  
let x = 5;                       // Mutable variable
const NAME = "Ifá";              // Constant

// With type hints
ayanmo age: Int = 25;
let name: Str = "Alice";

Assignment

Assignments 
x = 10;                         // Simple assignment
*ptr = 20;                       // Pointer dereference
arr[0] = 30;                     // Array indexing

Function Definition

Functions 
// Yoruba
ese add(a, b) {
    pada a + b;
}

// Async function
ese fetch_data() async {
    // Async operations here
}

// English
fn add(a, b) {
    return a + b;
}

// Public function
public ese calculate(x) {
    // Accessible from other modules
}

Control Flow

If Statement

Conditional Logic 
// Yoruba
bi x > 0 {
    Irosu.fo("Positive");
} bi kosi {
    Irosu.fo("Non-positive");
}

// English
if x > 0 {
    Irosu.fo("Positive");
} else {
    Irosu.fo("Non-positive");
}

Loops

Iteration 
// While loop
nigita x < 10 {
    x = x + 1;
}

// For loop
fun i ninu 0..10 {
    Irosu.fo(i);
}

// English equivalents
while x < 10 {
    x = x + 1;
}

for i in 0..10 {
    Irosu.fo(i);
}

Match Statement

Pattern Matching 
match (x) {
    0 => Irosu.fo("Zero"),
    1..10 => Irosu.fo("Small"),
    _ => Irosu.fo("Large")
}

Error Handling

Try-Catch-Finally

Exception Handling 
// Yoruba
gbiyanju {
    risky_operation();
} gba (error) {
    Irosu.fo("Error: " + error.message);
} nipari {
    cleanup();
}

// English
try {
    risky_operation();
} catch (error) {
    Irosu.fo("Error: " + error.message);
} finally {
    cleanup();
}

Assertions

Debug Assertions 
// Yoruba
ewo x >= 0, "x must be non-negative";
ewo balance >= 0;

// English
assert x >= 0, "x must be non-negative";
verify balance >= 0;

Imports and Modules

When You Need Imports

NOTE: The 16 Odù domains (Ogbe, Irosu, Obara, etc.) are built-in and do NOT need imports. They're fundamental language primitives.

Import Statements

Use imports for external modules, utilities, or specific functions:

Module Imports 
// Import external utilities
iba std.utils;
import std.utils;

// Import specific functions from modules
iba { fo, ka } from std.irosu;  // Specific Irosu functions
import { println, read } from std.irosu;

// Import from sub-modules
iba std.crypto.sha256;
import std.crypto.sha256;

// Import third-party packages
iba external_crate;
import external_crate;

What DOESN'T Need Imports

Built-in Domains (No Imports) 
// These are WRONG - don't import the 16 Odù domains!
// iba std.irosu;    // NO! Irosu is built-in
// iba std.obara;    // NO! Obara is built-in  
// import std.ogbe;   // NO! Ogbe is built-in

// These are CORRECT - use domains directly
ese main() {
    Irosu.fo("Hello");     // Built-in, no import needed
    ayanmo sum = Obara.fikun(1, 2);  // Built-in, no import needed
    Ogbe.jade(0);          // Built-in, no import needed
}

Domain Calls

Call methods on the 16 Odù domains - NO IMPORTS NEEDED:

Domain Operations 
// The 16 Odù domains are built-in and globally available
// No imports required - they're fundamental language primitives

// Console I/O (Irosu - 0011)
Irosu.fo("Hello, World!");
ayanmo input = Irosu.ka("Enter name: ");

// Mathematics (Obara - 1000)
ayanmo result = Obara.fikun(10, 20);
ayanmo sqrt = Obara.akako(16);

// Cryptography (Irete - 1101)
ayanmo hash = Irete.sha256("data");
ayanmo encoded = Irete.base64("data");

// Time operations (Iwori - 0110)
ayanmo now = Iwori.bayi();
ayanmo formatted = Iwori.ohun(now);

// System operations (Ogbe - 1111)
ayanmo args = Ogbe.args();
Ogbe.jade(0);

// And all other 16 Odù domains work the same way

Memory Management

Opon Directives

Memory Configuration 
// Memory size directives
opon: kekere;              // Small memory
#opon arinrin;               // Medium memory  
mem: nla;                   // Large memory
#mem ailopin;               // Unlimited memory

Resource Management ( )

Resource Sacrifice 
// Resource acquisition with obligation
ebo "server_config.json";

// File handling with automatic cleanup
ebo file = Odi.open("data.txt");
// Use file...
// File automatically closed when scope ends

Unsafe Code

For low-level operations when needed:

Unsafe Blocks 
// Yoruba
ailewu {
    // Unsafe operations here
    // Direct memory manipulation
    // FFI calls
}

// English
unsafe {
    // Unsafe operations here
}

Generators

Function Return

Program Termination 
// Yoruba
ase;

Type System

Basic Types

Primitive Types 
// Numbers
ayanmo i: Int = 42;
ayanmo f: Float = 3.14;
ayanmo u8: u8 = 255;
ayanmo i64: i64 = 1000000;

// Text
ayanmo s: Str = "Hello";
ayanmo c: char = 'A';

// Boolean
ayanmo b: Bool = otito;

// Collections
ayanmo list: List[Int] = [1, 2, 3];
ayanmo map: Map[Str, Int] = {"a": 1, "b": 2};

// Universal
ayanmo anything: Any = 42;

Type Hints

Type Annotations 
// Function parameters with types
ese add(a: Int, b: Int) -> Int {
    pada a + b;
}

// Variable with type hint
ayanmo name: Str = "Alice";

// Complex types
ayanmo matrix: List[List[Float]] = [[1.0, 2.0], [3.0, 4.0]];

Taboo Declarations

Declare architectural constraints:

Taboo Rules 
// Taboo: Graphics operations in error context
èèwò: Ose -> Odi;

// Taboo: Network operations in crypto context  
taboo: Otura -> Irete;

// This means certain domain combinations are forbidden
// to prevent architectural violations
