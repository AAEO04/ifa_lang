Grammar Reference

Complete Language Grammar Specification

Overview

Ifá-Lang uses a Pest grammar definition for parsing. The grammar supports bilingual keywords (Yoruba/English) and implements the full language specification including the 16 Odù domains.

Program Structure

Top-Level Structure

Program Grammar 
program = { SOI ~ statement* ~ EOI }

// A program is zero or more statements
statement = {
    import_stmt
    | from_import_stmt
    | opon_stmt
    | ebo_stmt
    | ailewu_stmt
    | var_decl
    | const_stmt
    | assignment_stmt
    | odu_def
    | ese_def
    | if_stmt
    | while_stmt
    | for_stmt
    | return_stmt
    | taboo_stmt
    | ewo_stmt
    | instruction
    | ase_stmt
    | match_stmt
    | try_stmt
    | yield_stmt
    | COMMENT
}

Statements

Variable Declarations

Variable Grammar 
// Variable declaration
var_decl = { public_mod? ~ var_kw ~ ident ~ type_hint? ~ "=" ~ expression ~ ";" }

// Type hint
type_hint = { ":" ~ type_name }

// Variable keywords
var_kw = { "ayanmo" | "let" }

// Public modifier
public_mod = { "public" }

// Examples:
ayanmo x = 5;
public let name: Str = "Alice";

Constants

Constant Grammar 
// Constant declaration
const_stmt = { public_mod? ~ const_kw ~ ident ~ "=" ~ expression ~ ";" }

// Constant keywords
const_kw = { "const" | "loruko" | "ka" | "ayanfe" | "àyànfê" }

// Examples:
const PI = 3.14159;
public loruko VERSION = "1.3.0";

Assignments

Assignment Grammar 
// Assignment statement
assignment_stmt = { lvalue ~ "=" ~ expression ~ ";" }

// Left-hand values
lvalue = { deref_lvalue | index_lvalue | ident }
deref_lvalue = { "*" ~ expression }        // *ptr
index_lvalue = { ident ~ "[" ~ expression ~ "]" }  // arr[0]

// Examples:
x = 10;
*ptr = 20;
arr[0] = 30;

Functions and Classes

Function Definitions

Function Grammar 
// Function definition
ese_def = { public_mod? ~ async_mod? ~ ese_kw ~ ident ~ "(" ~ params? ~ ")" ~ "{" ~ statement* ~ "}" }

// Parameters
params = { param ~ ("," ~ param)* }
param = { ident ~ (":" ~ type_name)? }

// Function keywords
ese_kw = { "ese" | "fn" }
async_mod = { "async" }

// Examples:
ese add(a, b) { pada a + b; }
public async fetch(url: Str) { ... }
fn greet(name: Str) -> Str { ... }

Odù Definitions

Odù Grammar 
// Odù (class) definition
odu_def = { public_mod? ~ odu_kw ~ ident ~ "{" ~ odu_body ~ "}" }

odu_body = { (ese_def | var_decl)* }

// Odù keyword
odu_kw = { "odu" | "class" }

// Examples:
odu Server {
    ayanmo host: Str;
    ese start() { ... }
}

class Database {
    let connection: Resource;
    fn query(sql: Str) { ... }
}

Control Flow

If Statements

If Grammar 
// If statement
if_stmt = { if_kw ~ expression ~ "{" ~ statement* ~ "}" ~ else_clause? }

// Else clause
else_clause = { else_kw ~ "{" ~ statement* ~ "}" }

// Keywords
if_kw = { "bi" | "if" }
else_kw = { "bi kosi" | "else" }

// Examples:
bi x > 0 {
    Irosu.fo("Positive");
} bi kosi {
    Irosu.fo("Non-positive");
}

if condition {
    // then block
} else {
    // else block
}

Loops

Loop Grammar 
// While loop
while_stmt = { while_kw ~ expression ~ "{" ~ statement* ~ "}" }

// For loop
for_stmt = { for_kw ~ ident ~ in_kw ~ expression ~ "{" ~ statement* ~ "}" }

// Keywords
while_kw = { "nigita" | "while" }
for_kw = { "fun" | "for" }
in_kw = { "ninu" | "in" }

// Examples:
nigita x < 10 {
    x = x + 1;
}

fun i ninu 0..10 {
    Irosu.fo(i);
}

while condition {
    // loop body
}

for item in collection {
    // loop body
}

Match Statements

Match Grammar 
// Match statement
match_stmt = { match_kw ~ "(" ~ expression ~ ")" ~ "{" ~ match_arm* ~ "}" }

// Match arms
match_arm = { match_pattern ~ "=>" ~ (statement | ("{" ~ statement* ~ "}")) }

// Patterns
match_pattern = { range_pattern | literal_pattern | wildcard_pattern }
range_pattern = { expression ~ ".." ~ expression }
literal_pattern = { expression }
wildcard_pattern = { "_" }

// Keywords
match_kw = { "match" }

// Examples:
match (x) {
    0 => Irosu.fo("Zero"),
    1..10 => Irosu.fo("Small"),
    _ => Irosu.fo("Large")
}

Imports and Modules

Import Statements

Import Grammar 
// Import statement
import_stmt = { import_kw ~ module_path ~ ";" }

// Named import
from_import_stmt = { import_kw ~ "{" ~ ident ~ ("," ~ ident)* ~ "}" ~ from_kw ~ module_path ~ ";" }

// Module path
module_path = { ident ~ ("." ~ ident)* }

// Keywords
import_kw = { "iba" | "import" }
from_kw = { "lati" | "from" }

// Examples:
iba std.irosu;
import std.utils;

iba { fo, ka } from std.irosu;
import { println, read } from std.irosu;

Error Handling

Try-Catch-Finally

Try Grammar 
// Try statement
try_stmt = { try_kw ~ "{" ~ statement* ~ "}" ~ catch_clause ~ nipari_clause? }

// Catch clause
catch_clause = { catch_kw ~ "(" ~ ident ~ ")" ~ "{" ~ statement* ~ "}" }

// Finally clause
nipari_clause = { nipari_kw ~ "{" ~ statement* ~ "}" }

// Keywords
try_kw = { "gbiyanju" | "gbìyànjú" | "try" }
catch_kw = { "gba" | "gbà" | "catch" }
nipari_kw = { "nipari" | "nípàrí" | "finally" }

// Examples:
gbiyanju {
    risky_operation();
} gba (error) {
    Irosu.fo("Error: " + error.message);
} nipari {
    cleanup();
}

try {
    // try block
} catch (error) {
    // catch block
} finally {
    // finally block
}

Assertions

Assert Grammar 
// Assertion statement
ewo_stmt = { ewo_kw ~ expression ~ ("," ~ string)? ~ ";" }

// Keywords
ewo_kw = { "ewo" | "ewo" | "assert" | "verify" }

// Examples:
ewo x >= 0, "x must be non-negative";
ewo balance >= 0;

assert x >= 0, "x must be non-negative";
verify balance >= 0;

Domain Calls

Odù Domain Instructions

Domain Grammar 
// Domain instruction
instruction = { odu_call ~ ";" }

// Odù call
odu_call = { odu_name ~ "." ~ ident ~ "(" ~ arguments? ~ ")" }

// Arguments
arguments = { expression ~ ("," ~ expression)* }

// Odù names (16 core domains)
odu_name = {
    "Ogbe" | "Oyeku" | "Iwori" | "Odi" | "Irosu" | "Owonrin"
    | "Obara" | "Okanran" | "Ogunda" | "Osa" | "Ika" | "Oturupon"
    | "Otura" | "Irete" | "Ose" | "Ofun"
}

// Examples:
Irosu.fo("Hello, World!");
Obara.fikun(10, 20);
Irete.sha256("data");
Ogbe.args();

Memory Management

Opon Directives

Opon Grammar 
// Opon (memory) directive
opon_stmt = { opon_kw ~ ":"? ~ opon_size ~ ";"? }

// Opon keywords
opon_kw = { "#opon" | "opon" | "#Opon" | "Opon" | "#mem" | "mem" }

// Memory sizes
opon_size = {
    "kekere" | "kékéré" | "small" | "tiny" | "embedded"
    | "arinrin" | "àrínrin" | "medium" | "standard" | "default"
    | "nla" | "nlá" | "large" | "big"
    | "ailopin" | "àìlópin" | "unlimited" | "dynamic" | "max"
}

// Examples:
opon: kekere;
#opon arinrin;
mem: nla;
#mem ailopin;

Resource Management ( )

Ebo Grammar 
// Ebo (offering/initiation) statement
ebo_stmt = { ebo_kw ~ expression ~ ";"? }

// Ebo keywords
ebo_kw = { "ebo" | "éb" | "sacrifice" }

// Examples:
ebo "server_config.json";
sacrifice file = Odi.open("data.txt");

Special Constructs

Unsafe Code

Unsafe Grammar 
// Ailewu (unsafe) block
ailewu_stmt = { ailewu_kw ~ "{" ~ statement* ~ "}" }

// Keywords
ailewu_kw = { "ailewu" | "àìléwu" | "unsafe" }

// Examples:
ailewu {
    // Unsafe operations here
    // Direct memory manipulation
    // FFI calls
}

    // Unsafe operations here
}

Yield and Return

Control Flow Grammar 
// Return statement
return_stmt = { return_kw ~ expression? ~ ";" }

// Yield statement
yield_stmt = { yield_kw ~ expression ~ ";" }

// End statement
ase_stmt = { ase_kw ~ ";"? }

// Keywords
return_kw = { "pada" | "return" }
yield_kw = { "jowo" | "jwó" | "yield" }
ase_kw = { "ase" | "end" }

// Examples:
pada result;
return 42;

jowo 1000;
yield 1000;

ase;
end;

Taboo Declarations

Taboo Grammar 
// Taboo declaration
taboo_stmt = { taboo_kw ~ ":" ~ odu_name ~ "->" ~ odu_name ~ ";" }

// Keywords
taboo_kw = { "èèw" | "ew" | "taboo" }

// Examples:
èèw: Ose -> Odi;
taboo: Otura -> Irete;

Expressions

Literals

Literal Grammar 
// Various literal types
expression = { literal | ... }

literal = {
    int_literal
    | float_literal
    | string_literal
    | char_literal
    | boolean_literal
    | list_literal
    | map_literal
}

// Integer literals
int_literal = { "-"? ~ ASCII_DIGIT+ }

// Float literals
float_literal = { "-"? ~ ASCII_DIGIT+ ~ "." ~ ASCII_DIGIT* }

// String literals
string_literal = { "\"" ~ (~"\"" ~ ANY)* ~ "\"" }

// Boolean literals
boolean_literal = { "otito" | "true" | "eke" | "false" }

// Examples:
42
-100
3.14159
-2.5
"Hello, World!"
'c'
otito / true
eke / false

Collections

Collection Grammar 
// List literal
list_literal = { "[" ~ expression ~ ("," ~ expression)* ~ "]" }

// Map literal
map_literal = {
    "{" ~ string_literal ~ ":" ~ expression ~ ("," ~ string_literal ~ ":" ~ expression)* ~ "}"
}

// Examples:
[1, 2, 3, 4, 5]
["hello", "world", 42, otito]

{
    "name": "Alice",
    "age": 30,
    "active": otito
}

Operators

Operator Grammar 
// Arithmetic operators (high precedence)
expression = { term ~ (("+" | "-") ~ term)* }
term = { factor ~ (("*" | "/" | "%") ~ factor)* }

// Comparison operators
comparison = { expression ~ (("==" | "!=" | "<" | ">" | "<=" | ">=") ~ expression) }

// Logical operators
logical = { comparison ~ (("&&" | "||") ~ comparison)* }

// Unary operators
factor = { ("!" | "-") ~ primary }

// Examples:
x + y * z
a == b && c != d
!condition
-value

Type Names

Supported Types

Type Grammar 
// Type names in type hints
type_name = {
    // Basic types
    "Int" | "Float" | "Str" | "Bool" | "List" | "Map" | "Any"
    | "int" | "float" | "str" | "bool" | "list" | "map" | "any"
    
    // Numeric types
    | "i8" | "i16" | "i32" | "i64"
    | "u8" | "u16" | "u32" | "u64"
    | "f32" | "f64"
    
    // Special types
    | "void"
}

// Examples:
ayanmo x: Int = 42;
ayanmo name: Str = "Alice";
ayanmo items: List[Int] = [1, 2, 3];
ayanmo data: Map[Str, Any] = {"key": "value"};

Comments

Comment Grammar

Comment Grammar 
// Comment patterns
COMMENT = {
    // Single line comment
    | "/*" ~ (!"*/" ~ ANY)* ~ "*/"  // Multi-line comment
}

// Examples:
// This is a single line comment

/*
   This is a
   multi-line comment
*/

/// Documentation comment

Grammar Implementation Notes

Pest Features Used

  • PEG Grammar: Parsing Expression Grammar for unambiguous parsing
  • Unicode Support: Full Unicode support for Yoruba characters
  • Error Recovery: Built-in error recovery for better error messages
  • Whitespace Handling: Automatic whitespace skipping between tokens

Bilingual Support

  • Yoruba Keywords: Full Yoruba keyword support with diacritics
  • English Keywords: English alternatives for accessibility
  • Mixed Usage: Can mix Yoruba and English in same program
  • Cultural Integration: Keywords reflect Yoruba philosophical concepts

Odù Domain Integration

  • 16 Core Domains: All traditional Odù domains supported
  • Method Calls: Domain.method() syntax for operations
  • Built-in Access: No imports required for core domains
  • Type Safety: Domain methods are type-checked

Performance Considerations

  • Left Recursion: Avoided for better parsing performance
  • Operator Precedence: Clear precedence rules for expressions
  • Lookahead: Minimal lookahead for efficient parsing
  • Error Messages: Contextual error messages for debugging
