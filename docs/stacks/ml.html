Features

Tensors

N-dimensional arrays with broadcasting

// Create tensor
ayanmo t = ML.tensor([[1, 2], [3, 4]]);

// Zeros/ones
ayanmo zeros = ML.zeros([3, 3]);
ayanmo ones = ML.ones([2, 2]);

Matrix Operations

Matrix multiply, transpose, reshape

// Matrix multiply
ayanmo result = ML.matmul(a, b);

// Transpose
ayanmo t = ML.transpose(matrix);

// Reshape
ayanmo reshaped = ML.reshape(tensor, [4, 2]);

Activations

ReLU, Sigmoid, Softmax, Tanh

ayanmo activated = ML.relu(tensor);
ayanmo probs = ML.softmax(logits);
ayanmo squashed = ML.sigmoid(values);

Layers

Dense, Conv2D, BatchNorm

// Dense layer
ayanmo out = ML.dense(input, weights, bias);

// Convolution
ayanmo conv = ML.conv2d(image, kernel);

Loss Functions

MSE, CrossEntropy

ayanmo loss = ML.mse(predicted, actual);
ayanmo ce = ML.cross_entropy(logits, labels);
