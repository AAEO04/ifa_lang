?? IoT Stack

Embedded systems and hardware interfaces

Features

GPIO

Digital input/output pins

// Configure pin
ayanmo led = IoT.gpio(13, "output");
ayanmo button = IoT.gpio(2, "input");

// Read/write
IoT.write(led, 1);  // HIGH
ayanmo state = IoT.read(button);

Serial

UART communication

// Open serial port
ayanmo port = IoT.serial("/dev/ttyUSB0", 9600);

// Read/write
IoT.serial_write(port, "Hello");
ayanmo data = IoT.serial_read(port);

I2C

I2C bus communication

// Initialize I2C
ayanmo bus = IoT.i2c(1);  // Bus 1

// Read sensor
ayanmo temp = IoT.i2c_read(bus, 0x48, 2);  // Address, bytes

SPI

SPI bus communication

ayanmo spi = IoT.spi(0, 1000000);  // Bus 0, 1MHz

ayanmo response = IoT.spi_transfer(spi, [0x01, 0x02]);

Sensors

Common sensor interfaces

// Temperature sensor
ayanmo temp = IoT.temperature();

// Humidity
ayanmo humidity = IoT.humidity();

// Accelerometer
ayanmo accel = IoT.accelerometer();  // Returns [x, y, z]
