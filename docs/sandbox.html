Manual Container Control

from src.sandbox import IgbaleRuntime, OgbeConfig

runtime = IgbaleRuntime()
config = OgbeConfig(
    name="my-container",
    command=["python", "-c", "print('Hello')"],
    auto_remove=True
)

container = runtime.create(config)
container.start()

# Use virtual filesystem
container.odi_fs.write_file("workspace/data.txt", "Some data")
content = container.odi_fs.read_file("workspace/data.txt")

container.stop()
runtime.gc()  # Cleanup
