// ============================================================
// CONWAY'S GAME OF LIFE - Ifá-Lang Showcase
// ============================================================
// Rules:
// 1. Any live cell with 2-3 neighbors survives
// 2. Any dead cell with exactly 3 neighbors becomes alive
// 3. All other cells die or stay dead

// Grid settings
ayanmo WIDTH = 20;
ayanmo HEIGHT = 12;
ayanmo GENERATIONS = 5;

ayanmo ALIVE = "¦";
ayanmo DEAD = "·";

// =============== GRID FUNCTIONS ===============

// Create empty grid
ise create_grid(w, h) {
    ayanmo grid = [];
    ayanmo y = 0;
    nigba (y < h) {
        ayanmo row = [];
        ayanmo x = 0;
        nigba (x < w) {
            ayanmo row = Ogunda.push(row, 0);
            ayanmo x = x + 1;
        }
        ayanmo grid = Ogunda.push(grid, row);
        ayanmo y = y + 1;
    }
    padap? grid;
}

// Set cell value (returns new grid)
ise set_cell(grid, x, y, val) {
    ayanmo new_grid = [];
    ayanmo row_idx = 0;
    
    fun row ninu grid {
        ti (row_idx == y) {
            ayanmo new_row = [];
            ayanmo col_idx = 0;
            fun cell ninu row {
                ti (col_idx == x) {
                    ayanmo new_row = Ogunda.push(new_row, val);
                } bib?k? {
                    ayanmo new_row = Ogunda.push(new_row, cell);
                }
                ayanmo col_idx = col_idx + 1;
            }
            ayanmo new_grid = Ogunda.push(new_grid, new_row);
        } bib?k? {
            ayanmo new_grid = Ogunda.push(new_grid, row);
        }
        ayanmo row_idx = row_idx + 1;
    }
    padap? new_grid;
}

// Get cell value
ise get_cell(grid, x, y) {
    ti (x < 0 || x >= WIDTH || y < 0 || y >= HEIGHT) {
        padap? 0;
    }
    ayanmo row = Ogunda.get(grid, y);
    padap? Ogunda.get(row, x);
}

// Count live neighbors (8-way)
ise count_neighbors(grid, x, y) {
    ayanmo count = 0;
    ayanmo dy = -1;
    
    nigba (dy <= 1) {
        ayanmo dx = -1;
        nigba (dx <= 1) {
            ti (!(dx == 0 && dy == 0)) {
                ayanmo neighbor = get_cell(grid, x + dx, y + dy);
                ti (neighbor == 1) {
                    ayanmo count = count + 1;
                }
            }
            ayanmo dx = dx + 1;
        }
        ayanmo dy = dy + 1;
    }
    
    padap? count;
}

// Compute next generation
ise next_generation(grid) {
    ayanmo new_grid = create_grid(WIDTH, HEIGHT);
    ayanmo y = 0;
    
    nigba (y < HEIGHT) {
        ayanmo x = 0;
        nigba (x < WIDTH) {
            ayanmo cell = get_cell(grid, x, y);
            ayanmo neighbors = count_neighbors(grid, x, y);
            
            // Apply rules
            ti (cell == 1) {
                // Live cell
                ti (neighbors == 2 || neighbors == 3) {
                    ayanmo new_grid = set_cell(new_grid, x, y, 1);
                }
            } bib?k? {
                // Dead cell
                ti (neighbors == 3) {
                    ayanmo new_grid = set_cell(new_grid, x, y, 1);
                }
            }
            
            ayanmo x = x + 1;
        }
        ayanmo y = y + 1;
    }
    
    padap? new_grid;
}

// Display grid
ise display(grid, gen) {
    Irosu.fo("--- Generation " + gen + " ---");
    
    fun row ninu grid {
        ayanmo line = "";
        fun cell ninu row {
            ti (cell == 1) {
                ayanmo line = line + ALIVE;
            } bib?k? {
                ayanmo line = line + DEAD;
            }
        }
        Irosu.fo(line);
    }
    Irosu.fo("");
}

// =============== PATTERNS ===============

// Place a glider at position
ise place_glider(grid, ox, oy) {
    //  .¦.
    //  ..¦
    //  ¦¦¦
    ayanmo g = set_cell(grid, ox+1, oy, 1);
    ayanmo g = set_cell(g, ox+2, oy+1, 1);
    ayanmo g = set_cell(g, ox, oy+2, 1);
    ayanmo g = set_cell(g, ox+1, oy+2, 1);
    ayanmo g = set_cell(g, ox+2, oy+2, 1);
    padap? g;
}

// Place a blinker (oscillator)
ise place_blinker(grid, ox, oy) {
    //  ¦¦¦
    ayanmo g = set_cell(grid, ox, oy, 1);
    ayanmo g = set_cell(g, ox+1, oy, 1);
    ayanmo g = set_cell(g, ox+2, oy, 1);
    padap? g;
}

// =============== SIMULATION ===============

Irosu.fo("?? Conway's Game of Life");
Irosu.fo("Built with Ifá-Lang\n");

// Create grid with patterns
ayanmo grid = create_grid(WIDTH, HEIGHT);
ayanmo grid = place_glider(grid, 2, 2);
ayanmo grid = place_blinker(grid, 10, 5);

// Run simulation
ayanmo gen = 0;
nigba (gen < GENERATIONS) {
    display(grid, gen);
    ayanmo grid = next_generation(grid);
    ayanmo gen = gen + 1;
}

display(grid, gen);
Irosu.fo("? Simulation complete!");
