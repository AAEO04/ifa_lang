FizzBuzz

fizzbuzz.ifa 
// Classic FizzBuzz problem
ayanmo i = 1;

nigba (i <= 20) {
    ti (i % 15 == 0) {
        Irosu.fo("FizzBuzz");
    } bib?k? ti (i % 3 == 0) {
        Irosu.fo("Fizz");
    } bib?k? ti (i % 5 == 0) {
        Irosu.fo("Buzz");
    } bib?k? {
        Irosu.fo(i);
    }
    ayanmo i = i + 1;
}
// Classic FizzBuzz problem
let i = 1;

while (i <= 20) {
    if (i % 15 == 0) {
        Irosu.println("FizzBuzz");
    } else if (i % 3 == 0) {
        Irosu.println("Fizz");
    } else if (i % 5 == 0) {
        Irosu.println("Buzz");
    } else {
        Irosu.println(i);
    }
    let i = i + 1;
}

Fibonacci Sequence

// Generate first 15 Fibonacci numbers
ayanmo a = 0;
ayanmo b = 1;
ayanmo n = 15;

Irosu.fo("Fibonacci:");
ayanmo i = 0;

nigba (i < n) {
    Irosu.fo(a);
    ayanmo temp = a + b;
    ayanmo a = b;
    ayanmo b = temp;
    ayanmo i = i + 1;
}
