Ogbe Domain Examples

System Operations and Infrastructure (1111)

Overview

Ogbe (1111) is the system domain representing creation, expansion, and system operations. It provides comprehensive system-level functionality including file operations, process management, hardware control, and system information.

Domain Characteristics

  • Binary Pattern: 1111 (all ones - completeness)
  • Yoruba Meaning: "He who creates the path" or "The expander"
  • System Role: System operations and infrastructure
  • Element: Fire/Air (creation and expansion)

Core Capabilities

  • File System: File and directory operations
  • Process Management: Process creation and control
  • System Information: Hardware and OS details
  • Network Operations: Basic network functions
  • Hardware Control: GPIO and device management

File Reading and Writing 
// Read file contents
ese read_config_file(path: Str) -> Map {
    gbiyanju {
        ayanmo content = Ogbe.file_read(path);
        ayanmo lines = Ika.split(content, "\n");
        ayanmo config = Map::::new();
        
        fun line ninu lines {

                ayanmo parts = Ika.split(line, "=");

            }
        }
        
        pada config;
    } gba (error) {
        Irosu.kigbe("Error reading config: " + error.message);
        pada Map::::new();
    }
}

// Write file with backup
ese safe_write(path: Str, content: Str) -> Bool {
    gbiyanju {
        // Create backup
        ayanmo backup_path = path + ".backup";
        if Ogbe.file_exists(path) {
            Ogbe.file_copy(path, backup_path);
        }
        
        // Write new content
        Ogbe.file_write(path, content);
        
        // Verify write
        ayanmo written = Ogbe.file_read(path);
        if written == content {
            Irosu.fo("File written successfully: " + path);
            pada otito;
        } else {
            Irosu.kigbe("File verification failed for: " + path);
            pada eke;
        }
    } gba (error) {
        Irosu.kigbe("Error writing file: " + error.message);
        pada eke;
    }
}

Directory Management 
// Create directory with parents
ese create_directory_tree(path: Str) -> Bool {
    gbiyanju {
        ayanmo parts = Ika.split(path, "/");
        ayanmo current = "";
        
        fun part ninu parts {
            if part != "" {
                current = current + "/" + part;
                if !Ogbe.dir_exists(current) {
                    Ogbe.dir_create(current);
                    Irosu.fo("Created directory: " + current);
                }
            }
        }
        
        pada otito;
    } gba (error) {
        Irosu.kigbe("Error creating directory: " + error.message);
        pada eke;
    }
}

// Directory listing with filtering
ese list_files_by_extension(dir: Str, extension: Str) -> Vec {
    ayanmo files = Vec::::new();
    
    gbiyanju {
        ayanmo entries = Ogbe.dir_list(dir);
        
        fun entry ninu entries {
            ayanmo full_path = dir + "/" + entry;
            if Ogbe.file_exists(full_path) {
                ayanmo ext = Ika.substring(entry, Ika.length(entry) - Ika.length(extension));
                if ext == extension {
                    files.push(entry);
                }
            }
        }
    } gba (error) {
        Irosu.kigbe("Error listing directory: " + error.message);
    }
    
    pada files;
}

Process Management

Process Creation and Control

Process Operations 
// Execute command with timeout
ese execute_command(cmd: Str, timeout_ms: u32) -> Map {
    ayanmo result = Map::::new();
    
    gbiyanju {
        ayanmo start_time = Iwori.timestamp();
        ayanmo process = Ogbe.process_start(cmd);
        
        // Wait for completion or timeout
        while Ogbe.process_running(process) {
            ayanmo elapsed = Iwori.timestamp() - start_time;
            if elapsed > timeout_ms {
                Ogbe.process_kill(process);
                result.insert("status", "timeout");
                result.insert("error", "Command timed out after " + timeout_ms + "ms");
                pada result;
            }
            Iwori.sleep(10); // Sleep 10ms
        }
        
        ayanmo exit_code = Ogbe.process_exit_code(process);
        ayanmo stdout = Ogbe.process_stdout(process);
        ayanmo stderr = Ogbe.process_stderr(process);
        
        result.insert("status", "completed");
        result.insert("exit_code", Obara.to_str(exit_code));
        result.insert("stdout", stdout);
        result.insert("stderr", stderr);
        result.insert("duration", Obara.to_str(Iwori.timestamp() - start_time));
        
    } gba (error) {
        result.insert("status", "error");
        result.insert("error", error.message);
    }
    
    pada result;
}

System Information

Hardware and OS Details

System Information Gathering 
// Get comprehensive system information
ese get_system_info() -> Map {
    ayanmo info = Map::::new();
    
    gbiyanju {
        // Operating System
        info.insert("os_name", Ogbe.os_name());
        info.insert("os_version", Ogbe.os_version());
        info.insert("os_arch", Ogbe.os_arch());
        
        // Hardware
        info.insert("cpu_model", Ogbe.cpu_model());
        info.insert("cpu_cores", Obara.to_str(Ogbe.cpu_cores()));
        info.insert("cpu_frequency", Obara.to_str(Ogbe.cpu_frequency()) + "MHz");
        
        // Memory
        ayanmo total_mem = Ogbe.memory_total();
        ayanmo free_mem = Ogbe.memory_free();
        ayanmo used_mem = total_mem - free_mem;
        
        info.insert("memory_total", Obara.format_bytes(total_mem));
        info.insert("memory_used", Obara.format_bytes(used_mem));
        info.insert("memory_free", Obara.format_bytes(free_mem));
        info.insert("memory_usage", Obara.to_str((used_mem * 100) / total_mem) + "%");
        
        // Disk
        ayanmo disk_info = Ogbe.disk_info("/");
        info.insert("disk_total", Obara.format_bytes(disk_info.total));
        info.insert("disk_used", Obara.format_bytes(disk_info.used));
        info.insert("disk_free", Obara.format_bytes(disk_info.free));
        info.insert("disk_usage", Obara.to_str((disk_info.used * 100) / disk_info.total) + "%");
        
        // Network
        ayanmo network = Ogbe.network_interfaces();
        info.insert("network_interfaces", Obara.to_str(network.length()));
        
        // Uptime
        info.insert("uptime", Obara.format_duration(Ogbe.uptime()));
        
    } gba (error) {
        Irosu.kigbe("Error getting system info: " + error.message);
    }
    
    pada info;
}

// System health check
ese health_check() -> Map {
    ayanmo health = Map::::new();
    
    // Memory health
    ayanmo mem_total = Ogbe.memory_total();
    ayanmo mem_free = Ogbe.memory_free();
    ayanmo mem_usage = ((mem_total - mem_free) * 100) / mem_total;
    
    if mem_usage > 90 {
        health.insert("memory", "critical");
    } else if mem_usage > 80 {
        health.insert("memory", "warning");
    } else {
        health.insert("memory", "ok");
    }
    
    // Disk health
    ayanmo disk_info = Ogbe.disk_info("/");
    ayanmo disk_usage = (disk_info.used * 100) / disk_info.total;
    
    if disk_usage > 95 {
        health.insert("disk", "critical");
    } else if disk_usage > 85 {
        health.insert("disk", "warning");
    } else {
        health.insert("disk", "ok");
    }
    
    // CPU health (simplified)
    ayanmo cpu_usage = Ogbe.cpu_usage();
    if cpu_usage > 90 {
        health.insert("cpu", "critical");
    } else if cpu_usage > 80 {
        health.insert("cpu", "warning");
    } else {
        health.insert("cpu", "ok");
    }
    
    // Overall health
    ayanmo critical_count = 0;
    ayanmo warning_count = 0;
    
    fun (key, status) ninu health {
        if status == "critical" {
            critical_count += 1;
        } else if status == "warning" {
            warning_count += 1;
        }
    }
    
    if critical_count > 0 {
        health.insert("overall", "critical");
    } else if warning_count > 0 {
        health.insert("overall", "warning");
    } else {
        health.insert("overall", "ok");
    }
    
    health.insert("timestamp", Obara.to_str(Iwori.timestamp()));
    
    pada health;
}

// Port scanning

            result.insert("error", error.message);
        }

}

// Button reading with debouncing
ese read_button(pin: u8, debounce_ms: u32) -> Bool {
    gbiyanju {
        // Configure GPIO pin
        Ogbe.gpio_config(pin, "input");
        Ogbe.gpio_set_pull(pin, "pullup");
        
        ayanmo last_state = Ogbe.gpio_read(pin);
        ayanmo last_time = Iwori.timestamp();
        
        // Simple debouncing
        while otito {
            ayanmo current_state = Ogbe.gpio_read(pin);
            ayanmo current_time = Iwori.timestamp();
            
            if current_state != last_state {
                if (current_time - last_time) >= debounce_ms {
                    // State changed and debounced
                    if current_state == eke { // Button pressed (pullup -> low)
                        pada otito;
                    }
                    last_state = current_state;
                    last_time = current_time;
                }
            }
            
            Iwori.sleep(10); // Check every 10ms
        }
    } gba (error) {
        Irosu.kigbe("Error reading button: " + error.message);
        pada eke;
    }
}

// Temperature sensor reading (I2C)
ese read_temperature_sensor(i2c_bus: u8, device_addr: u8) -> Float {
    gbiyanju {
        // Initialize I2C
        Ogbe.i2c_init(i2c_bus);
        
        // Write register pointer
        Ogbe.i2c_write(device_addr, [0x00]); // Temperature register
        
        // Read temperature data (2 bytes)
        ayanmo data = Ogbe.i2c_read(device_addr, 2);
        
        // Convert to temperature (simplified for LM75)
        ayanmo raw_temp = (data[0] << 8) | data[1];
        ayanmo temperature = raw_temp as Float / 256.0;
        
        pada temperature;
        
    } gba (error) {
        Irosu.kigbe("Error reading temperature sensor: " + error.message);
        pada 0.0;
    }
}

Best Practices

Error Handling

  • Always wrap system operations in gbiyanju/gba blocks
  • Provide meaningful error messages with context
  • Use Okanran domain for structured error handling
  • Implement graceful degradation for non-critical failures

Resource Management

  • Use  (sacrifice) blocks for resource cleanup
  • Close file handles and network connections properly
  • Monitor resource usage to prevent leaks
  • Implement timeout mechanisms for blocking operations

Security Considerations

  • Validate file paths to prevent directory traversal
  • Sanitize command arguments to prevent injection
  • Use proper permissions for file operations
  • Implement access controls for sensitive operations

Performance Optimization

  • Use asynchronous operations for I/O-bound tasks
  • Cache frequently accessed system information
  • Implement efficient data structures for large datasets
  • Profile and optimize hot paths in system operations
