Data Pipeline

Reactive Data Processing with Ifá-Lang

Overview

This example demonstrates building a complete data pipeline using Ifá-Lang's reactive programming system (Àjõsé), domain-based data processing, and cultural integration. The pipeline includes data ingestion, transformation, validation, aggregation, and output with real-time monitoring and error handling.

// CSV data source
struct CsvDataSource {
    file_path: Str,
    delimiter: Str,
    has_header: Bool
}

impl DataSource for CsvDataSource {
    se connect(&self) -> Result<(), Str> {
        if !Ogbe.file_exists(self.file_path) {
            pada Err("File not found: " + self.file_path);
        }
        
        Irosu.fo("Connected to CSV source: " + self.file_path);
        pada Ok(());
    }
    
    se read_batch(&self, batch_size: u32) -> Result, Str> {
        gbiyanju {
            ayanmo content = Ogbe.file_read(self.file_path);
            ayanmo lines = Ika.split(content, "\n");
            
            // Skip header if present
            ayanmo start_line = if self.has_header { 1 } else { 0 };
            ayanmo records = Vec::::new();
            
            fun i ninu start_line..Obara.min(start_line + batch_size, lines.length()) {
                ayanmo line = lines[i];
                if Ika.trim(line).length() > 0 {
                    ayanmo fields = Ika.split(line, self.delimiter);
                    
                    ayanmo record = DataRecord {
                        id: "csv_" + Obara.to_str(i),
                        timestamp: Iwori.timestamp(),
                        source: "csv_file",
                        data: parse_csv_fields(fields),
                        metadata: Map::::new()
                    };
                    
                    records.push(record);
                }
            }
            
            pada Ok(records);
        } gba (error) {
            pada Err("Failed to read CSV: " + error.message);
        }
    }
    
    se disconnect(&self) -> Result<(), Str> {
        Irosu.fo("Disconnected from CSV source");
        pada Ok(());
    }
}

            pada Err("Database connection failed: " + error.message);

    }
    
    se read_batch(&self, batch_size: u32) -> Result, Str> {
        gbiyanju {
            ayanmo conn = Odi.connect(self.connection_string);
            ayanmo limited_query = self.query + " LIMIT " + Obara.to_str(batch_size);
            ayanmo rows = conn.query(limited_query);
            
            ayanmo records = Vec::::new();
            
            fun row ninu rows {
                ayanmo record = DataRecord {
                    id: row.get("id"),
                    timestamp: Iwori.parse_timestamp(row.get("created_at")),
                    source: "database",
                    data: Map::::new(),
                    metadata: Map::::new()
                };
                
                // Add all columns to data
                fun (column, value) ninu row {
                    if column != "id" and column != "created_at" {
                        record.data.insert(column, value);
                    }
                }
                
                records.push(record);
            }
            
            pada Ok(records);
        } gba (error) {
            pada Err("Database query failed: " + error.message);
        }
    }
    
    se disconnect(&self) -> Result<(), Str> {
        Irosu.fo("Disconnected from database");
        pada Ok(());
    }
}

// Multi-source ingestion manager
se ingest_from_multiple_sources(sources: Vec>, batch_size: u32) -> Vec {
    ayanmo all_records = Vec::::new();
    
    // Process sources in parallel using Osa domain
    ayanmo futures = Vec:: Vec>::new>();
    
    fun source ninu sources {
        futures.push(|| {
            gbiyanju {
                source.connect()?;
                ayanmo records = source.read_batch(batch_size)?;
                source.disconnect()?;
                pada records;
            } gba (error) {
                Irosu.kigbe("Source ingestion failed: " + error.message);
                pada Vec::::new();
            }
        });
    }
    
    // Execute in parallel
    ayanmo results = Osa.parallel_execute(futures);
    
    fun result ninu results {
        fun record ninu result {
            all_records.push(record);
        }
    }
    
    Irosu.fo("Ingested " + Obara.to_str(all_records.length()) + " records from " + 
             Obara.to_str(sources.length()) + " sources");
    
    pada all_records;
}

// Validate single record
se validate_record(record: DataRecord, schema: DataSchema) -> ValidationResult {
    ayanmo mut result = ValidationResult {
        is_valid: otito,
        errors: Vec::::new(),
        warnings: Vec::::new()
    };
    
    // Check required fields
    fun field ninu schema.required_fields {
        if !record.data.contains_key(field) {
            result.errors.push("Missing required field: " + field);
            result.is_valid = eke;
        }
    }
    
    // Apply validation rules
    fun rule ninu schema.validation_rules {
        if record.data.contains_key(rule.field) {
            ayanmo field_value = record.data.get(rule.field);
            ayanmo validation_result = apply_validation_rule(field_value, rule);
            
            if !validation_result.is_valid {
                result.errors.push(validation_result.error);
                result.is_valid = eke;
            }
            
            if validation_result.warning != "" {
                result.warnings.push(validation_result.warning);
            }
        }
    }
    
    // Check data types
    fun (field, expected_type) ninu schema.data_types {
        if record.data.contains_key(field) {
            ayanmo field_value = record.data.get(field);
            if !check_data_type(field_value, expected_type) {
                result.errors.push("Field " + field + " should be of type " + expected_type);
                result.is_valid = eke;
            }
        }
    }
    
    pada result;
}

// Apply individual validation rule
se apply_validation_rule(value: Any, rule: ValidationRule) -> ValidationResult {
    match rule.rule_type {
        "required" => {
            ValidationResult {
                is_valid: value != "",
                error: if value == "" { rule.error_message } else { "" },
                warning: ""
            }
        }
        
        "email_format" => {
            ayanmo email_str = value.to_string();
            ayanmo is_valid = Ika.contains(email_str, "@") and Ika.contains(email_str, ".");
            ValidationResult {
                is_valid: is_valid,
                error: if !is_valid { rule.error_message } else { "" },
                warning: ""
            }
        }
        
        "range" => {
            ayanmo num_value = value.to_f64();
            ayanmo min_val = rule.parameters.get("min").to_f64();
            ayanmo max_val = rule.parameters.get("max").to_f64();
            ayanmo is_valid = num_value >= min_val and num_value <= max_val;
            
            ValidationResult {
                is_valid: is_valid,
                error: if !is_valid { rule.error_message } else { "" },
                warning: ""
            }
        }
        
        "string_length" => {
            ayanmo str_value = value.to_string();
            ayanmo length = Ika.length(str_value);
            ayanmo min_len = rule.parameters.get("min").to_u32();
            ayanmo max_len = rule.parameters.get("max").to_u32();
            ayanmo is_valid = length >= min_len and length <= max_len;
            
            ValidationResult {
                is_valid: is_valid,
                error: if !is_valid { rule.error_message } else { "" },
                warning: ""
            }
        }
        
        "timestamp_range" => {
            ayanmo timestamp = value.to_u64();
            ayanmo min_time = rule.parameters.get("min").to_u64();
            ayanmo max_time = rule.parameters.get("max").to_u64();
            ayanmo is_valid = timestamp >= min_time and timestamp <= max_time;
            
            ValidationResult {
                is_valid: is_valid,
                error: if !is_valid { rule.error_message } else { "" },
                warning: ""
            }
        }
        
        _ => ValidationResult {
            is_valid: otito,
            error: "Unknown validation rule: " + rule.rule_type,
            warning: ""
        }
    }
}

// Check data type
se check_data_type(value: Any, expected_type: Str) -> Bool {
    match expected_type {
        "str" => value.is_string(),
        "int" => value.is_integer(),
        "float" => value.is_float(),
        "bool" => value.is_boolean(),
        _ => otito  // Unknown type, assume valid
    }
}

// Batch validation
se validate_batch(records: Vec, schema: DataSchema) -> (Vec, Vec) {
    ayanmo valid_records = Vec::::new();
    ayanmo validation_results = Vec::::new();
    
    fun record ninu records {
        ayanmo result = validate_record(record, schema);
        validation_results.push(result);
        
        if result.is_valid {
            valid_records.push(record);
        }
    }
    
    Irosu.fo("Validation complete: " + Obara.to_str(valid_records.length()) + "/" + 
             Obara.to_str(records.length()) + " records passed");
    
    pada (valid_records, validation_results);
}

// Transformation functions
se normalize_timestamp(record: DataRecord) -> DataRecord {
    ayanmo mut normalized_record = record;
    
    // Ensure timestamp is in UTC
    if record.metadata.contains_key("timezone") {
        ayanmo timezone = record.metadata.get("timezone");
        ayanmo utc_timestamp = Iwori.convert_timezone(record.timestamp, timezone, "UTC");
        normalized_record.timestamp = utc_timestamp;
    }
    
    normalized_record.metadata.insert("normalized", "true");
    pada normalized_record;
}

se enrich_data(record: DataRecord) -> DataRecord {
    ayanmo mut enriched_record = record;
    
    // Add computed fields
    enriched_record.data.insert("processed_at", Iwori.timestamp());
    enriched_record.data.insert("day_of_week", Obara.to_str(Iwori.get_date_info(record.timestamp).day_of_week));
    
    // Add cultural context
    if enriched_record.data.contains_key("name") {
        ayanmo name = enriched_record.data.get("name").to_string();
        enriched_record.data.insert("name_length", Obara.to_str(Ika.length(name)));
        
        // Add Yoruba cultural significance if name matches patterns
        if Ika.contains(name.to_lower(), "ifa") or Ika.contains(name.to_lower(), "orisa") {
            enriched_record.metadata.insert("cultural_significance", "spiritual");
        }
    }
    
    enriched_record.metadata.insert("enriched", "true");
    pada enriched_record;
}

se filter_quality(record: DataRecord) -> Option {
    // Quality filter based on data completeness
    ayanmo completeness_score = 0.0;
    ayanmo total_fields = 10.0;  // Expected fields
    
    if record.data.contains_key("id") { completeness_score += 1.0; }
    if record.data.contains_key("name") { completeness_score += 1.0; }
    if record.data.contains_key("email") { completeness_score += 1.0; }
    if record.data.contains_key("age") { completeness_score += 1.0; }
    if record.data.contains_key("location") { completeness_score += 1.0; }
    if record.data.contains_key("phone") { completeness_score += 1.0; }
    if record.data.contains_key("status") { completeness_score += 1.0; }
    if record.data.contains_key("created_at") { completeness_score += 1.0; }
    if record.data.contains_key("updated_at") { completeness_score += 1.0; }
    if record.data.contains_key("category") { completeness_score += 1.0; }
    
    ayanmo quality_score = completeness_score / total_fields;
    
    if quality_score >= 0.7 {  // 70% completeness threshold
        ayanmo mut filtered_record = record;
        filtered_record.metadata.insert("quality_score", Obara.format(quality_score, 3));
        filtered_record.metadata.insert("quality_status", "high");
        pada Some(filtered_record);
    } else {
        Irosu.fo("Filtered low quality record (score: " + Obara.format(quality_score, 3) + ")");
        None
    }
}

se aggregate_metrics(record: DataRecord) -> DataRecord {
    ayanmo mut aggregated_record = record;
    
    // Add aggregation metrics
    aggregated_record.data.insert("record_hash", Irete.hash_sha256(record.to_string()));
    aggregated_record.data.insert("batch_id", "batch_" + Obara.to_str(Iwori.timestamp() / 60000));  // Hourly batches
    
    // Add cultural metrics
    if aggregated_record.data.contains_key("category") {
        ayanmo category = aggregated_record.data.get("category").to_string();
        aggregated_record.data.insert("category_weight", get_category_weight(category));
    }
    
    aggregated_record.metadata.insert("aggregated", "true");
    pada aggregated_record;
}

// Get cultural weight for category
se get_category_weight(category: Str) -> f64 {
    match category.to_lower() {
        "spiritual" => 1.0,
        "cultural" => 0.9,
        "educational" => 0.8,
        "community" => 0.7,
        "personal" => 0.6,
        "business" => 0.5,
        _ => 0.4
    }
}

// Create reactive processing pipeline
se create_reactive_pipeline(input_data: Vec) -> Stream {
    // Create input stream
    ayanmo input_stream = Stream::from_vec(input_data);
    
    // Set up reactive processing using Àjõsé
    ajose! {
        // Create processing stream
        ayanmo processing_stream = input_stream
            .map(normalize_timestamp)
            .map(enrich_data)
            .filter_map(filter_quality)
            .map(aggregate_metrics);
        
        // Add monitoring
        processing_stream.subscribe(|record| {
            Irosu.fo("Processed record: " + record.id);
        });
        
        // Add error handling
        processing_stream.subscribe_error(|error| {
            Irosu.kigbe("Processing error: " + error.message);
        });
        
        pada processing_stream;
    }
}

// Process data through reactive pipeline
se process_reactive_pipeline(data: Vec) -> Vec {

    ayanmo mut processed_data = Vec::::new();

se count_by_category(records: Vec) -> Vec {
    ayanmo category_counts = Map::::new();
    
    fun record ninu records {
        if record.data.contains_key("category") {
            ayanmo category = record.data.get("category").to_string();
            if category_counts.contains_key(category) {
                category_counts.insert(category, category_counts.get(category) + 1);
            } else {
                category_counts.insert(category, 1);
            }
        }
    }
    
    ayanmo results = Vec::::new();
    fun (category, count) ninu category_counts {
        results.push(AggregationResult {
            metric: "count_by_category",
            value: count as f64,
            timestamp: Iwori.timestamp(),
            metadata: Map::::new()
                .insert("category", category)
        });
    }
    
    pada results;
}

se average_by_field(records: Vec, field: Str) -> AggregationResult {
    ayanmo mut sum = 0.0;
    ayanmo mut count = 0;
    
    fun record ninu records {
        if record.data.contains_key(field) {
            ayanmo value = record.data.get(field).to_f64();
            sum = Obara.fikun(sum, value);
            count += 1;
        }
    }
    
    ayanmo average = if count > 0 { sum / count as f64 } else { 0.0 };
    
    AggregationResult {
        metric: "average_" + field,
        value: average,
        timestamp: Iwori.timestamp(),
        metadata: Map::::new()
            .insert("count", Obara.to_str(count))
    }
}

se percentiles_by_field(records: Vec, field: Str) -> Map {
    ayanmo values = Vec::::new();
    
    fun record ninu records {
        if record.data.contains_key(field) {
            values.push(record.data.get(field).to_f64());
        }
    }
    
    if values.length() == 0 {
        return Map::::new();
    }
    
    values.sort();
    
    ayanmo percentiles = Map::::new();
    
    // Calculate percentiles
    percentiles.insert("p50", values[values.length() / 2]);  // Median
    percentiles.insert("p25", values[values.length() / 4]);
    percentiles.insert("p75", values[values.length() * 3 / 4]);
    percentiles.insert("p90", values[values.length() * 9 / 10]);
    percentiles.insert("p95", values[values.length() * 95 / 100]);
    percentiles.insert("p99", values[values.length() * 99 / 100]);
    
    percentiles.insert("min", values[0]);
    percentiles.insert("max", values[values.length() - 1]);
    
    pada percentiles;
}

se time_series_aggregation(records: Vec, time_window: u64) -> Vec {
    if records.length() == 0 {
        return Vec::::new();
    }
    
    // Group records by time window
    ayanmo time_groups = Map::>::new();
    
    fun record ninu records {
        ayanmo window_start = (record.timestamp / time_window) * time_window;
        
        if time_groups.contains_key(window_start) {
            time_groups.get_mut(window_start).push(record);
        } else {
            time_groups.insert(window_start, Vec::::new().push(record));
        }
    }
    
    ayanmo results = Vec::::new();
    
    fun (window_start, window_records) ninu time_groups {
        ayanmo count = window_records.length() as f64;
        ayanmo avg_quality = window_records
            .map(|r| r.metadata.get("quality_score", "0.0").to_f64())
            .reduce(Obara.fikun, 0.0) / count;
        
        results.push(AggregationResult {
            metric: "time_series_count",
            value: count,
            timestamp: window_start,
            metadata: Map::::new()
                .insert("window_start", Iwori.format_timestamp(window_start))
                .insert("record_count", Obara.to_str(count))
        });
        
        results.push(AggregationResult {
            metric: "time_series_avg_quality",
            value: avg_quality,
            timestamp: window_start,
            metadata: Map::::new()
                .insert("window_start", Iwori.format_timestamp(window_start))
        });
    }
    
    // Sort by timestamp
    results.sort_by(|a, b| Obara.compare(a.timestamp, b.timestamp));
    
    pada results;
}

// Cultural aggregation
se cultural_significance_aggregation(records: Vec) -> Map {
    ayanmo cultural_scores = Map::::new();
    
    // Initialize cultural categories
    cultural_scores.insert("spiritual", 0.0);
    cultural_scores.insert("community", 0.0);
    cultural_scores.insert("traditional", 0.0);
    cultural_scores.insert("modern", 0.0);
    cultural_scores.insert("educational", 0.0);
    
    fun record ninu records {
        if record.metadata.contains_key("cultural_significance") {
            ayanmo significance = record.metadata.get("cultural_significance");
            ayanmo weight = record.metadata.get("quality_score", "0.5").to_f64();
            
            if cultural_scores.contains_key(significance) {
                cultural_scores.insert(significance, 
                    cultural_scores.get(significance) + weight);
            }
        }
        
        // Also check category for cultural context
        if record.data.contains_key("category") {
            ayanmo category = record.data.get("category").to_string();
            ayanmo weight = record.metadata.get("quality_score", "0.5").to_f64();
            
            match category.to_lower() {
                "spiritual" => cultural_scores.insert("spiritual", 
                    cultural_scores.get("spiritual") + weight),
                "community" => cultural_scores.insert("community", 
                    cultural_scores.get("community") + weight),
                "traditional" => cultural_scores.insert("traditional", 
                    cultural_scores.get("traditional") + weight),
                "modern" => cultural_scores.insert("modern", 
                    cultural_scores.get("modern") + weight),
                "educational" => cultural_scores.insert("educational", 
                    cultural_scores.get("educational") + weight),
                _ => {}
            }
        }
    }
    
    // Normalize scores
    ayanmo total_score = 0.0;
    fun (_, score) ninu cultural_scores {
        total_score = Obara.fikun(total_score, score);
    }
    
    if total_score > 0.0 {

// Pipeline monitor
struct PipelineMonitor {
    start_time: u64,
    records_processed: u64,
    records_failed: u64,
    error_threshold: f64,
    alerts: Vec
}

impl PipelineMonitor {
    se new(error_threshold: f64) -> PipelineMonitor {
        PipelineMonitor {
            start_time: Iwori.timestamp(),
            records_processed: 0,
            records_failed: 0,
            error_threshold: error_threshold,
            alerts: Vec::::new()
        }
    }
    
    se record_processed(&mut self) {
        self.records_processed += 1;
        
        // Check for performance issues every 1000 records
        if self.records_processed % 1000 == 0 {
            self.check_performance();
        }
    }
    
    se record_failed(&mut self) {
        self.records_failed += 1;
        
        // Check error rate
        ayanmo error_rate = self.get_error_rate();
        if error_rate > self.error_threshold {
            self.add_alert("High error rate detected: " + Obara.format(error_rate * 100, 2) + "%");
        }
    }
    
    se get_error_rate(&self) -> f64 {
        ayanmo total = self.records_processed + self.records_failed;
        if total == 0 {
            0.0
        } else {
            self.records_failed as f64 / total as f64
        }
    }
    
    se get_throughput(&self) -> f64 {
        ayanmo elapsed = (Iwori.timestamp() - self.start_time) / 1000.0;  // seconds
        if elapsed > 0.0 {
            self.records_processed as f64 / elapsed
        } else {
            0.0
        }
    }
    
    se check_performance(&self) {
        ayanmo throughput = self.get_throughput();
        ayanmo error_rate = self.get_error_rate();
        
        Irosu.fo("Performance Check:");
        Irosu.fo("  Records processed: " + Obara.to_str(self.records_processed));
        Irosu.fo("  Records failed: " + Obara.to_str(self.records_failed));
        Irosu.fo("  Error rate: " + Obara.format(error_rate * 100, 2) + "%");
        Irosu.fo("  Throughput: " + Obara.format(throughput, 2) + " records/sec");
        
        if error_rate > self.error_threshold {
            Irosu.kigbe("ERROR: Error rate exceeds threshold!");
        }
        
        if throughput < 100.0 {  // Less than 100 records/sec
            Irosu.kigbe("WARNING: Low throughput detected!");
        }
    }
    
    se add_alert(&mut self, message: Str) {
        ayanmo timestamp = Iwori.format_timestamp(Iwori.timestamp());
        ayanmo alert = "[" + timestamp + "] " + message;
        self.alerts.push(alert);
        Irosu.kigbe("ALERT: " + alert);
    }
    
    se get_summary(&self) -> Map {
        ayanmo summary = Map::::new();
        ayanmo elapsed = (Iwori.timestamp() - self.start_time) / 1000.0;
        
        summary.insert("total_records", Obara.to_str(self.records_processed + self.records_failed));
        summary.insert("processed_records", Obara.to_str(self.records_processed));
        summary.insert("failed_records", Obara.to_str(self.records_failed));
        summary.insert("error_rate", Obara.format(self.get_error_rate() * 100, 2) + "%");
        summary.insert("throughput", Obara.format(self.get_throughput(), 2) + " records/sec");
        summary.insert("elapsed_time", Obara.format(elapsed, 2) + " seconds");
        summary.insert("alerts_count", Obara.to_str(self.alerts.length()));
        
        pada summary;
    }
}
