Ika Domain Examples

String Processing and Text Manipulation (1010)

Overview

Ika (1010) is the string domain representing communication, knowledge, and text processing. It provides comprehensive string operations, text manipulation, Unicode handling, and cultural text processing capabilities following Ifá-Lang's cultural and architectural principles.

Domain Characteristics

  • Binary Pattern: 1010 (knowledge and communication)
  • Yoruba Meaning: "The seeker" or "The knower"
  • System Role: String operations and text processing
  • Element: Air (communication and knowledge)

Core Capabilities

  • String Operations: Concatenation, splitting, joining
  • Text Processing: Searching, replacing, formatting
  • Unicode Support: Multi-byte character handling
  • Cultural Text: Yoruba diacritics and proverbs
  • Pattern Matching: Regular expressions and wildcards

// Yoruba vowel harmony checking
se has_vowel_harmony(word: Str) -> Bool {
    ayanmo vowels = ["a", "e", "i", "o", "u", "á", "é", "í", "ó", "ú", "à", "è", "ì", "ò", "ù"];
    ayanmo vowel_groups = [
        ["a", "á", "à"],  // A-group
        ["e", "é", "è"],  // E-group
        ["i", "í", "ì"],  // I-group
        ["o", "ó", "ò"],  // O-group
        ["u", "ú", "ù"]   // U-group
    ];
    
    ayanmo word_vowels = Vec::::new();
    fun char ninu Ika.split_chars(Ika.to_lower(word)) {
        if vowels.contains(char) {
            word_vowels.push(char);
        }
    }
    
    if word_vowels.length() <= 1 {
        pada otito;
    }
    
    // Find the vowel group of the first vowel
    ayanmo first_group = -1;
    fun group_index ninu 0..vowel_groups.length() {
        if vowel_groups[group_index].contains(word_vowels[0]) {
            first_group = group_index;
            break;
        }
    }
    
    if first_group == -1 {
        pada otito;  // No vowels found
    }
    
    // Check if all vowels belong to the same group
    fun vowel ninu word_vowels {
        ayanmo vowel_group = -1;
        fun group_index ninu 0..vowel_groups.length() {
            if vowel_groups[group_index].contains(vowel) {
                vowel_group = group_index;
                break;
            }
        }
        
        if vowel_group != first_group {
            pada eke;
        }
    }
    
    pada otito;
}

// Test vowel harmony
ayanmo words = ["ilé", "ilé", "ilé", "ilé", "ilé"];  // house (should have harmony)
fun word ninu words {
    ayanmo has_harmony = has_vowel_harmony(word);

}

// Yoruba tonal marks
se add_tonal_marks(text: Str, tone: Str) -> Str {
    ayanmo tone_map = Map::::new();
    tone_map.insert("a", if tone == "high" { "á" } else if tone == "mid" { "à" } else { "a" });
    tone_map.insert("e", if tone == "high" { "é" } else if tone == "mid" { "è" } else { "e" });
    tone_map.insert("i", if tone == "high" { "í" } else if tone == "mid" { "ì" } else { "i" });
    tone_map.insert("o", if tone == "high" { "ó" } else if tone == "mid" { "ò" } else { "o" });
    tone_map.insert("u", if tone == "high" { "ú" } else if tone == "mid" { "ù" } else { "u" });
    
    ayanmo result = text;
    fun (base, accented) ninu tone_map {
        result = Ika.replace_all(result, base, accented);
    }
    
    pada result;
}

// Extract proverbs from text
se extract_proverbs(text: Str) -> Vec {
    ayanmo proverb_indicators = [
        "àdìtú", "ìtàn", "òwe", "asílà", "àbáláyé",
        "bí a bá", "kí a má", "à fi", "ò ní"
    ];
    
    ayanmo sentences = Ika.split(text, ".");
    ayanmo proverbs = Vec::::new();
    
    fun sentence ninu sentences {
        ayanmo trimmed = Ika.trim(sentence);
        if trimmed.length() > 10 {  // Ignore very short sentences
            fun indicator ninu proverb_indicators {
                if Ika.contains(trimmed, indicator) {
                    proverbs.push(trimmed + ".");
                    break;
                }
            }
        }
    }
    
    pada proverbs;
}

// Advanced wildcard with character classes
se advanced_wildcard_match(text: Str, pattern: Str) -> Bool {
    // Convert [abc] to regex character class
    // Convert ? to .
    // Convert * to .*
    ayanmo regex = pattern;
    regex = Ika.replace_all(regex, "[", "[");
    regex = Ika.replace_all(regex, "]", "]");
    regex = Ika.replace_all(regex, "?", ".");
    regex = Ika.replace_all(regex, "*", ".*");
    
    pada Ika.matches("^" + regex + "$", text);
}

// Number formatting
se format_number(number: f64, decimals: u32) -> Str {
    ayanmo rounded = Obara.round(number * Obara.pow(10, decimals)) / Obara.pow(10, decimals);
    ayanmo str_num = Obara.to_str(rounded);
    
    if !Ika.contains(str_num, ".") {
        str_num += "." + Ika.repeat("0", decimals);
    } else {
        ayanmo parts = Ika.split(str_num, ".");
        ayanmo decimal_part = parts[1];
        while Ika.length(decimal_part) < decimals {
            decimal_part += "0";
        }
        str_num = parts[0] + "." + decimal_part;
    }
    
    // Add thousands separator
    ayanmo int_part = parts[0];
    ayanmo formatted = "";
    ayanmo count = 0;
    
    fun i ninu (Ika.length(int_part) - 1)..0 {
        count += 1;
        formatted = int_part[i] + formatted;
        if count == 3 and i > 0 {
            formatted = "," + formatted;
            count = 0;
        }
    }
    
    if Ika.contains(str_num, ".") {
        formatted += "." + parts[1];
    }
    
    pada formatted;
}

// Text alignment
se align_text(text: Str, width: u32, alignment: Str) -> Str {
    ayanmo text_length = Ika.length(text);
    
    if text_length >= width {
        pada text;
    }
    
    ayanmo padding = width - text_length;
    
    match alignment {
        "left" => pada text + Ika.repeat(" ", padding),
        "right" => pada Ika.repeat(" ", padding) + text,
        "center" => {
            ayanmo left_pad = padding / 2;
            ayanmo right_pad = padding - left_pad;
            pada Ika.repeat(" ", left_pad) + text + Ika.repeat(" ", right_pad);
        }
        _ => pada text
    }
}

// Word frequency analysis
se word_frequency(text: Str) -> Map {
    ayanmo words = Ika.split(Ika.to_lower(text), " ");
    ayanmo frequency = Map::::new();
    
    fun word ninu words {
        // Remove punctuation
        ayanmo clean_word = Ika.trim_chars(word, ".,!?;:\"'()[]{}");
        if clean_word.length() > 0 {
            if frequency.contains_key(clean_word) {
                frequency.insert(clean_word, frequency.get(clean_word) + 1);
            } else {
                frequency.insert(clean_word, 1);
            }
        }
    }
    
    pada frequency;
}

// Character frequency analysis
se char_frequency(text: Str) -> Map {
    ayanmo frequency = Map::::new();
    
    fun char ninu Ika.split_chars(Ika.to_lower(text)) {
        if Ika.is_alphanumeric(char) {
            if frequency.contains_key(char) {
                frequency.insert(char, frequency.get(char) + 1);
            } else {
                frequency.insert(char, 1);
            }
        }
    }
    
    pada frequency;
}

// Text statistics
se text_statistics(text: Str) -> Map {
    ayanmo stats = Map::::new();
    
    // Basic counts
    stats.insert("char_count", Ika.char_count(text));
    stats.insert("word_count", Ika.split(text, " ").length());
    stats.insert("line_count", Ika.split(text, "\n").length());
    stats.insert("sentence_count", Ika.split(text, ".").length());
    
    // Average word length
    ayanmo words = Ika.split(text, " ");
    ayanmo total_length = 0;
    fun word ninu words {
        total_length += Ika.length(Ika.trim_chars(word, ".,!?;:\"'()[]{}"));
    }
    ayanmo avg_word_length = if words.length() > 0 {
        total_length as f64 / words.length() as f64
    } else {
        0.0
    };
    stats.insert("avg_word_length", avg_word_length);
    
    // Reading time (assuming 200 words per minute)
    ayanmo reading_time_minutes = words.length() as f64 / 200.0;
    stats.insert("reading_time_minutes", reading_time_minutes);
    
    pada stats;
}

// Read and process text file
se process_text_file(file_path: Str) -> Map {
    gbiyanju {
        ayanmo content = Ogbe.file_read(file_path);
        ayanmo lines = Ika.split(content, "\n");
        
        ayanmo results = Map::::new();
        results.insert("total_lines", lines.length());
        results.insert("total_chars", Ika.char_count(content));
        results.insert("total_words", word_frequency(content).keys().length());
        
        // Find longest line
        ayanmo longest_line = "";
        ayanmo longest_length = 0;
        fun line ninu lines {
            if Ika.length(line) > longest_length {
                longest_line = line;
                longest_length = Ika.length(line);
            }
        }
        results.insert("longest_line", longest_line);
        results.insert("longest_line_length", longest_length);
        
        // Count empty lines
        ayanmo empty_lines = 0;
        fun line ninu lines {
            if Ika.trim(line).length() == 0 {
                empty_lines += 1;
            }
        }
        results.insert("empty_lines", empty_lines);
        
        pada results;
    } gba (error) {
        Irosu.kigbe("Error processing file: " + error.message);
        pada Map::::new();
    }
}

// Search and replace in file
se search_replace_in_file(file_path: Str, search: Str, replace: Str, backup: Bool) -> Bool {
    gbiyanju {
        ayanmo content = Ogbe.file_read(file_path);
        
        if !Ika.contains(content, search) {
            Irosu.fo("Search pattern not found in file");
            pada eke;
        }
        
        // Create backup if requested
        if backup {
            ayanmo backup_path = file_path + ".backup";
            Ogbe.file_write(backup_path, content);
            Irosu.fo("Backup created: " + backup_path);
        }
        
        // Perform replacement
        ayanmo new_content = Ika.replace_all(content, search, replace);
        Ogbe.file_write(file_path, new_content);
        
        ayanmo replacements = (Ika.count(content, search) - Ika.count(new_content, search)) / 
                           (Ika.length(search) - Ika.length(replace));
        
        Irosu.fo("Replaced " + Obara.to_str(replacements) + " occurrences");
        pada otito;
        
    } gba (error) {
        Irosu.kigbe("Error processing file: " + error.message);
        pada eke;
    }
}

// Split large text file
se split_text_file(file_path: Str, lines_per_file: u32, output_prefix: Str) -> Bool {
    gbiyanju {
        ayanmo content = Ogbe.file_read(file_path);
        ayanmo lines = Ika.split(content, "\n");
        
        ayanmo total_lines = lines.length();
        ayanmo file_count = (total_lines + lines_per_file - 1) / lines_per_file;
        
        Irosu.fo("Splitting " + Obara.to_str(total_lines) + " lines into " + 
                 Obara.to_str(file_count) + " files");
        
        fun file_index ninu 0..file_count {
            ayanmo start_line = file_index * lines_per_file;
            ayanmo end_line = Obara.min(start_line + lines_per_file, total_lines);
            
            ayanmo file_lines = Vec::::new();
            fun i ninu start_line..end_line {
                file_lines.push(lines[i]);
            }
            
            ayanmo file_content = Ika.join(file_lines, "\n");
            ayanmo output_file = output_prefix + Obara.to_str(file_index + 1) + ".txt";
            
            Ogbe.file_write(output_file, file_content);
            Irosu.fo("Created: " + output_file + " (" + Obara.to_str(end_line - start_line) + " lines)");
        }
        
        pada otito;
        
    } gba (error) {
        Irosu.kigbe("Error splitting file: " + error.message);
        pada eke;
    }
}

Best Practices

Unicode Handling

  • Unicode-Aware: Always use Unicode-aware functions for text processing
  • Normalization: Normalize Unicode strings before comparison
  • Encoding: Specify encoding when reading/writing files
  • Diacritics: Handle diacritics properly for Yoruba text

Performance

  • Avoid Repeated Operations: Cache string lengths and positions
  • Use Builders: Use string builders for concatenation
  • Minimize Allocations: Reuse string buffers when possible
  • Lazy Evaluation: Process large text files line by line

Cultural Sensitivity

  • Tonal Marks: Preserve Yoruba tonal marks in processing
  • Proverbs: Handle proverbs and cultural expressions appropriately
  • Context: Consider cultural context in text analysis
  • Respect: Handle sacred texts with appropriate care

Error Handling

  • Validation: Validate input strings before processing
  • Encoding Errors: Handle encoding conversion errors gracefully
  • Pattern Errors: Validate regex patterns before use
  • File Errors: Handle file I/O errors with proper messages
