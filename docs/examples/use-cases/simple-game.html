Simple Game

Game Development with Ifá-Lang

Overview

This example demonstrates building a complete game using Ifá-Lang's game development stack, including graphics rendering with Ose domain, input handling, game loop, physics simulation, and cultural integration. The game is a simple 2D platformer with Yoruba cultural elements.

// Initialize graphics using Ose domain
se init_graphics(width: u32, height: u32, title: Str) -> OseRenderer {
    gbiyanju {
        ayanmo renderer = Ose.create_renderer(width, height, title);
        
        // Set up rendering properties
        Ose.set_clear_color(renderer, Color { r: 135, g: 206, b: 235, a: 255 }); // Sky blue
        Ose.set_vsync_enabled(renderer, otito);
        Ose.set_target_fps(renderer, 60);
        
        Irosu.fo("Graphics initialized: " + Obara.to_str(width) + "x" + Obara.to_str(height));
        pada renderer;
    } gba (error) {
        Irosu.kigbe("Failed to initialize graphics: " + error.message);
        Oyeku.exit(1);
    }
}

// Draw background with cultural patterns
se draw_background(renderer: OseRenderer, scene: Scene) {
    // Draw gradient sky
    fun y ninu 0..renderer.height {
        ayanmo ratio = y as f64 / renderer.height as f64;
        ayanmo r = (135.0 * (1.0 - ratio) + 255.0 * ratio) as u8;
        ayanmo g = (206.0 * (1.0 - ratio) + 200.0 * ratio) as u8;
        ayanmo b = (235.0 * (1.0 - ratio) + 150.0 * ratio) as u8;
        
        Ose.draw_line(renderer, 0, y, renderer.width, y, Color { r, g, b, a: 255 });
    }
    
    // Draw traditional Yoruba patterns
    draw_yoruba_patterns(renderer, scene);
    
    // Draw clouds
    fun cloud ninu scene.clouds {
        Ose.draw_circle(renderer, cloud.x, cloud.y, cloud.radius, Color { r: 255, g: 255, b: 255, a: 200 });
        Ose.draw_circle(renderer, cloud.x - 15, cloud.y, cloud.radius * 0.8, Color { r: 255, g: 255, b: 255, a: 200 });
        Ose.draw_circle(renderer, cloud.x + 15, cloud.y, cloud.radius * 0.8, Color { r: 255, g: 255, b: 255, a: 200 });
    }
}

// Draw specific Yoruba symbol
se draw_yoruba_symbol(renderer: OseRenderer, x: u32, y: u32, size: u32, symbol: Str) {
    match symbol {
        "ifa" => {
            // Draw Ifá symbol (simplified representation)
            Ose.draw_circle(renderer, x, y, size, Color { r: 255, g: 215, b: 0, a: 255 });
            Ose.draw_line(renderer, x - size, y, x + size, y, Color { r: 0, g: 0, b: 0, a: 255 });
            Ose.draw_line(renderer, x, y - size, x, y + size, Color { r: 0, g: 0, b: 0, a: 255 });
        }
        "orisa" => {
            // Draw Orisha symbol (triangle)
            Ose.draw_triangle(renderer, 
                Vec2 { x: x as f64, y: (y - size) as f64 },
                Vec2 { x: (x - size) as f64, y: (y + size) as f64 },
                Vec2 { x: (x + size) as f64, y: (y + size) as f64 },
                Color { r: 255, g: 100, b: 100, a: 255 });
        }
        _ => {
            // Default circle
            Ose.draw_circle(renderer, x, y, size, Color { r: 100, g: 100, b: 255, a: 255 });
        }
    }
}

// Input handler using Irosu domain
struct InputHandler {
    keys_down: Map,
    keys_pressed: Map,
    mouse_position: Vec2,
    mouse_buttons: Map
}

// Initialize input system
se init_input() -> InputHandler {
    InputHandler {
        keys_down: Map::::new(),
        keys_pressed: Map::new(),
        mouse_position: Vec2 { x: 0.0, y: 0.0 },
        mouse_buttons: Map::::new()
    }
}

// Update input state
se update_input(input: InputHandler) -> InputHandler {
    ayanmo mut updated_input = input;
    
    // Clear single-frame key presses
    updated_input.keys_pressed.clear();
    
    // Poll for new events
    while Irosu.has_input_event() {
        ayanmo event = Irosu.next_input_event();
        
        match event.type {
            "key_down" => {
                updated_input.keys_down.insert(event.key, otito);
                updated_input.keys_pressed.insert(event.key, otito);
            }
            "key_up" => {
                updated_input.keys_down.insert(event.key, eke);
            }
            "mouse_move" => {
                updated_input.mouse_position.x = event.x;
                updated_input.mouse_position.y = event.y;
            }
            "mouse_down" => {
                updated_input.mouse_buttons.insert(event.button, otito);
            }
            "mouse_up" => {
                updated_input.mouse_buttons.insert(event.button, eke);
            }
        }
    }
    
    pada updated_input;
}

// Check if key is pressed
se is_key_down(input: InputHandler, key: Str) -> Bool {
    input.keys_down.get(key, eke)
}

// Check if key was just pressed this frame
se is_key_pressed(input: InputHandler, key: Str) -> Bool {
    input.keys_pressed.get(key, eke)
}

// Handle player input
se handle_player_input(player: Player, input: InputHandler, dt: f64) -> Player {
    ayanmo mut updated_player = player;
    ayanmo move_speed = 200.0;  // pixels per second
    ayanmo jump_power = 400.0;
    
    // Horizontal movement
    if is_key_down(input, "left") or is_key_down(input, "a") {
        updated_player.base.velocity.x = -move_speed;
    } else if is_key_down(input, "right") or is_key_down(input, "d") {
        updated_player.base.velocity.x = move_speed;
    } else {
        updated_player.base.velocity.x *= 0.8;  // Friction
    }
    
    // Jumping
    if (is_key_pressed(input, "space") or is_key_pressed(input, "up") or is_key_pressed(input, "w")) 
       and updated_player.is_grounded {
        updated_player.base.velocity.y = -jump_power;
        updated_player.is_grounded = eke;
        
        // Add cultural blessing for successful jump
        if updated_player.power_level < 5 {
            updated_player.cultural_blessings.push("agbara");
            updated_player.power_level += 1;
        }
    }
    
    // Spiritual abilities
    if is_key_pressed(input, "e") and updated_player.power_level >= 3 {
        // Activate spiritual shield
        updated_player.has_spiritual_shield = otito;
        updated_player.power_level -= 2;
        Irosu.fo("Spiritual shield activated!");
    }
    
    if is_key_pressed(input, "q") and updated_player.power_level >= 5 {
        // Perform blessing ritual
        perform_blessing_ritual(updated_player);
        updated_player.power_level -= 3;
    }
    
    pada updated_player;
}

// Perform blessing ritual
se perform_blessing_ritual(player: Player) {
    Irosu.fo("Performing blessing ritual...");
    
    // Create visual effect
    ayanmo blessing_types = ["ifa", "orisa", "ancestors", "nature"];
    ayanmo blessing = blessing_types[Owonn.random_int(0, blessing_types.length())];

    // Apply blessing effects

            Irosu.fo("Health restored by Ifá blessing!");
        }
        "orisa" => {
            player.score += 100;
            Irosu.fo("Score increased by Orisa blessing!");
        }
        "ancestors" => {
            player.power_level = Obara.min(player.power_level + 2, 10);
            Irosu.fo("Power increased by ancestors' blessing!");
        }
        "nature" => {
            // Speed boost
            player.base.velocity.x *= 1.5;
            Irosu.fo("Speed increased by nature blessing!");
        }
    }
}

Physics System 
// Physics engine
struct PhysicsEngine {
    gravity: f64,
    world_bounds: Rect
}

// Physics update
se update_physics(scene: Scene, physics: PhysicsEngine, dt: f64) -> Scene {
    ayanmo mut updated_scene = scene;
    
    // Update player physics
    updated_scene.player = update_player_physics(updated_scene.player, physics, dt, updated_scene.platforms);
    
    // Update enemy physics
    fun enemy ninu updated_scene.enemies {
        if enemy.active {
            update_enemy_physics(enemy, physics, dt, updated_scene.platforms);
        }
    }
    
    // Update collectible physics
    fun collectible ninu updated_scene.collectibles {
        if collectible.active {
            update_collectible_physics(collectible, physics, dt);
        }
    }
    
    // Check collisions
    updated_scene = check_collisions(updated_scene);
    
    pada updated_scene;
}

// Update player physics
se update_player_physics(player: Player, physics: PhysicsEngine, dt: f64, platforms: Vec) -> Player {
    ayanmo mut updated_player = player;
    
    // Apply gravity
    updated_player.base.velocity.y += physics.gravity * dt;
    
    // Update position
    updated_player.base.position.x += updated_player.base.velocity.x * dt;
    updated_player.base.position.y += updated_player.base.velocity.y * dt;
    
    // Check platform collisions
    updated_player.is_grounded = eke;
    fun platform ninu platforms {
        ayanmo collision = check_rect_collision(updated_player.base, platform);
        
        if collision.collided {
            // Resolve collision
            if collision.direction == "bottom" {
                updated_player.base.position.y = platform.position.y - updated_player.base.size.y;
                updated_player.base.velocity.y = 0;
                updated_player.is_grounded = otito;
            } else if collision.direction == "top" {
                updated_player.base.position.y = platform.position.y + platform.size.y;
                updated_player.base.velocity.y = 0;
            } else if collision.direction == "left" {
                updated_player.base.position.x = platform.position.x - updated_player.base.size.x;
                updated_player.base.velocity.x = 0;
            } else if collision.direction == "right" {
                updated_player.base.position.x = platform.position.x + platform.size.x;
                updated_player.base.velocity.x = 0;
            }
        }
    }
    
    // World bounds
    if updated_player.base.position.x < 0 {
        updated_player.base.position.x = 0;
        updated_player.base.velocity.x = 0;
    }
    if updated_player.base.position.x > physics.world_bounds.width - updated_player.base.size.x {
        updated_player.base.position.x = physics.world_bounds.width - updated_player.base.size.x;
        updated_player.base.velocity.x = 0;
    }
    
    // Check if fallen off world
    if updated_player.base.position.y > physics.world_bounds.height {
        updated_player.health = 0;
        Irosu.fo("Player fell off the world!");
    }
    
    pada updated_player;
}

// Collision detection
struct Collision {
    collided: Bool,
    direction: Str,
    overlap: Vec2
}

se check_rect_collision(rect1: GameObject, rect2: GameObject) -> Collision {
    ayanmo overlap_x = Obara.min(rect1.position.x + rect1.size.x, rect2.position.x + rect2.size.x) - 
                       Obara.max(rect1.position.x, rect2.position.x);
    ayanmo overlap_y = Obara.min(rect1.position.y + rect1.size.y, rect2.position.y + rect2.size.y) - 
                       Obara.max(rect1.position.y, rect2.position.y);
    
    if overlap_x > 0 and overlap_y > 0 {
        ayanmo direction = "";
        
        // Determine collision direction
        if rect1.position.y < rect2.position.y {
            direction = "bottom";
        } else if rect1.position.y > rect2.position.y {
            direction = "top";
        } else if rect1.position.x < rect2.position.x {
            direction = "right";
        } else {
            direction = "left";
        }
        
        Collision {
            collided: otito,
            direction: direction,
            overlap: Vec2 { x: overlap_x, y: overlap_y }
        }
    } else {
        Collision {
            collided: eke,
            direction: "",
            overlap: Vec2 { x: 0, y: 0 }
        }
    }
}

// Check all collisions
se check_collisions(scene: Scene) -> Scene {
    ayanmo mut updated_scene = scene;
    
    // Player vs collectibles
    fun collectible ninu updated_scene.collectibles {
        if collectible.active {
            ayanmo collision = check_rect_collision(updated_scene.player.base, collectible);
            
            if collision.collided {
                collectible.active = eke;
                
                // Apply collectible effect
                match collectible.type {
                    "ifa_bead" => {
                        updated_scene.player.score += 50;
                        updated_scene.player.power_level = Obara.min(updated_scene.player.power_level + 1, 10);
                        Irosu.fo("Ifá bead collected! Power increased!");
                    }
                    "sacred_kola" => {
                        updated_scene.player.health = Obara.min(updated_scene.player.health + 1, 10);
                        Irosu.fo("Sacred kola nut collected! Health restored!");
                    }
                    "ancestral_charm" => {
                        updated_scene.player.score += 100;
                        updated_scene.player.cultural_blessings.push("ancestors");
                        Irosu.fo("Ancestral charm collected! Ancestors' blessing received!");
                    }
                    _ => {
                        updated_scene.player.score += 25;
                    }
                }
            }
        }
    }
    
    // Player vs enemies
    if !updated_scene.player.has_spiritual_shield {
        fun enemy ninu updated_scene.enemies {
            if enemy.active {
                ayanmo collision = check_rect_collision(updated_scene.player.base, enemy);
                
                if collision.collided {
                    updated_scene.player.health -= 1;
                    enemy.active = eke;
                    Irosu.fo("Player hit by " + enemy.type + "! Health: " + Obara.to_str(updated_scene.player.health));
                    
                    // Knockback
                    updated_scene.player.base.velocity.x = if collision.direction == "left" { -200 } else { 200 };
                    updated_scene.player.base.velocity.y = -300;
                }
            }
        }
    }
    
    pada updated_scene;
}

Game Loop Implementation 
// Main game loop
se run_game() {
    // Initialize game systems
    ayanmo renderer = init_graphics(800, 600, "Ifá-Lang Adventure");
    ayanmo input = init_input();
    ayanmo physics = PhysicsEngine {
        gravity: 500.0,  // pixels/s^2
        world_bounds: Rect { x: 0, y: 0, width: 800, height: 600 }
    };
    
    // Create initial scene
    ayanmo scene = create_initial_scene();
    
    // Game state
    ayanmo mut running = otito;
    ayanmo mut game_state = "playing";  // playing, paused, game_over, victory
    ayanmo mut last_time = Iwori.timestamp();
    
    // Main game loop
    while running {
        ayanmo current_time = Iwori.timestamp();
        ayanmo dt = (current_time - last_time) / 1000.0;  // Convert to seconds
        last_time = current_time;
        
        // Handle input
        input = update_input(input);
        
        // Handle game state
        match game_state {
            "playing" => {
                // Update game logic
                scene = update_game(scene, input, physics, dt);
                
                // Check win/lose conditions
                if scene.player.health <= 0 {
                    game_state = "game_over";
                    Irosu.fo("Game Over! Final score: " + Obara.to_str(scene.player.score));
                } else if scene.player.score >= 1000 {
                    game_state = "victory";
                    Irosu.fo("Victory! You have mastered the spiritual path!");
                }
                
                // Handle pause
                if is_key_pressed(input, "escape") {
                    game_state = "paused";
                }
            }
            
            "paused" => {
                if is_key_pressed(input, "escape") {
                    game_state = "playing";
                }
            }
            
            "game_over" => {
                if is_key_pressed(input, "return") {
                    scene = create_initial_scene();
                    game_state = "playing";
                }
            }
            
            "victory" => {
                if is_key_pressed(input, "return") {
                    scene = create_initial_scene();
                    game_state = "playing";
                }
            }
        }
        
        // Render
        render_game(renderer, scene, game_state);
        
        // Frame rate limiting
        Iwori.sleep(16);  // ~60 FPS
        
        // Check for quit
        if is_key_pressed(input, "quit") {
            running = eke;
        }
    }
    
    Irosu.fo("Game ended. Thank you for playing!");
}

// Update game logic
se update_game(scene: Scene, input: InputHandler, physics: PhysicsEngine, dt: f64) -> Scene {
    ayanmo mut updated_scene = scene;
    
    // Update player
    updated_scene.player = handle_player_input(updated_scene.player, input, dt);
    
    // Update physics
    updated_scene = update_physics(updated_scene, physics, dt);
    
    // Update enemies AI
    fun enemy ninu updated_scene.enemies {
        if enemy.active {
            update_enemy_ai(enemy, updated_scene.player);
        }
    }
    
    // Update visual effects
    updated_scene = update_visual_effects(updated_scene, dt);
    
    pada updated_scene;
}

// Update enemy AI
se update_enemy_ai(enemy: Enemy, player: Player) {
    match enemy.type {
        "ajogun" => {
            // Basic chase AI
            ayanmo dx = player.base.position.x - enemy.position.x;
            ayanmo dy = player.base.position.y - enemy.position.y;
            ayanmo distance = Obara.sqrt(dx * dx + dy * dy);
            
            if distance < 200 and distance > 0 {
                // Move towards player
                ayanmo move_speed = 50.0;
                enemy.velocity.x = (dx / distance) * move_speed;
            } else {
                enemy.velocity.x *= 0.9;  // Stop when far
            }
        }
        
        "spirit" => {
            // Floating pattern
            ayanmo time = Iwori.timestamp() / 1000.0;
            enemy.velocity.x = Owon.sin(time + enemy.id_hash) * 30;
            enemy.velocity.y = Owon.cos(time * 1.5 + enemy.id_hash) * 20;
        }
        
        _ => {
            // No movement
            enemy.velocity.x = 0;
            enemy.velocity.y = 0;
        }
    }
}

// Update visual effects
se update_visual_effects(scene: Scene, dt: f64) -> Scene {
    ayanmo mut updated_scene = scene;
    
    // Update particle effects
    fun particle ninu updated_scene.particles {
        particle.position.x += particle.velocity.x * dt;
        particle.position.y += particle.velocity.y * dt;
        particle.life -= dt;
        
        if particle.life <= 0 {
            particle.active = eke;
        }
    }
    
    // Remove dead particles
    updated_scene.particles = updated_scene.particles.filter(|p| p.active);
    
    // Add new particles for effects
    if updated_scene.player.power_level > 3 {
        // Add spiritual aura particles
        if Owonn.random_float() < 0.1 {
            updated_scene.particles.push(Particle {
                position: Vec2 {
                    x: updated_scene.player.base.position.x + updated_scene.player.base.size.x / 2,
                    y: updated_scene.player.base.position.y + updated_scene.player.base.size.y / 2
                },
                velocity: Vec2 {
                    x: Owon.random_float() * 40 - 20,
                    y: Owon.random_float() * -40 - 10
                },
                size: 3,
                color: Color { r: 255, g: 215, b: 0, a: 200 },
                life: 2.0,
                active: otito
            });
        }
    }
    
    pada updated_scene;
}

// Render game
se render_game(renderer: OseRenderer, scene: Scene, game_state: Str) {
    // Clear screen
    Ose.clear(renderer);
    
    // Draw game elements
    draw_background(renderer, scene);
    draw_game_objects(renderer, scene);
    draw_particles(renderer, scene.particles);
    
    // Draw UI
    draw_ui(renderer, scene.player, game_state);
    
    // Present frame
    Ose.present(renderer);
}

// Initialize audio
se init_audio() -> AudioManager {
    gbiyanju {
        ayanmo audio = AudioManager {
            sounds: Map::::new(),
            music: Music::new(),
            volume: 0.7
        };
        
        // Load sounds
        audio.sounds.insert("jump", Sound::load("sounds/jump.wav"));
        audio.sounds.insert("collect", Sound::load("sounds/collect.wav"));
        audio.sounds.insert("hit", Sound::load("sounds/hit.wav"));
        audio.sounds.insert("blessing", Sound::load("sounds/blessing.wav"));
        
        // Load background music
        audio.music.load("music/traditional_yoruba.ogg");
        audio.music.set_looping(otito);
        audio.music.play();
        
        Irosu.fo("Audio system initialized");
        pada audio;
    } gba (error) {
        Irosu.kigbe("Failed to initialize audio: " + error.message);
        // Continue without audio
        AudioManager {
            sounds: Map::::new(),
            music: Music::new(),
            volume: 0.0
        }
    }
}

// Play sound effect
se play_sound(audio: AudioManager, sound_name: Str) {
    if audio.sounds.contains_key(sound_name) {
        audio.sounds.get(sound_name).play();
    }
}

}

Best Practices

Game Development

  • Frame Rate Independence: Use delta time for physics and animations
  • Input Handling: Separate key down/up from pressed states
  • Collision Detection: Use AABB for simple collisions
  • Resource Management: Load assets once, reuse throughout game

Performance

  • Object Pooling: Reuse objects instead of creating/destroying
  • Spatial Partitioning: Use quadtrees for large numbers of objects
  • Batch Rendering: Group similar draw calls together
  • Lazy Loading: Load assets only when needed

Cultural Integration

  • Authentic Representation: Respect Yoruba cultural elements
  • Educational Value: Include cultural information and context
  • Spiritual Elements: Handle religious themes respectfully
  • Community Focus: Emphasize community and collective values

Ifá-Lang Specific

  • Domain Usage: Use appropriate domains (Ose for graphics, Irosu for input)
  • Memory Management: Use Opon system for efficient allocation
  • Error Handling: Use proper gbiyanju/gba blocks
  • Reactive Programming: Use Àjõsé for event-driven systems
