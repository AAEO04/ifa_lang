REST API Server

Web Service Development with Ifá-Lang

Overview

This example demonstrates building a complete REST API server using Ifá-Lang's web development capabilities. The server includes CRUD operations, middleware, authentication, database integration, and proper error handling following Ifá-Lang's cultural and architectural principles.

// Validation functions
ese validate_user_request(request: CreateUserRequest) -> Result<(), Str> {
    // Username validation
    if Ika.length(request.username) < 3 {
        pada Err("Username must be at least 3 characters long");
    }
    
    if Ika.length(request.username) > 50 {
        pada Err("Username must not exceed 50 characters");
    }
    
    if !Ika.matches(request.username, "^[a-zA-Z0-9_]+$") {
        pada Err("Username can only contain letters, numbers, and underscores");
    }
    
    // Email validation
    if !Ika.contains(request.email, "@") or !Ika.contains(request.email, ".") {
        pada Err("Invalid email format");
    }
    
    // Password validation
    if Ika.length(request.password) < 8 {
        pada Err("Password must be at least 8 characters long");
    }
    
    if !Ika.matches(request.password, "^(?=.*[a-z])(?=.*[A-Z])(?=.*\\d).+$") {
        pada Err("Password must contain at least one lowercase letter, one uppercase letter, and one number");
    }
    
    pada Ok(());
}

ese validate_post_request(request: CreatePostRequest) -> Result<(), Str> {
    if Ika.length(Ika.trim(request.title)) == 0 {
        pada Err("Title cannot be empty");
    }
    
    if Ika.length(request.title) > 200 {
        pada Err("Title must not exceed 200 characters");
    }
    
    if Ika.length(Ika.trim(request.content)) == 0 {
        pada Err("Content cannot be empty");
    }
    
    if Ika.length(request.content) > 10000 {
        pada Err("Content must not exceed 10,000 characters");
    }
    
    if request.tags.length() > 10 {
        pada Err("Cannot have more than 10 tags");
    }
    
    fun tag ninu request.tags {
        if Ika.length(tag) > 50 {
            pada Err("Tag '" + tag + "' exceeds 50 characters");
        }
    }
    
    pada Ok(());
}

// Database migrations
ese run_migrations(connection: DatabaseConnection) {
    ayanmo migrations = [
        "CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            username TEXT UNIQUE NOT NULL,
            email TEXT UNIQUE NOT NULL,
            password_hash TEXT NOT NULL,
            created_at INTEGER NOT NULL,
            updated_at INTEGER NOT NULL,
            is_active BOOLEAN NOT NULL DEFAULT 1
        )",
        
        "CREATE TABLE IF NOT EXISTS blog_posts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            author_id INTEGER NOT NULL,
            created_at INTEGER NOT NULL,
            updated_at INTEGER NOT NULL,
            published BOOLEAN NOT NULL DEFAULT 0,
            FOREIGN KEY (author_id) REFERENCES users(id)
        )",
        
        "CREATE TABLE IF NOT EXISTS post_tags (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            post_id INTEGER NOT NULL,
            tag TEXT NOT NULL,
            FOREIGN KEY (post_id) REFERENCES blog_posts(id) ON DELETE CASCADE
        )",
        
        "CREATE INDEX IF NOT EXISTS idx_users_username ON users(username)",
        "CREATE INDEX IF NOT EXISTS idx_users_email ON users(email)",
        "CREATE INDEX IF NOT EXISTS idx_blog_posts_author ON blog_posts(author_id)",
        "CREATE INDEX IF NOT EXISTS idx_blog_posts_published ON blog_posts(published)",
        "CREATE INDEX IF NOT EXISTS idx_post_tags_post ON post_tags(post_id)",
        "CREATE INDEX IF NOT EXISTS idx_post_tags_tag ON post_tags(tag)"
    ];
    
    fun migration ninu migrations {
        gbiyanju {
            connection.execute(migration);
            Irosu.fo("Migration executed: " + migration);
        } gba (error) {
            Irosu.kigbe("Migration failed: " + error.message);
            Oyeku.exit(1);
        }
    }
}

se update_user(db: Database, user_id: u64, request: UpdateUserRequest) -> Result {
    gbiyanju {
        ayanmo user = get_user_by_id(db, user_id);
        if user.is_none() {
            pada Err("User not found");
        }
        
        ayanmo mut user = user.unwrap();
        ayanmo now = Iwori.timestamp();
        
        // Build update query dynamically
        ayanmo updates = Vec::::new();
        ayanmo values = Vec::::new();
        
        if request.username.is_some() {
            ayanmo username = request.username.unwrap();
            if Ika.length(username) < 3 or Ika.length(username) > 50 {
                pada Err("Username must be between 3 and 50 characters");
            }
            
            // Check if username already exists
            ayanmo existing = get_user_by_username(db, username);
            if existing.is_some() and existing.unwrap().id != user_id {
                pada Err("Username already exists");
            }
            
            updates.push("username = ?");
            values.push(username);
            user.username = username;
        }
        
        if request.email.is_some() {
            ayanmo email = request.email.unwrap();
            if !Ika.contains(email, "@") or !Ika.contains(email, ".") {
                pada Err("Invalid email format");
            }
            
            // Check if email already exists
            ayanmo existing = get_user_by_email(db, email);
            if existing.is_some() and existing.unwrap().id != user_id {
                pada Err("Email already exists");
            }
            
            updates.push("email = ?");
            values.push(email);
            user.email = email;
        }
        
        if request.is_active.is_some() {
            updates.push("is_active = ?");
            values.push(Obara.to_str(request.is_active.unwrap()));
            user.is_active = request.is_active.unwrap();
        }
        
        if updates.length() > 0 {
            updates.push("updated_at = ?");
            values.push(Obara.to_str(now));
            values.push(Obara.to_str(user_id));
            
            ayanmo query = "UPDATE users SET " + Ika.join(updates, ", ") + " WHERE id = ?";
            connection.execute(query, values);
            
            user.updated_at = now;
        }
        
        pada user;
    } gba (error) {
        pada Err("Failed to update user: " + error.message);
    }
}

// JWT token structure
struct JWTToken {
    user_id: u64,
    username: Str,
    email: Str,
    issued_at: u64,
    expires_at: u64
}

// Authentication middleware result
struct AuthResult {
    user: User,
    token: Str
}

// Generate JWT token
se generate_jwt_token(user: User) -> Str {
    ayanmo now = Iwori.timestamp();
    ayanmo expires_at = now + (24 * 60 * 60); // 24 hours
    
    ayanmo payload = JWTToken {
        user_id: user.id,
        username: user.username,
        email: user.email,
        issued_at: now,
        expires_at: expires_at
    };
    
    ayanmo payload_json = Ika.to_json(payload);
    ayanmo signature = Irete.hmac_sha256(payload_json, "secret_key");
    
    // Base64 encode payload and signature
    ayanmo encoded_payload = Irete.base64_encode(payload_json);
    ayanmo encoded_signature = Irete.base64_encode(signature);
    
    pada encoded_payload + "." + encoded_signature;
}

// Verify JWT token
se verify_jwt_token(token: Str) -> Option {
    gbiyanju {
        ayanmo parts = Ika.split(token, ".");
        if parts.length() != 2 {
            pada None;
        }
        
        ayanmo encoded_payload = parts[0];
        ayanmo encoded_signature = parts[1];
        
        ayanmo payload_json = Irete.base64_decode(encoded_payload);
        ayanmo signature = Irete.base64_decode(encoded_signature);
        
        // Verify signature
        ayanmo expected_signature = Irete.hmac_sha256(payload_json, "secret_key");
        if signature != expected_signature {
            pada None;
        }
        
        // Parse payload
        ayanmo payload = Ika.from_json(payload_json);
        
        // Check expiration
        if Iwori.timestamp() > payload.expires_at {
            pada None;
        }
        
        pada Some(payload);
    } gba (error) {
        Irosu.kigbe("JWT verification failed: " + error.message);
        pada None;
    }
}

// Authentication middleware
se authenticate_middleware(request: Request, next: ese (Request) -> Response) -> Response {
    ayanmo auth_header = request.get_header("Authorization");
    
    if auth_header.is_none() {
        pada Response::unauthorized("Missing authorization header");
    }
    
    ayanmo header = auth_header.unwrap();
    if !Ika.starts_with(header, "Bearer ") {
        pada Response::unauthorized("Invalid authorization format");
    }
    
    ayanmo token = Ika.substring(header, 7); // Remove "Bearer "
    
    ayanmo jwt_payload = verify_jwt_token(token);
    if jwt_payload.is_none() {
        pada Response::unauthorized("Invalid or expired token");
    }
    
    ayanmo payload = jwt_payload.unwrap();
    ayanmo user = get_user_by_id(database, payload.user_id);
    
    if user.is_none() {
        pada Response::unauthorized("User not found");
    }
    
    // Add user to request context
    request.set_context("user", user.unwrap());
    
    // Call next middleware/handler
    next(request);
}

// User login
se login_user(db: Database, username: Str, password: Str) -> Result {
    gbiyanju {
        ayanmo user = get_user_by_username(db, username);
        if user.is_none() {
            pada Err("Invalid username or password");
        }
        
        ayanmo user_data = user.unwrap();
        if !user_data.is_active {
            pada Err("Account is deactivated");
        }
        
        if !Irete.verify_password(password, user_data.password_hash) {
            pada Err("Invalid username or password");
        }
        
        ayanmo token = generate_jwt_token(user_data);
        
        pada AuthResult {
            user: user_data,
            token: token
        };
    } gba (error) {
        pada Err("Login failed: " + error.message);
    }
}

// Password hashing
se hash_password(password: Str) -> Str {
    ayanmo salt = Irete.generate_salt();
    ayanmo hash = Irete.pbkdf2(password, salt, 10000);
    pada salt + ":" + hash;
}

se verify_password(password: Str, hash: Str) -> Bool {
    gbiyanju {
        ayanmo parts = Ika.split(hash, ":");
        if parts.length() != 2 {
            pada eke;
        }
        
        ayanmo salt = parts[0];
        ayanmo stored_hash = parts[1];
        ayanmo computed_hash = Irete.pbkdf2(password, salt, 10000);
        
        pada computed_hash == stored_hash;
    } gba (error) {
        Irosu.kigbe("Password verification failed: " + error.message);
        pada eke;
    }
}

// Authentication handlers
se login_handler(request: Request) -> Response {
    gbiyanju {
        ayanmo body = request.get_json_body();
        if body.is_none() {
            pada Response::bad_request("Invalid JSON body");
        }
        
        ayanmo json_data = body.unwrap();
        ayanmo username = json_data.get("username");
        ayanmo password = json_data.get("password");
        
        if username.is_none() or password.is_none() {

// CORS middleware
se cors_middleware(request: Request, next: ese (Request) -> Response) -> Response {
    ayanmo response = next(request);
    
    // Add CORS headers
    response.set_header("Access-Control-Allow-Origin", "*");
    response.set_header("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS");
    response.set_header("Access-Control-Allow-Headers", "Content-Type, Authorization");
    response.set_header("Access-Control-Max-Age", "86400");
    
    // Handle preflight requests
    if request.method == "OPTIONS" {
        pada Response::no_content(204);
    }
    
    pada response;
}

// Logging middleware
se logging_middleware(request: Request, next: ese (Request) -> Response) -> Response {
    ayanmo start_time = Iwori.timestamp();
    
    // Log request
    Irosu.fo("[" + Iwori.format_timestamp(start_time) + "] " + 
             request.method + " " + request.path + 
             " from " + request.get_remote_addr());
    
    ayanmo response = next(request);
    
    ayanmo end_time = Iwori.timestamp();
    ayanmo duration = end_time - start_time;
    
    // Log response
    Irosu.fo("[" + Iwori.format_timestamp(end_time) + "] " + 
             Obara.to_str(response.status_code) + " " + 
             request.method + " " + request.path + 
             " - " + Obara.to_str(duration) + "ms");
    
    pada response;
}

// Rate limiting middleware
se rate_limit_middleware(request: Request, next: ese (Request) -> Response) -> Response {
    ayanmo client_ip = request.get_remote_addr();
    ayanmo current_time = Iwori.timestamp();
    ayanmo window_size = 60; // 1 minute window
    
    // Simple in-memory rate limiting (in production, use Redis)
    static mut RATE_LIMITS: Map> = Map::>::new();
    
    unsafe {
        if !RATE_LIMITS.contains_key(client_ip) {
            RATE_LIMITS.insert(client_ip, Vec::::new());
        }
        
        ayanmo requests = RATE_LIMITS.get_mut(client_ip).unwrap();
        
        // Remove old requests outside the window
        requests.retain(|timestamp| *timestamp > current_time - window_size);
        
        // Check if rate limit exceeded (100 requests per minute)
        if requests.length() >= 100 {
            ayanmo response = ErrorResponse {
                success: eke,
                error: "Rate limit exceeded",
                code: 429,
                timestamp: current_time
            };
            
            pada Response::json(429, response);
        }
        
        // Add current request
        requests.push(current_time);
    }
    
    pada next(request);
}

// Error handling middleware
se error_handling_middleware(request: Request, next: ese (Request) -> Response) -> Response {
    gbiyanju {
        pada next(request);
    } gba (error) {
        Irosu.kigbe("Unhandled error: " + error.message);
        
        ayanmo response = ErrorResponse {
            success: eke,
            error: "Internal server error",
            code: 500,
            timestamp: Iwori.timestamp()
        };
        
        pada Response::json(500, response);
    }
}

Best Practices

Security

  • Input Validation: Validate all input data with proper error messages
  • Authentication: Use JWT tokens with proper expiration and refresh
  • Authorization: Implement proper role-based access control
  • Password Security: Use strong hashing algorithms (PBKDF2, bcrypt, Argon2)
  • HTTPS: Always use HTTPS in production with proper certificates
  • CORS: Configure CORS properly for your frontend domains

Performance

  • Database Connection Pooling: Use connection pools for database access
  • Caching: Implement caching for frequently accessed data
  • Pagination: Always paginate list endpoints to prevent large responses
  • Async Operations: Use async/await for I/O-bound operations
  • Rate Limiting: Implement rate limiting to prevent abuse

Reliability

  • Error Handling: Provide consistent error responses with proper HTTP status codes
  • Logging: Implement structured logging with appropriate log levels
  • Health Checks: Provide health check endpoints for monitoring
  • Graceful Shutdown: Handle shutdown signals properly
  • Circuit Breakers: Implement circuit breakers for external services

Testing

  • Unit Tests: Test individual functions and methods
  • Integration Tests: Test API endpoints with real database
  • Load Testing: Test performance under load
  • Security Testing: Test for common vulnerabilities
  • Contract Testing: Test API contracts with consumers
