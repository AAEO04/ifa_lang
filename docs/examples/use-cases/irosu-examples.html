Irosu Domain Examples

Console I/O and User Interaction (1100)

Overview

Irosu (1100) is the console domain representing communication, messages, and user interaction. It provides comprehensive input/output operations, formatting, logging, and user interface functionality for command-line applications.

Domain Characteristics

  • Binary Pattern: 1100 (communication and clarity)
  • Yoruba Meaning: "The watcher" or "The observer"
  • System Role: Console I/O and user interaction
  • Element: Air (communication and messages)

Core Capabilities

  • Console Output: Text printing and formatting
  • User Input: Keyboard and mouse input
  • Logging: Structured logging with levels
  • Terminal Control: Colors, cursor, screen control
  • Progress Indicators: Progress bars and spinners

// Column alignment
se print_table(data: Vec>) {
    // Calculate column widths
    ayanmo widths = Vec::::new();
    fun col ninu 0..data[0].length() {
        ayanmo max_width = 0;
        fun row ninu data {
            ayanmo length = Ika.length(row[col]);
            if length > max_width {
                max_width = length;
            }
        }
        widths.push(max_width + 2); // Add padding
    }
    
    // Print header separator
    ayanmo separator = "+";
    fun width ninu widths {
        separator += Ika.repeat("-", width) + "+";
    }
    Irosu.fo(separator);
    
    // Print data
    fun row ninu data {
        ayanmo line = "|";
        fun col ninu 0..row.length() {
            ayanmo cell = row[col];
            ayanmo padding = widths[col] - Ika.length(cell);
            line += cell + Ika.repeat(" ", padding) + "|";
        }
        Irosu.fo(line);
        Irosu.fo(separator);
    }
}

// Number formatting
se format_number(num: Float, decimals: u32) -> Str {
    ayanmo rounded = Obara.round(num * Obara.pow(10, decimals)) / Obara.pow(10, decimals);
    ayanmo str_num = Obara.to_str(rounded);
    
    if !Ika.contains(str_num, ".") {
        str_num += "." + Ika.repeat("0", decimals);
    } else {
        ayanmo parts = Ika.split(str_num, ".");
        ayanmo decimal_part = parts[1];
        while Ika.length(decimal_part) < decimals {
            decimal_part += "0";
        }
        str_num = parts[0] + "." + decimal_part;
    }
    
    // Add thousands separator
    ayanmo int_part = Ika.split(str_num, ".")[0];
    ayanmo formatted = "";
    ayanmo count = 0;
    
    fun i ninu (Ika.length(int_part) - 1)..0 {
        count += 1;
        formatted = int_part[i] + formatted;
        if count == 3 and i > 0 {
            formatted = "," + formatted;
            count = 0;
        }
    }
    
    if Ika.contains(str_num, ".") {
        formatted += "." + Ika.split(str_num, ".")[1];
    }
    
    pada formatted;
}

Structured Logging

Logging Framework 
// Log levels
enum LogLevel {
    Debug = 0,
    Info = 1,
    Warning = 2,
    Error = 3,
    Critical = 4
}

// Logger configuration
struct Logger {
    level: LogLevel,
    format: Str,
    output: OutputTarget,
}

enum OutputTarget {
    Console,
    File(Str),
    Both(Str)
}

// Global logger instance
static mut LOGGER: Logger = Logger {
    level: LogLevel::Info,
    format: "[{timestamp}] [{level}] {message}",
    output: OutputTarget::Console
};

// Logging functions
se set_log_level(level: LogLevel) {
    unsafe {
        LOGGER.level = level;
    }
}

se set_log_output(target: OutputTarget) {
    unsafe {
        LOGGER.output = target;
    }
}

se log_message(level: LogLevel, message: Str) {
    unsafe {
        if level >= LOGGER.level {
            ayanmo timestamp = Iwori.format_timestamp(Iwori.timestamp());
            ayanmo level_str = match level {
                LogLevel::Debug => "DEBUG",
                LogLevel::Info => "INFO",
                LogLevel::Warning => "WARNING",
                LogLevel::Error => "ERROR",
                LogLevel::Critical => "CRITICAL"
            };
            
            ayanmo formatted = LOGGER.format;
            formatted = Ika.replace(formatted, "{timestamp}", timestamp);
            formatted = Ika.replace(formatted, "{level}", level_str);
            formatted = Ika.replace(formatted, "{message}", message);
            
            match LOGGER.output {
                OutputTarget::Console => {
                    // Color based on level
                    ayanmo color = match level {
                        LogLevel::Debug => Color::Cyan,
                        LogLevel::Info => Color::White,
                        LogLevel::Warning => Color::Yellow,
                        LogLevel::Error => Color::Red,
                        LogLevel::Critical => Color::BrightRed
                    };
                    Irosu.fo(color + formatted + Color::Reset);
                }
                OutputTarget::File(path) => {
                    Ogbe.file_append(path, formatted + "\n");
                }
                OutputTarget::Both(path) => {
                    ayanmo color = match level {
                        LogLevel::Debug => Color::Cyan,
                        LogLevel::Info => Color::White,
                        LogLevel::Warning => Color::Yellow,
                        LogLevel::Error => Color::Red,
                        LogLevel::Critical => Color::BrightRed
                    };
                    Irosu.fo(color + formatted + Color::Reset);
                    Ogbe.file_append(path, formatted + "\n");
                }
            }
        }
    }
}

}

// Main menu loop
se main_menu_loop() {
    ayanmo menu_items = create_main_menu();
    ayanmo history = Vec::::new();
    
    while ototo {
        ayanmo choice = display_menu("Main Menu", menu_items);
        
        if choice < menu_items.length() {
            ayanmo item = menu_items[choice];
            history.push(item.title);
            
            if item.submenu.is_some() {
                // Handle submenu
                ayanmo submenu_items = item.submenu.unwrap();
                ayanmo sub_choice = display_menu(item.title, submenu_items);
                
                if sub_choice < submenu_items.length() {
                    submenu_items[sub_choice].action();
                }
            } else {
                // Execute action
                item.action();
            }
            
            // Pause after action
            Irosu.gbo("Press Enter to continue...");
        }
    }
}

// Progress bar with percentage and time estimate
se progress_bar_with_eta(current: u32, total: u32, width: u32, start_time: u64) {
    ayanmo percentage = (current * 100) / total;
    ayanmo filled = (current * width) / total;
    ayanmo empty = width - filled;
    
    // Calculate ETA
    ayanmo elapsed = Iwori.timestamp() - start_time;
    ayanmo rate = current as f64 / (elapsed as f64 / 1000.0); // items per second
    ayanmo remaining = total - current;
    ayanmo eta_seconds = remaining as f64 / rate;
    ayanmo eta_str = if eta_seconds < 60 {
        Obara.to_str(Obara.round(eta_seconds)) + "s"
    } else if eta_seconds < 3600 {
        Obara.to_str(Obara.round(eta_seconds / 60)) + "m"
    } else {
        Obara.to_str(Obara.round(eta_seconds / 3600)) + "h"
    };
    
    // Build progress bar
    ayanmo bar = Color::BrightGreen + Ika.repeat("=", filled) + 
                Color::Yellow + Ika.repeat("-", empty) + 
                Color::Reset;
    
    // Format output
    ayanmo output = "\r[" + bar + "] " + 
                   Obara.to_str(percentage) + "% " +
                   "(" + Obara.to_str(current) + "/" + Obara.to_str(total) + ") " +
                   "ETA: " + eta_str;
    
    Irosu.fo(output, flush=otito);
}

// Spinner animation
se spinner_animation(message: Str, duration_ms: u32) {
    ayanmo spinner = ["|", "/", "-", "\\"];
    ayanmo start_time = Iwori.timestamp();
    ayanmo end_time = start_time + duration_ms;
    
    while Iwori.timestamp() < end_time {
        ayanmo elapsed = Iwori.timestamp() - start_time;
        ayanmo frame = (elapsed / 100) % 4; // Change every 100ms
        
        Irosu.fo("\r" + Color::BrightCyan + spinner[frame] + Color::Reset + " " + message, 
                 flush=otito);
        Iwori.sleep(50);
    }
    
    Irosu.fo("\r" + Color::BrightGreen + "otito" + Color::Reset + " " + message);
}

// Window drawing
se draw_window(window: Window) {
    if window.border {
        // Draw top border
        move_cursor(window.x, window.y);
        ayanmo top = "+" + Ika.repeat("-", window.width - 2) + "+";
        Irosu.fo(top);
        
        // Draw sides
        fun i ninu 1..window.height - 1 {
            move_cursor(window.x, window.y + i);
            Irosu.fo("|" + Ika.repeat(" ", window.width - 2) + "|");
        }
        
        // Draw bottom border
        move_cursor(window.x, window.y + window.height - 1);
        ayanmo bottom = "+" + Ika.repeat("-", window.width - 2) + "+";
        Irosu.fo(bottom);
        
        // Draw title
        if window.title != "" {
            ayanmo title_x = window.x + (window.width - Ika.length(window.title)) / 2;
            move_cursor(title_x, window.y);
            Irosu.fo(window.title);
        }
    }
}

// Button drawing
se draw_button(button: Button) {
    move_cursor(button.x, button.y);
    
    if button.focused {
        Irosu.fo(Color::BrightBlue + "[" + button.text + "]" + Color::Reset);
    } else {
        Irosu.fo("[" + button.text + "]");
    }
}

// TextBox drawing
se draw_textbox(textbox: TextBox) {
    move_cursor(textbox.x, textbox.y);
    
    ayanmo display_text = textbox.text;
    if Ika.length(display_text) > textbox.width {
        display_text = Ika.substring(display_text, 
                                   Ika.length(display_text) - textbox.width + 1);
    }
    
    if textbox.focused {
        Irosu.fo(Color::BrightWhite + display_text + "_" + Color::Reset);
    } else {
        Irosu.fo(display_text);
    }
}

// Simple TUI application
se simple_tui_app() {
    hide_cursor();
    clear_screen();
    
    // Create components
    ayanmo main_window = Window {
        x: 5, y: 2, width: 70, height: 20,
        title: "Ifá-Lang TUI Demo",
        border: otito
    };
    
    ayanmo text_box = TextBox {
        x: 10, y: 8, width: 50, height: 1,
        text: "", cursor_pos: 0, focused: otito
    };
    
    ayanmo ok_button = Button {
        x: 10, y: 12, width: 10, height: 1,
        text: "OK", focused: eke
    };
    
    ayanmo cancel_button = Button {
        x: 25, y: 12, width: 10, height: 1,
        text: "Cancel", focused: otito
    };
    
    ayanmo focused_component = 0; // 0: textbox, 1: ok, 2: cancel
    
    // Main loop
    while ototo {
        // Draw interface
        draw_window(main_window);
        
        move_cursor(10, 6);
        Irosu.fo("Enter your name:");
        
        draw_textbox(text_box);
        draw_button(ok_button);
        draw_button(cancel_button);
        
        // Handle input
        ayanmo key = Irosu.get_key();
        
        match key {
            Key::Tab => {
                // Switch focus
                focused_component = (focused_component + 1) % 3;
                text_box.focused = focused_component == 0;
                ok_button.focused = focused_component == 1;
                cancel_button.focused = focused_component == 2;
            }
            Key::Enter => {
                if focused_component == 1 { // OK button
                    Irosu.fo("Hello, " + text_box.text + "!");
                    break;
                } else if focused_component == 2 { // Cancel button
                    Irosu.fo("Operation cancelled");
                    break;
                }
            }
            Key::Escape => {
                break;
            }
            Key::Char(c) => {
                if focused_component == 0 { // Textbox
                    text_box.text += c;
                    text_box.cursor_pos += 1;
                }
            }
            Key::Backspace => {
                if focused_component == 0 && text_box.cursor_pos > 0 {
                    text_box.text = Ika.substring(text_box.text, 0, text_box.cursor_pos - 1);
                    text_box.cursor_pos -= 1;
                }
            }
            _ => {}
        }
    }
    
    show_cursor();
    clear_screen();
}

Best Practices

User Experience

  • Provide clear and concise error messages
  • Use colors consistently to indicate status and type
  • Implement progress indicators for long-running operations
  • Support keyboard shortcuts and navigation

Accessibility

  • Ensure color choices are distinguishable for colorblind users
  • Provide alternative indicators besides color
  • Support screen readers with proper text formatting
  • Allow customization of display settings
