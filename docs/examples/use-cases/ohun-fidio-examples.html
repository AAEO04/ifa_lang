Use Case 1: Music Player

// music_player.ifa - Simple audio player

ayanmo playlist = [];
ayanmo current_track = 0;
ayanmo volume = 0.8;
ayanmo playing = iro;

ise add_to_playlist(path) {
    ti (!Odi.exists(path)) {
        Irosu.kigbe("File not found: " + path);
        padap?;
    }
    
    ayanmo metadata = Ohun.get_metadata(path);
    ayanmo playlist = Ogunda.push(playlist, {
        "path": path,
        "title": metadata["title"],
        "artist": metadata["artist"],
        "duration": metadata["duration"]
    });
    
    Irosu.fo("Added: " + metadata["title"]);
}

ise play() {
    ti (Ogunda.len(playlist) == 0) {
        Irosu.fo("Playlist empty");
        padap?;
    }
    
    ayanmo track = Ogunda.get(playlist, current_track);
    Ohun.play(track["path"], volume);
    ayanmo playing = otito;
    
    Irosu.fo("? Now playing: " + track["title"] + " - " + track["artist"]);
}

ise pause() {
    Ohun.pause();
    ayanmo playing = iro;
    Irosu.fo("? Paused");
}

ise next_track() {
    ayanmo current_track = Oturupon.mod(current_track + 1, Ogunda.len(playlist));
    ti (playing) { play(); }
}

ise prev_track() {
    ayanmo current_track = current_track - 1;
    ti (current_track < 0) {
        ayanmo current_track = Ogunda.len(playlist) - 1;
    }
    ti (playing) { play(); }
}

ise set_volume(vol) {
    ayanmo volume = Obara.max(0, Obara.min(1, vol));
    Ohun.set_volume(volume);
    Irosu.fo("?? Volume: " + (volume * 100) + "%");
}

// Usage
add_to_playlist("music/song1.mp3");
add_to_playlist("music/song2.mp3");
add_to_playlist("music/song3.mp3");

play();
Oyeku.sleep(5000);
next_track();
Oyeku.sleep(3000);
set_volume(0.5);
Oyeku.sleep(2000);
pause();

Use Case 3: Video Thumbnail Generator

// thumbnails.ifa - Extract video thumbnails

ise generate_thumbnail(video_path, time_seconds, output_path) {
    ti (!Odi.exists(video_path)) {
        padap? Okanran.error("Video not found: " + video_path);
    }
    
    // Get video info
    ayanmo info = Fidio.get_info(video_path);
    Irosu.fo("Video: " + info["width"] + "x" + info["height"] + 
             ", Duration: " + info["duration"] + "s");
    
    // Validate time
    ti (time_seconds >= info["duration"]) {
        ayanmo time_seconds = info["duration"] / 2;
    }
    
    // Extract frame
    ayanmo frame = Fidio.extract_frame(video_path, time_seconds);
    
    // Resize to thumbnail
    ayanmo thumbnail = Fidio.resize_image(frame, 320, 180);
    
    // Save
    Fidio.save_image(thumbnail, output_path);
    Irosu.fo("? Saved thumbnail: " + output_path);
    
    padap? output_path;
}

ise batch_thumbnails(video_dir, output_dir) {
    Odi.mkdir(output_dir);
    
    ayanmo files = Odi.list(video_dir);
    ayanmo count = 0;
    
    fun file ninu files {
        ti (Ika.ends_with(file, ".mp4") || 
            Ika.ends_with(file, ".avi") ||
            Ika.ends_with(file, ".mkv")) {
            
            ayanmo video = video_dir + "/" + file;
            ayanmo name = Ika.split(file, ".")[0];
            ayanmo thumb = output_dir + "/" + name + ".jpg";
            
            generate_thumbnail(video, 10, thumb);
            ayanmo count = count + 1;
        }
    }
    
    Irosu.fo("\n? Generated " + count + " thumbnails");
}

// Usage
batch_thumbnails("./videos", "./thumbnails");

Use Case 4: Video Playback Controller

// player.ifa - Video player with controls

ayanmo player = nil;
ayanmo current_time = 0;
ayanmo duration = 0;

ise open_video(path) {
    ayanmo player = Fidio.open(path);
    ayanmo info = Fidio.get_info(path);
    ayanmo duration = info["duration"];
    ayanmo current_time = 0;
    
    Irosu.fo("??? Opened: " + path);
    Irosu.fo("   Resolution: " + info["width"] + "x" + info["height"]);
    Irosu.fo("   Duration: " + format_time(duration));
    Irosu.fo("   FPS: " + info["fps"]);
}

ise format_time(seconds) {
    ayanmo mins = Oturupon.div(seconds, 60);
    ayanmo secs = Oturupon.mod(seconds, 60);
    padap? mins + ":" + Ika.pad_left(secs, 2, "0");
}

ise play_video() {
    Fidio.play(player);
    Irosu.fo("? Playing");
}

ise pause_video() {
    Fidio.pause(player);
    Irosu.fo("? Paused at " + format_time(current_time));
}

ise seek(seconds) {
    ayanmo new_time = Obara.max(0, Obara.min(current_time + seconds, duration));
    Fidio.seek(player, new_time);
    ayanmo current_time = new_time;
    Irosu.fo("? Seeked to " + format_time(current_time));
}

ise show_status() {
    ayanmo pos = (current_time * 30) / duration;
    ayanmo bar = Ika.repeat("¦", pos) + Ika.repeat("¦", 30 - pos);
    Irosu.fo("[" + bar + "] " + format_time(current_time) + " / " + format_time(duration));
}

// Usage
open_video("movie.mp4");
play_video();
Oyeku.sleep(3000);
show_status();
seek(60);  // Forward 1 minute
show_status();
seek(-30);  // Back 30 seconds
show_status();
pause_video();
