// Main entry point
ese main() {
    ayanmo args = Ogbe.parse_args();
    
    // Handle empty arguments
    if args.length() == 0 {
        show_help();
        Oyeku.exit(0);
    }
    
    // Parse global options
    ayanmo config = parse_global_options(args);
    
    // Execute command
    gbiyanju {
        execute_command(config, args);
    } gba (error) {
        Irosu.kigbe("Error: " + error.message);
        if config.verbose {
            Irosu.kigbe("Stack trace: " + error.stack_trace);
        }
        Oyeku.exit(1);
    }
}

// Parse global options
se parse_global_options(args: Vec) -> Config {
    ayanmo mut config = Config::load_default();
    
    fun i ninu 0..args.length() {
        ayanmo arg = args[i];
        
        if arg == "--help" or arg == "-h" {
            show_help();
            Oyeku.exit(0);
        } else if arg == "--version" or arg == "-V" {
            show_version();
            Oyeku.exit(0);
        } else if arg == "--verbose" or arg == "-v" {
            config.verbose = otito;
        } else if arg == "--quiet" or arg == "-q" {
            config.quiet = otito;
        } else if arg == "--config" {
            if i + 1 < args.length() {
                config.config_file = args[i + 1];
                config = Config::load_file(config.config_file);
            } else {
                Irosu.kigbe("Error: --config requires a file path");
                Oyeku.exit(1);
            }
        } else if arg == "--color" {
            if i + 1 < args.length() {
                ayanmo color_mode = args[i + 1];
                if color_mode == "auto" or color_mode == "always" or color_mode == "never" {
                    config.color_mode = color_mode;
                } else {
                    Irosu.kigbe("Error: --color must be auto, always, or never");
                    Oyeku.exit(1);
                }
            } else {
                Irosu.kigbe("Error: --color requires a mode");
                Oyeku.exit(1);
            }
        }
    }
    
    pada config;
}

// Execute command
se execute_command(config: Config, args: Vec) {
    if args.length() == 0 {
        show_help();
        return;
    }
    
    ayanmo command = args[0];
    ayanmo command_args = args.slice(1, args.length());
    
    match command {
        "system" => commands::system_command(config, command_args),
        "file" => commands::file_command(config, command_args),
        "network" => commands::network_command(config, command_args),
        "process" => commands::process_command(config, command_args),
        "config" => commands::config_command(config, command_args),
        "interactive" => commands::interactive_command(config, command_args),
        "help" => show_help(),
        "version" => show_version(),
        _ => {
            Irosu.kigbe("Unknown command: " + command);
            Irosu.fo("Use 'ifa-tool help' for available commands");
            Oyeku.exit(1);
        }
    }
}

// System health check
se system_health(config: Config, args: Vec) {
    ayanmo verbose = args.contains("--verbose") or args.contains("-v");
    
    if !config.quiet {
        Irosu.fo("=== System Health Check ===");
        Irosu.fo("");
    }
    
    ayanmo health = utils::check_system_health();
    ayanmo issues = Vec::::new();
    
    // Check each component
    fun (component, status) ninu health {
        ayanmo status_icon = match status {
            "ok" => "otito",
            "warning" => "warning",
            "critical" => "CRITICAL",
            _ => "unknown"
        };
        
        if verbose {
            Irosu.fo(component + ": " + status_icon);
        }
        
        if status == "critical" or status == "warning" {
            issues.push(component + " (" + status + ")");
        }
    }
    
    if !verbose {
        ayanmo overall = health.get("overall");
        ayanmo overall_icon = match overall {
            "ok" => "otito",
            "warning" => "warning", 
            "critical" => "CRITICAL",
            _ => "unknown"
        };
        
        Irosu.fo("Overall Status: " + overall_icon);
        
        if issues.length() > 0 {
            Irosu.fo("");
            Irosu.fo("Issues:");
            fun issue ninu issues {
                Irosu.fo("  - " + issue);
            }
        }
    }
    
    // Exit with appropriate code
    if health.get("overall") == "critical" {
        Oyeku.exit(2);
    } else if health.get("overall") == "warning" {
        Oyeku.exit(1);
    }
}

// File listing
se file_list(config: Config, args: Vec) {
    ayanmo path = if args.length() > 0 { args[0] } else { "." };
    ayanmo detailed = args.contains("--detailed") or args.contains("-l");
    ayanmo all = args.contains("--all") or args.contains("-a");
    ayanmo recursive = args.contains("--recursive") or args.contains("-r");
    
    if !Ogbe.dir_exists(path) {
        Irosu.kigbe("Directory not found: " + path);
        return;
    }
    
    list_directory(path, detailed, all, recursive, 0);
}

// Recursive directory listing
se list_directory(path: Str, detailed: Bool, all: Bool, recursive: Bool, depth: u32) {
    if depth > 0 and !recursive {
        return;
    }
    
    ayanmo entries = Ogbe.dir_list(path);
    
    // Sort entries
    entries.sort_by(|a, b| {
        if a.is_dir and !b.is_dir { -1 }
        else if !a.is_dir and b.is_dir { 1 }
        else { Ika.compare(a.name, b.name) }
    });
    
    fun entry ninu entries {
        ayanmo full_path = path + "/" + entry.name;
        
        if !all and entry.name.starts_with(".") {
            continue;
        }
        
        if detailed {
            ayanmo permissions = Ogbe.file_permissions(full_path);
            ayanmo size = if entry.is_dir { "0" } else { Obara.to_str(entry.size) };
            ayanmo modified = Iwori.format_date(entry.modified);
            
            Irosu.fo(permissions + " " + 
                     Ika.pad_right(size, 10) + " " +
                     Ika.pad_right(modified, 20) + " " +
                     (if entry.is_dir { Color::Blue + entry.name + "/" } else { entry.name }) +
                     Color::Reset);
        } else {
            Irosu.fo((if entry.is_dir { Color::Blue + entry.name + "/" } else { entry.name }) + Color::Reset);
        }
        
        if recursive and entry.is_dir and depth < 10 { // Limit depth to prevent infinite loops
            Irosu.fo("");
            list_directory(full_path, detailed, all, recursive, depth + 1);
        }
    }
}

se file_search(config: Config, args: Vec) {
    if args.length() < 2 {
        Irosu.kigbe("Usage: ifa-tool file search  ");
        return;
    }
    
    ayanmo pattern = args[0];
    ayanmo directory = args[1];
    ayanmo case_sensitive = args.contains("--case-sensitive");
    ayanmo recursive = args.contains("--recursive") or args.contains("-r");
    
    Irosu.fo("Searching for: " + pattern);
    Irosu.fo("Directory: " + directory);
    Irosu.fo("Case sensitive: " + Obara.to_str(case_sensitive));
    Irosu.fo("Recursive: " + Obara.to_str(recursive));
    Irosu.fo("");

    if results.length() == 0 {
        Irosu.fo("No matches found.");
    } else {
        Irosu.fo("Found " + Obara.to_str(results.length()) + " matches:");
        fun result ninu results {
            Irosu.fo("  " + result);
        }
    }
}

// Recursive file search
se search_files(directory: Str, pattern: Str, case_sensitive: Bool, recursive: Bool, depth: u32) -> Vec {
    ayanmo results = Vec::::new();
    
    if depth > 0 and !recursive {
        pada results;
    }
    
    gbiyanju {
        ayanmo entries = Ogbe.dir_list(directory);
        
        fun entry ninu entries {
            ayanmo full_path = directory + "/" + entry.name;
            
            // Check if name matches pattern
            ayanmo name_matches = if case_sensitive {
                Ika.contains(entry.name, pattern)
            } else {
                Ika.contains(Ika.to_lower(entry.name), Ika.to_lower(pattern))
            };
            
            if name_matches {
                results.push(full_path);
            }
            
            // Search in file contents if it's a text file
            if entry.is_file and utils::is_text_file(entry.name) {
                ayanmo content = Ogbe.file_read(full_path);
                ayanmo content_matches = if case_sensitive {
                    Ika.contains(content, pattern)
                } else {
                    Ika.contains(Ika.to_lower(content), Ika.to_lower(pattern))
                };
                
                if content_matches {
                    results.push(full_path + " (content match)");
                }
            }
            
            // Recursively search subdirectories
            if recursive and entry.is_dir and depth < 10 {
                ayanmo sub_results = search_files(full_path, pattern, case_sensitive, recursive, depth + 1);
                fun sub_result ninu sub_results {
                    results.push(sub_result);
                }
            }
        }
    } gba (error) {
        // Skip directories we can't access
        if config.verbose {
            Irosu.kigbe("Cannot access " + directory + ": " + error.message);
        }
    }
    
    pada results;
}

// Network ping
se network_ping(config: Config, args: Vec) {
    if args.length() == 0 {
        Irosu.kigbe("Usage: ifa-tool network ping ");
        return;
    }
    
    ayanmo host = args[0];
    ayanmo count = 4;
    ayanmo timeout = 1000;
    
    // Parse optional arguments
    fun i ninu 0..args.length() {
        if args[i] == "--count" and i + 1 < args.length() {
            count = Obara.parse_u32(args[i + 1]).unwrap_or(4);
        } else if args[i] == "--timeout" and i + 1 < args.length() {
            timeout = Obara.parse_u32(args[i + 1]).unwrap_or(1000);
        }
    }
    
    Irosu.fo("PING " + host + " (" + count + " packets)");
    Irosu.fo("");
    
    ayanmo total_time = 0;
    ayanmo received = 0;
    
    fun i ninu 1..count + 1 {
        ayanmo start_time = Iwori.timestamp();
        
        gbiyanju {
            ayanmo result = Otura.ping(host, timeout);
            ayanmo end_time = Iwori.timestamp();
            ayanmo rtt = (end_time - start_time) * 1000; // Convert to milliseconds
            
            if result.success {
                Irosu.fo("Reply from " + host + ": bytes=" + result.bytes + 
                         " time=" + Obara.to_str(rtt) + "ms TTL=" + Obara.to_str(result.ttl));
                total_time += rtt;
                received += 1;
            } else {
                Irosu.kigbe("Request timeout.");
            }
        } gba (error) {
            Irosu.kigbe("Ping failed: " + error.message);
        }
        
        if i < count {
            Iwori.sleep(1000); // Wait 1 second between pings
        }
    }
    
    Irosu.fo("");
    
    // Statistics
    ayanmo loss = ((count - received) * 100) / count;
    ayanmo avg_time = if received > 0 { total_time / received } else { 0 };

    Irosu.fo("    Packets: Sent = " + Obara.to_str(count) + 
             ", Received = " + Obara.to_str(received) + 
             ", Lost = " + Obara.to_str(count - received) + 
             " (" + Obara.to_str(loss) + "% loss)");
    
    if received > 0 {
        Irosu.fo("Approximate round trip times:");
        Irosu.fo("    Minimum = 0ms, Maximum = " + Obara.to_str(total_time) + "ms, Average = " + Obara.to_str(avg_time) + "ms");
    }
}

// Process listing
se process_list(config: Config, args: Vec) {
    ayanmo sort_by = "cpu";
    ayanmo limit = 20;
    
    // Parse arguments
    fun i ninu 0..args.length() {
        if args[i] == "--sort" and i + 1 < args.length() {
            sort_by = args[i + 1];
        } else if args[i] == "--limit" and i + 1 < args.length() {
            limit = Obara.parse_u32(args[i + 1]).unwrap_or(20);
        }
    }
    
    ayanmo processes = Ogbe.get_all_processes();
    
    // Sort processes
    match sort_by {
        "cpu" => processes.sort_by(|a, b| Obara.compare(b.cpu_percent, a.cpu_percent)),
        "memory" => processes.sort_by(|a, b| Obara.compare(b.memory_mb, a.memory_mb)),
        "name" => processes.sort_by(|a, b| Ika.compare(a.name, b.name)),
        "pid" => processes.sort_by(|a, b| Obara.compare(a.pid, b.pid)),
        _ => {}
    }
    
    // Limit results
    if processes.length() > limit {
        processes = processes.slice(0, limit);
    }
    
    // Display header
    Irosu.fo("PID     NAME                    CPU%    MEMORY");
    Irosu.fo("----    --------------------  -----   -------");
    
    // Display processes
    fun proc ninu processes {
        Irosu.fo(Ika.pad_left(Obara.to_str(proc.pid), 8) + "  " +
                 Ika.pad_right(proc.name, 20) + "  " +
                 Ika.pad_left(Obara.to_str(proc.cpu_percent), 5) + "%  " +
                 utils::format_bytes(proc.memory_mb * 1024 * 1024));
    }
}

// Set configuration value
se config_set(config: Config, args: Vec) {
    if args.length() < 2 {
        Irosu.kigbe("Usage: ifa-tool config set  ");
        return;
    }
    
    ayanmo key = args[0];
    ayanmo value = args[1];
    
    // Update configuration
    ayanmo mut updated_config = config;
    
    match key {
        "verbose" => {
            if value == "true" or value == "1" {
                updated_config.verbose = otito;
            } else if value == "false" or value == "0" {
                updated_config.verbose = eke;
            } else {
                Irosu.kigbe("Invalid value for verbose: " + value);
                return;
            }
        }
        
        "quiet" => {
            if value == "true" or value == "1" {
                updated_config.quiet = otito;
            } else if value == "false" or value == "0" {
                updated_config.quiet = eke;
            } else {
                Irosu.kigbe("Invalid value for quiet: " + value);
                return;
            }
        }
        
        "color_mode" => {
            if value == "auto" or value == "always" or value == "never" {
                updated_config.color_mode = value;
            } else {
                Irosu.kigbe("Invalid color_mode: " + value + " (must be auto, always, or never)");
                return;
            }
        }
        
        "default_shell" => {
            updated_config.default_shell = value;
        }
        
        "editor" => {
            updated_config.editor = value;
        }
        
        "pager" => {
            updated_config.pager = value;
        }
        
        "default_timeout" => {
            ayanmo timeout = Obara.parse_u32(value);
            if timeout.is_ok() {
                updated_config.default_timeout = timeout.unwrap();
            } else {
                Irosu.kigbe("Invalid timeout value: " + value);
                return;
            }
        }
        
        "checksum_algorithm" => {
            if value == "md5" or value == "sha1" or value == "sha256" or value == "sha512" {
                updated_config.checksum_algorithm = value;
            } else {
                Irosu.kigbe("Invalid checksum algorithm: " + value);
                return;
            }
        }
        
        _ => {
            // Custom setting
            updated_config.custom.insert(key, value);
        }
    }
    
    // Save updated configuration
    save_config(updated_config);
    
    Irosu.fo("Set " + key + " = " + value);
}

// Format bytes to human readable format
se format_bytes(bytes: u64) -> Str {
    ayanmo units = ["B", "KB", "MB", "GB", "TB", "PB"];
    ayanmo mut size = bytes as f64;
    ayanmo mut unit_index = 0;
    
    while size >= 1024.0 and unit_index < units.length() - 1 {
        size /= 1024.0;
        unit_index += 1;
    }
    
    if unit_index == 0 {
        pada Obara.to_str(bytes) + " " + units[unit_index];
    } else {
        pada Obara.format(size, 2) + " " + units[unit_index];
    }
}

// Validate file path
se validate_file_path(path: Str) -> Result<(), Str> {
    if path == "" {
        pada Err("Path cannot be empty");
    }
    
    // Check for invalid characters
    ayanmo invalid_chars = ["<", ">", ":", "\"", "|", "?", "*"];
    fun char ninu invalid_chars {
        if Ika.contains(path, char) {
            pada Err("Path contains invalid character: " + char);
        }
    }
    
    // Check path length (platform dependent)
    ayanmo max_length = if Ogbe.os_name().contains("Windows") { 260 } else { 4096 };
    if Ika.length(path) > max_length {
        pada Err("Path too long (max " + Obara.to_str(max_length) + " characters)");
    }
    
    // Check if parent directory exists
    ayanmo parent_dir = Ika.substring(path, 0, Ika.last_index_of(path, "/"));
    if parent_dir != "" and !Ogbe.dir_exists(parent_dir) {
        pada Err("Parent directory does not exist: " + parent_dir);
    }
    
    pada Ok(());
}

// Parse size string (e.g., "1GB", "500MB")
se parse_size(size_str: Str) -> Result {
    ayanmo units = Map::::new();
    units.insert("B", 1);
    units.insert("KB", 1024);
    units.insert("MB", 1024 * 1024);
    units.insert("GB", 1024 * 1024 * 1024);
    units.insert("TB", 1024 * 1024 * 1024 * 1024);
    
    // Find unit at end of string
    ayanmo unit = "";
    fun (unit_str, multiplier) ninu units {
        if Ika.ends_with(size_str, unit_str) {
            unit = unit_str;
            break;
        }
    }
    
    if unit == "" {
        pada Err("Invalid size format. Use units like B, KB, MB, GB, TB");
    }
    
    ayanmo number_part = Ika.substring(size_str, 0, Ika.length(size_str) - Ika.length(unit));
    ayanmo number = Obara.parse_f64(number_part);
    
    if number.is_err() {
        pada Err("Invalid number: " + number_part);
    }
    
    ayanmo bytes = (number.unwrap() * units.get(unit).unwrap() as f64) as u64;
    pada Ok(bytes);
}
