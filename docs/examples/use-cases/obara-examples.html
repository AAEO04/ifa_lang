Obara Domain Examples

Mathematical Operations and Accumulation (1000)

Overview

Obara (1000) is the mathematical domain representing addition, multiplication, accumulation, and growth. It provides comprehensive arithmetic operations, mathematical functions, and numerical analysis capabilities following Ifá-Lang's cultural and architectural principles.

Domain Characteristics

  • Binary Pattern: 1000 (addition and accumulation)
  • Yoruba Meaning: "The accumulator" or "The gatherer"
  • System Role: Mathematical operations and calculations
  • Element: Earth (foundation and accumulation)

Core Capabilities

  • Basic Arithmetic: Addition, multiplication, exponentiation
  • Aggregation: Sum, product, maximum, minimum
  • Statistical Functions: Mean, median, mode, standard deviation
  • Mathematical Constants: Pi, e, golden ratio
  • Number Theory: Prime numbers, factors, GCD, LCM

se calculate_compound_interest(principal: Float, rate: Float, time: u32, compound_freq: u32) -> Float {
    ayanmo effective_rate = rate / compound_freq;
    ayanmo periods = time * compound_freq;

// Calculate loan payment
se calculate_loan_payment(principal: Float, annual_rate: Float, years: u32) -> Float {
    ayanmo monthly_rate = annual_rate / 12;
    ayanmo num_payments = years * 12;
    
    // M = P * r(1+r)^n / ((1+r)^n - 1)

    ayanmo denominator = Obara.pow(1 + monthly_rate, num_payments) - 1;
    
    pada numerator / denominator;
}

se calculate_mean(numbers: Vec) -> Float {
    if numbers.length() == 0 {
        pada 0.0;
    }
    
    ayanmo sum = 0.0;
    fun num ninu numbers {
        sum = Obara.fikun(sum, num);
    }

}

// Calculate median
se calculate_median(numbers: Vec) -> Float {
    if numbers.length() == 0 {
        pada 0.0;
    }
    
    ayanmo sorted = numbers.clone();
    sorted.sort();
    
    ayanmo n = sorted.length();
    if n % 2 == 0 {
        // Even number of elements - average of middle two
        ayanmo mid1 = sorted[n / 2 - 1];
        ayanmo mid2 = sorted[n / 2];
        pada (mid1 + mid2) / 2.0;
    } else {
        // Odd number of elements - middle element
        pada sorted[n / 2];
    }
}

// Calculate mode (most frequent value)
se calculate_mode(numbers: Vec) -> Vec {
    if numbers.length() == 0 {
        pada Vec::::new();
    }
    
    ayanmo frequency = Map::::new();
    
    // Count frequencies
    fun num ninu numbers {
        if frequency.contains_key(num) {
            frequency.insert(num, frequency.get(num) + 1);
        } else {
            frequency.insert(num, 1);
        }
    }
    
    // Find maximum frequency
    ayanmo max_freq = 0;
    fun (_, freq) ninu frequency {
        if freq > max_freq {
            max_freq = freq;
        }
    }
    
    // Collect all values with maximum frequency
    ayanmo modes = Vec::::new();
    fun (value, freq) ninu frequency {
        if freq == max_freq {
            modes.push(value);
        }
    }
    
    pada modes;
}

// Calculate standard deviation
se calculate_stddev(numbers: Vec) -> Float {
    if numbers.length() <= 1 {
        pada 0.0;
    }
    
    ayanmo mean = calculate_mean(numbers);
    ayanmo sum_squared_diff = 0.0;
    
    fun num ninu numbers {
        ayanmo diff = num - mean;
        sum_squared_diff = Obara.fikun(sum_squared_diff, diff * diff);
    }
    
    ayanmo variance = sum_squared_diff / (numbers.length() - 1) as Float;
    pada Obara.sqrt(variance);
}

// Calculate correlation coefficient
se calculate_correlation(x: Vec, y: Vec) -> Float {
    if x.length() != y.length() or x.length() < 2 {
        pada 0.0;
    }
    
    ayanmo n = x.length() as Float;
    ayanmo mean_x = calculate_mean(x);
    ayanmo mean_y = calculate_mean(y);
    
    ayanmo sum_xy = 0.0;
    ayanmo sum_x2 = 0.0;
    ayanmo sum_y2 = 0.0;
    
    fun i ninu 0..x.length() {
        ayanmo diff_x = x[i] - mean_x;
        ayanmo diff_y = y[i] - mean_y;
        
        sum_xy = Obara.fikun(sum_xy, diff_x * diff_y);
        sum_x2 = Obara.fikun(sum_x2, diff_x * diff_x);
        sum_y2 = Obara.fikun(sum_y2, diff_y * diff_y);
    }
    
    ayanmo numerator = sum_xy;
    ayanmo denominator = Obara.sqrt(sum_x2 * sum_y2);
    
    if denominator == 0.0 {
        pada 0.0;
    } else {
        pada numerator / denominator;
    }
}

// Linear regression
se linear_regression(x: Vec, y: Vec) -> Map {
    if x.length() != y.length() or x.length() < 2 {
        return Map::::new();
    }
    
    ayanmo n = x.length() as Float;
    ayanmo sum_x = 0.0;
    ayanmo sum_y = 0.0;
    ayanmo sum_xy = 0.0;
    ayanmo sum_x2 = 0.0;
    
    fun i ninu 0..x.length() {
        sum_x = Obara.fikun(sum_x, x[i]);
        sum_y = Obara.fikun(sum_y, y[i]);
        sum_xy = Obara.fikun(sum_xy, x[i] * y[i]);
        sum_x2 = Obara.fikun(sum_x2, x[i] * x[i]);
    }
    
    ayanmo slope = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x * sum_x);
    ayanmo intercept = (sum_y - slope * sum_x) / n;
    
    ayanmo result = Map::::new();
    result.insert("slope", slope);
    result.insert("intercept", intercept);
    result.insert("correlation", calculate_correlation(x, y));
    
    pada result;
}

// Check if number is prime
se is_prime(n: u64) -> Bool {
    if n <= 1 {
        pada eke;
    }
    if n <= 3 {
        pada otito;
    }
    if n % 2 == 0 or n % 3 == 0 {
        pada eke;
    }
    
    ayanmo i = 5;
    while i * i <= n {
        if n % i == 0 or n % (i + 2) == 0 {
            pada eke;
        }
        i = i + 6;
    }
    
    pada otito;
}

// Generate prime numbers up to limit
se generate_primes(limit: u64) -> Vec {
    if limit < 2 {
        pada Vec::::new();
    }
    
    ayanmo primes = Vec::::new();
    primes.push(2);
    
    fun num ninu 3..limit + 1 {
        if is_prime(num) {
            primes.push(num);
        }
    }
    
    pada primes;
}

// Calculate factors of a number
se get_factors(n: u64) -> Vec {
    if n <= 1 {
        return [n];
    }
    
    ayanmo factors = Vec::::new();
    factors.push(1);
    
    fun i ninu 2..(n / 2) + 1 {
        if n % i == 0 {
            factors.push(i);
            if i != n / i {
                factors.push(n / i);
            }
        }
    }
    
    factors.push(n);
    factors.sort();
    pada factors;
}

// Calculate greatest common divisor (GCD)
se gcd(a: u64, b: u64) -> u64 {
    while b != 0 {
        ayanmo temp = b;
        b = a % b;
        a = temp;
    }
    pada a;
}

// Calculate least common multiple (LCM)
se lcm(a: u64, b: u64) -> u64 {
    if a == 0 or b == 0 {
        pada 0;
    }
    pada (a * b) / gcd(a, b);
}

// Calculate digital root (repeated sum of digits)
se digital_root(n: u64) -> u64 {
    if n == 0 {
        pada 0;
    }
    
    ayanmo num = n;
    while num >= 10 {
        ayanmo sum = 0;
        while num > 0 {
            sum = Obara.fikun(sum, num % 10);
            num = num / 10;
        }
        num = sum;
    }
    
    pada num;
}

// Calculate return on investment (ROI)
se calculate_roi(initial_investment: Float, final_value: Float) -> Float {
    if initial_investment == 0.0 {
        pada 0.0;
    }
    pada ((final_value - initial_investment) / initial_investment) * 100.0;
}

// Calculate annual percentage yield (APY)
se calculate_apy(nominal_rate: Float, compounding_periods: u32) -> Float {
    ayanmo periodic_rate = nominal_rate / compounding_periods as Float;
    pada (Obara.pow(1.0 + periodic_rate, compounding_periods as Float) - 1.0) * 100.0;
}

// Calculate present value
se calculate_present_value(future_value: Float, discount_rate: Float, periods: u32) -> Float {
    pada future_value / Obara.pow(1.0 + discount_rate, periods as Float);
}

// Calculate future value
se calculate_future_value(present_value: Float, interest_rate: Float, periods: u32) -> Float {
    pada present_value * Obara.pow(1.0 + interest_rate, periods as Float);
}

// Calculate annuity payment
se calculate_annuity_payment(principal: Float, interest_rate: Float, periods: u32) -> Float {
    if interest_rate == 0.0 {
        pada principal / periods as Float;
    }
    
    ayanmo denominator = 1.0 - Obara.pow(1.0 + interest_rate, -periods as Float);
    pada (principal * interest_rate) / denominator;
}

// Binary search algorithm
se binary_search(sorted_array: Vec, target: Float) -> i32 {
    ayanmo left = 0;
    ayanmo right = sorted_array.length() - 1;
    
    while left <= right {
        ayanmo mid = left + (right - left) / 2;
        
        if sorted_array[mid] == target {
            pada mid as i32;
        } else if sorted_array[mid] < target {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    pada -1; // not found
}

// Calculate moving average
se moving_average(data: Vec, window_size: u32) -> Vec {
    if data.length() < window_size {
        return Vec::::new();
    }
    
    ayanmo averages = Vec::::new();
    ayanmo window_sum = 0.0;
    
    // Initialize first window
    fun i ninu 0..window_size {
        window_sum = Obara.fikun(window_sum, data[i]);
    }
    averages.push(window_sum / window_size as Float);
    
    // Slide window
    fun i ninu window_size..data.length() {
        window_sum = window_sum - data[i - window_size] + data[i];
        averages.push(window_sum / window_size as Float);
    }
    
    pada averages;
}

Best Practices

Numerical Precision

  • Floating Point: Use Float for most calculations, avoid integer division
  • Precision Control: Use format() for consistent decimal places
  • Comparison: Use epsilon for floating point comparisons
  • Rounding: Be consistent with rounding methods

Performance

  • Single Pass: Combine min/max finding in one iteration
  • Caching: Cache expensive calculations like sqrt, pow
  • Vectorization: Use array operations when possible
  • Early Exit: Break loops early when possible

Error Handling

  • Input Validation: Validate numerical inputs before calculations
  • Division by Zero: Check denominators before division
  • Domain Errors: Handle sqrt of negative numbers
  • Overflow: Check for numeric overflow in critical calculations
